       ```ts
       /**
        * 260419-q2z Fix B — drain in-flight summarization promises so SIGTERM
        * does not truncate a registry update or a memory insert mid-flight.
        *
        * After calling drain(), new turn dispatches (streamFromAgent /
        * sendToAgent) reject with SessionError('shutting down, agent X is no
        * longer accepting turns'). This is the ONE surface that blocks new
        * work; stopAgent / reconcileRegistry continue to function so the
        * daemon can still clean up.
        *
        * @param timeoutMs - Hard ceiling on how long to wait for pending
        *                    summaries. On timeout, the promises are NOT
        *                    cancelled — drain just stops waiting. The
        *                    summaries continue in the background; any
        *                    SIGKILL that follows will kill them.
        */
       async drain(timeoutMs: number): Promise<{ readonly settled: number; readonly timedOut: number }> {
         this.draining = true;
         const count = this.pendingSummaries.size;
         if (count === 0) {
           return { settled: 0, timedOut: 0 };
         }
         const allSettled = Promise.allSettled([...this.pendingSummaries]);
         const timeout = new Promise<"timeout">((resolve) =>
           setTimeout(() => resolve("timeout"), timeoutMs).unref(),
         );
         const winner = await Promise.race([allSettled.then(() => "done" as const), timeout]);
         if (winner === "done") {
           return { settled: count, timedOut: 0 };
         }
         return { settled: 0, timedOut: this.pendingSummaries.size };
       }
       ```
