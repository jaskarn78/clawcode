// src/manager/filter-tools-by-capability.ts (NEW)

import type { McpServerState, CapabilityProbeStatus } from "./persistent-session-handle.js";

export interface ToolDef {
  readonly name: string;
  /** Server attribution. Required for filter — if missing, tool is treated as built-in (always passes). */
  readonly mcpServer?: string;
}

export interface FlapHistoryEntry {
  readonly windowStart: string;     // ISO8601 — start of the current 5min flap window
  readonly transitions: number;     // count of ready↔non-ready transitions in the window
  readonly stickyDegraded: boolean; // true if D-12 sticky window engaged
}

export const FLAP_WINDOW_MS = 5 * 60 * 1000;             // D-12: 5 minutes
export const FLAP_TRANSITION_THRESHOLD = 3;              // 3 flaps in window → sticky degraded

export interface FilterDeps {
  readonly snapshot: ReadonlyMap<string, McpServerState>;
  readonly flapHistory?: Map<string, FlapHistoryEntry>;  // mutable across calls — caller owns
  readonly now?: () => Date;
}

/**
 * Pure given (tools, deps): produces a filtered tool list where each tool's
 * mcpServer attribution is `capabilityProbe.status === 'ready'`. Built-in
 * tools (no mcpServer) always pass.
 *
 * Flap-stability (D-12): if a server has flapped ≥ FLAP_TRANSITION_THRESHOLD
 * times within FLAP_WINDOW_MS, it is treated as degraded for the rest of the
 * window even if currently ready. Prevents prompt-cache prefix-hash yo-yo.
 *
 * Returns Object.freeze'd readonly array.
 */
export function filterToolsByCapabilityProbe(
  tools: readonly ToolDef[],
  deps: FilterDeps,
): readonly ToolDef[];

/** Pure: returns true if server should be treated as ready (status==="ready" AND not in sticky-degraded window). */
export function isServerLlmAdvertisable(
  serverName: string,
  state: McpServerState | undefined,
  flapHistory: Map<string, FlapHistoryEntry> | undefined,
  now: Date,
): boolean;
```

    /**
     * Phase 94 Plan 02 — TOOL-03 dynamic tool advertising filter.
     * PURE module: no I/O, no SDK, no logger.
     */
    import type { McpServerState, CapabilityProbeStatus } from "./persistent-session-handle.js";

    export interface ToolDef {
      readonly name: string;
      readonly mcpServer?: string;
    }

    export interface FlapHistoryEntry {
      readonly windowStart: string;
      readonly transitions: number;
      readonly stickyDegraded: boolean;
    }

    export const FLAP_WINDOW_MS = 5 * 60 * 1000;
    export const FLAP_TRANSITION_THRESHOLD = 3;

    export interface FilterDeps {
      readonly snapshot: ReadonlyMap<string, McpServerState>;
      readonly flapHistory?: Map<string, FlapHistoryEntry>;
      readonly now?: () => Date;
    }

    /** Pure-ish: reads/writes flapHistory Map (caller-owned). */
    function updateFlapHistory(
      serverName: string,
      currentStatus: CapabilityProbeStatus | undefined,
      flapHistory: Map<string, FlapHistoryEntry>,
      now: Date,
    ): boolean {
      const prev = flapHistory.get(serverName);
      const isReady = currentStatus === "ready";
      const wasReady = prev ? !prev.stickyDegraded : true;

      if (!prev) {
        flapHistory.set(serverName, {
          windowStart: now.toISOString(),
          transitions: 0,
          stickyDegraded: false,
        });
        return false;
      }

      const windowMs = now.getTime() - new Date(prev.windowStart).getTime();
      if (windowMs > FLAP_WINDOW_MS) {
        // Window elapsed — reset
        flapHistory.set(serverName, {
          windowStart: now.toISOString(),
          transitions: 0,
          stickyDegraded: false,
        });
        return false;
      }

      const transitioned = isReady !== wasReady;
      const transitions = prev.transitions + (transitioned ? 1 : 0);
      const sticky = prev.stickyDegraded || transitions >= FLAP_TRANSITION_THRESHOLD;
      flapHistory.set(serverName, {
        windowStart: prev.windowStart,
        transitions,
        stickyDegraded: sticky,
      });
      return sticky;
    }

    export function isServerLlmAdvertisable(
      serverName: string,
      state: McpServerState | undefined,
      flapHistory: Map<string, FlapHistoryEntry> | undefined,
      now: Date,
    ): boolean {
      const status = state?.capabilityProbe?.status;
      if (status !== "ready") return false;
      if (!flapHistory) return true;
      const sticky = updateFlapHistory(serverName, status, flapHistory, now);
      return !sticky;
    }

    export function filterToolsByCapabilityProbe(
      tools: readonly ToolDef[],
      deps: FilterDeps,
    ): readonly ToolDef[] {
      const now = (deps.now ?? (() => new Date()))();
      const result: ToolDef[] = [];
      for (const tool of tools) {
        if (!tool.mcpServer) {
          result.push(tool);   // built-in — always passes
          continue;
        }
        const state = deps.snapshot.get(tool.mcpServer);
        if (isServerLlmAdvertisable(tool.mcpServer, state, deps.flapHistory, now)) {
          result.push(tool);
        }
      }
      return Object.freeze(result);
    }
    ```
