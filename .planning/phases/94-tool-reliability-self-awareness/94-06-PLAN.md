    ```ts
    export interface ResolvedDirective {
      readonly key: string;
      readonly text: string;
    }

    export function resolveSystemPromptDirectives(
      agentOverride: Record<string, { enabled?: boolean; text?: string }> | undefined,
      defaults: Record<string, SystemPromptDirective>,
    ): readonly ResolvedDirective[] {
      const allKeys = new Set<string>([...Object.keys(defaults), ...Object.keys(agentOverride ?? {})]);
      const merged: { key: string; enabled: boolean; text: string }[] = [];
      for (const key of allKeys) {
        const d = defaults[key];
        const o = agentOverride?.[key];
        const enabled = o?.enabled ?? d?.enabled ?? false;
        const text = o?.text ?? d?.text ?? "";
        if (enabled && text !== "") {
          merged.push({ key, enabled, text });
        }
      }
      // Deterministic order: sort by key alphabetically (cache stability)
      merged.sort((a, b) => a.key.localeCompare(b.key));
      return Object.freeze(merged.map(({ key, text }) => Object.freeze({ key, text })));
    }
    ```
