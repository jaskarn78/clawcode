<objective>
Implement D-06 honest ToolCallError schema: when a tool that PASSED capability probe still fails mid-turn (transient network, race, quota), wrap the failure into a structured discriminated-shape object the LLM receives in the tool-result slot. The LLM adapts naturally instead of silent-retrying or surfacing a raw exception. Includes the `findAlternativeAgents` pure helper used to populate the `alternatives` field (D-07 cross-agent suggestion data, consumed visually by 94-07).

Purpose: TOOL-07 = "no silent failures; the LLM sees what broke and what to do about it." The 5-value `errorClass` enum (`transient|auth|quota|permission|unknown`) gives the LLM enough signal to act differently per class without overwhelming detail. Classification rules are deterministic regex/status-code based — verifiable by tests, no LLM judgment in the wrapping.

Output: tool-call-error.ts new module + find-alternative-agents.ts pure helper + TurnDispatcher tool-execution-path extension that catches MCP rejections and routes through the wrapper. This plan is INDEPENDENT of 94-01 (Wave 1 parallel) — it doesn't read capabilityProbe; it writes structured error shapes when probe-passed tools fail at runtime.
</objective>

```ts
// src/manager/tool-call-error.ts (NEW)

export type ErrorClass =
  | "transient"    // network/socket/ECONNRESET/ETIMEDOUT/etc.
  | "auth"         // 401, invalid_key, unauthorized, expired
  | "quota"        // 429, rate limit, quota exceeded
  | "permission"   // 403, forbidden, insufficient permissions
  | "unknown";     // anything else — verbatim error preserved

export interface ToolCallError {
  readonly kind: "ToolCallError";          // discriminator for downstream consumers
  readonly tool: string;                    // e.g. "browser_snapshot"
  readonly errorClass: ErrorClass;
  readonly message: string;                 // verbatim from MCP — Phase 85 TOOL-04 pattern
  readonly suggestion?: string;             // operator-actionable hint (e.g., "auto-recovery scheduled" or "ask another agent")
  readonly alternatives?: readonly string[]; // healthy agents that could substitute (D-07)
}

/** Pure: classifies an error string/object into one of 5 ErrorClass values. */
export function classifyToolError(error: string | Error): ErrorClass;

export interface WrapToolErrorContext {
  readonly tool: string;
  readonly mcpServerName?: string;
  readonly findAlternatives?: () => readonly string[];   // injected from production via findAlternativeAgents
  readonly suggestionFor?: (errorClass: ErrorClass) => string | undefined;
}

/** Pure: wraps a runtime tool-call rejection into a structured ToolCallError. */
export function wrapMcpToolError(
  rawError: string | Error,
  context: WrapToolErrorContext,
): ToolCallError;
```

    **1. Implement `src/manager/tool-call-error.ts`** — pure module:

    ```ts
    /**
     * Phase 94 Plan 04 — D-06 ToolCallError schema.
     * Pure module: no I/O, no clock, no SDK imports.
     */

    export type ErrorClass = "transient" | "auth" | "quota" | "permission" | "unknown";

    export interface ToolCallError {
      readonly kind: "ToolCallError";
      readonly tool: string;
      readonly errorClass: ErrorClass;
      readonly message: string;
      readonly suggestion?: string;
      readonly alternatives?: readonly string[];
    }

    /** Order matters: auth/quota/permission checked before transient (more specific first). */
    const TRANSIENT_RE   = /(ECONNRESET|ECONNREFUSED|ETIMEDOUT|EAI_AGAIN|socket hang up|network|timeout)/i;
    const AUTH_RE        = /\b(401|unauthorized|invalid[_ ]?key|expired|authentication)\b/i;
    const QUOTA_RE       = /\b(429|rate[_ ]?limit|quota|too many requests)\b/i;
    const PERMISSION_RE  = /\b(403|forbidden|permission|insufficient)\b/i;

    export function classifyToolError(error: string | Error): ErrorClass {
      const msg = typeof error === "string" ? error : (error?.message ?? "");
      if (msg === "") return "unknown";
      if (AUTH_RE.test(msg))       return "auth";
      if (QUOTA_RE.test(msg))      return "quota";
      if (PERMISSION_RE.test(msg)) return "permission";
      if (TRANSIENT_RE.test(msg))  return "transient";
      return "unknown";
    }

    export interface WrapToolErrorContext {
      readonly tool: string;
      readonly mcpServerName?: string;
      readonly findAlternatives?: () => readonly string[];
      readonly suggestionFor?: (errorClass: ErrorClass) => string | undefined;
    }

    export function wrapMcpToolError(
      rawError: string | Error,
      context: WrapToolErrorContext,
    ): ToolCallError {
      const message = typeof rawError === "string" ? rawError : (rawError?.message ?? String(rawError));
      const errorClass = classifyToolError(rawError);
      const alternatives = context.findAlternatives?.();
      const suggestion = context.suggestionFor?.(errorClass);
      const obj: ToolCallError = Object.freeze({
        kind: "ToolCallError" as const,
        tool: context.tool,
        errorClass,
        message,
        ...(suggestion !== undefined ? { suggestion } : {}),
        ...(alternatives !== undefined && alternatives.length > 0 ? { alternatives: Object.freeze([...alternatives]) } : {}),
      });
      return obj;
    }
    ```

    ```ts
    import type { McpServerState } from "./persistent-session-handle.js";

    export interface McpStateProvider {
      readonly listAgents: () => readonly string[];
      readonly getStateFor: (agent: string) => ReadonlyMap<string, McpServerState>;
      readonly toolToServer?: (toolName: string) => string | undefined;
    }

    /** Heuristic default: tool names typically prefixed with `mcp__<server>__` or `<server>_`. */
    function defaultToolToServer(toolName: string): string | undefined {
      const m = toolName.match(/^mcp__([^_]+)__/);
      if (m) return m[1];
      const m2 = toolName.match(/^([a-z0-9-]+)_/i);
      return m2 ? m2[1] : undefined;
    }

    export function findAlternativeAgents(
      toolName: string,
      provider: McpStateProvider,
    ): readonly string[] {
      const toolToServer = provider.toolToServer ?? defaultToolToServer;
      const server = toolToServer(toolName);
      if (!server) return Object.freeze([]);
      const matches: string[] = [];
      for (const agent of provider.listAgents()) {
        const state = provider.getStateFor(agent);
        const serverState = state.get(server);
        if (serverState?.capabilityProbe?.status === "ready") {
          matches.push(agent);
        }
      }
      matches.sort();
      return Object.freeze(matches);
    }
    ```
