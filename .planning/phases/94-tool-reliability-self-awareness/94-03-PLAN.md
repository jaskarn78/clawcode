<objective>
Implement TOOL-04 + TOOL-05 + TOOL-06 (D-05): three default auto-recovery primitives wired into the heartbeat tick. When a capability probe (94-01) classifies a server as degraded, the recovery registry matches the error against handler regexes and runs the first match. Three handlers ship by default: Playwright Chromium-missing → `npx playwright install chromium --with-deps`; op:// auth-error → `op read` re-resolution + env swap; >5min degraded with no specific match → subprocess restart (last-resort). Bounded budget: 3 attempts/server/hour; 3rd failure alerts admin-clawdy. Discriminated-union outcome (`recovered | retry-later | give-up | not-applicable`) enforces compile-time exhaustiveness.

Purpose: TOOL-04/05/06 close the recovery feedback loop. 94-01 OBSERVES; 94-02 FILTERS; 94-03 RECOVERS. Without recovery, a degraded Playwright stays degraded forever and the LLM never sees `browser_snapshot` again. With recovery, the most common failure mode (Chromium binary missing after `playwright` SDK update) auto-heals within ~120s of detection. The pattern is extensible — RecoveryHandler interface lets future failure modes plug in without core changes.

Output: src/manager/recovery/{types,registry,playwright-chromium,op-refresh,subprocess-restart}.ts + TDD per handler + heartbeat integration. Bounded retries + admin-clawdy alert pinned by tests.

```ts
// src/manager/recovery/types.ts (NEW)

import type { Logger } from "pino";
import type { McpServerState } from "../persistent-session-handle.js";

/**
 * D-05 recovery outcome. Discriminated union — Plans consuming this MUST
 * exhaustive-switch + assertNever (compile-time enforcement).
 */
export type RecoveryOutcome =
  | { readonly kind: "recovered";        readonly serverName: string; readonly handlerName: string; readonly durationMs: number; readonly note?: string }
  | { readonly kind: "retry-later";      readonly serverName: string; readonly handlerName: string; readonly retryAfterMs: number; readonly reason: string }
  | { readonly kind: "give-up";          readonly serverName: string; readonly handlerName: string; readonly reason: string }
  | { readonly kind: "not-applicable";   readonly serverName: string };

/** Production wires execFile from node:child_process; tests stub. */
export interface RecoveryDeps {
  readonly execFile: (
    cmd: string,
    args: readonly string[],
    options?: { cwd?: string; env?: Record<string, string>; timeoutMs?: number },
  ) => Promise<{ stdout: string; stderr: string; exitCode: number }>;
  readonly killSubprocess: (serverName: string) => Promise<void>;
  readonly adminAlert: (text: string) => Promise<void>;
  readonly opRead: (reference: string) => Promise<string>;       // for op-refresh handler
  readonly readEnvForServer: (serverName: string) => Record<string, string>;
  readonly writeEnvForServer: (serverName: string, env: Record<string, string>) => Promise<void>;
  readonly now?: () => Date;
  readonly log: Logger;
}

export interface RecoveryHandler {
  readonly name: string;                            // e.g. "playwright-chromium"
  readonly priority: number;                         // lower = checked first; subprocess-restart highest priority number (last)
  /** Match the verbatim error string from capabilityProbe.error. */
  matches(error: string, state: McpServerState): boolean;
  recover(serverName: string, deps: RecoveryDeps): Promise<RecoveryOutcome>;
}

/** Bounded budget: 3 attempts per server per hour. */
export const MAX_ATTEMPTS_PER_HOUR = 3;
export const ATTEMPT_WINDOW_MS = 60 * 60 * 1000;

export interface AttemptRecord {
  readonly serverName: string;
  readonly attemptedAt: string;     // ISO8601
  readonly handlerName: string;
  readonly outcomeKind: RecoveryOutcome["kind"];
}

export type AttemptHistory = ReadonlyMap<string, readonly AttemptRecord[]>;
```

```ts
// src/manager/recovery/playwright-chromium.ts (NEW)

export const playwrightChromiumHandler: RecoveryHandler = {
  name: "playwright-chromium",
  priority: 10,
  matches(error: string): boolean {
    return /Executable doesn't exist at.*ms-playwright/i.test(error);
  },
  async recover(serverName, deps): Promise<RecoveryOutcome> {
    const start = (deps.now ?? (() => new Date()))();
    try {
      const result = await deps.execFile(
        "npx",
        ["playwright", "install", "chromium", "--with-deps"],
        { timeoutMs: 120_000 },
      );
      if (result.exitCode === 0) {
        return { kind: "recovered", serverName, handlerName: "playwright-chromium", durationMs: Date.now() - start.getTime(), note: "chromium installed" };
      }
      return { kind: "give-up", serverName, handlerName: "playwright-chromium", reason: `exit ${result.exitCode}: ${result.stderr.slice(0, 500)}` };
    } catch (err) {
      return { kind: "retry-later", serverName, handlerName: "playwright-chromium", retryAfterMs: 5 * 60_000, reason: err instanceof Error ? err.message : String(err) };
    }
  },
};
```

```ts
// src/manager/recovery/subprocess-restart.ts (NEW)

const FIVE_MIN_MS = 5 * 60_000;

export const subprocessRestartHandler: RecoveryHandler = {
  name: "subprocess-restart",
  priority: 100,
  matches(error: string, state: McpServerState): boolean {
    // Last-resort: only matches if degraded > 5min AND no specific handler has been wired
    const probe = state.capabilityProbe;
    if (!probe || probe.status !== "degraded") return false;
    const lastSuccess = probe.lastSuccessAt ? new Date(probe.lastSuccessAt).getTime() : 0;
    const now = Date.now();
    return (now - lastSuccess) > FIVE_MIN_MS;
  },
  async recover(serverName, deps): Promise<RecoveryOutcome> { /* killSubprocess; SDK respawns */ },
};
```

    ```ts
      async recover(serverName, deps): Promise<RecoveryOutcome> {
        const start = Date.now();
        try {
          const env = deps.readEnvForServer(serverName);
          const newEnv: Record<string, string> = {};
          for (const [key, value] of Object.entries(env)) {
            if (typeof value === "string" && value.startsWith("op://")) {
              const refs = value.match(OP_REF_RE) ?? [];
              if (refs.length === 0) {
                newEnv[key] = value;
                continue;
              }
              let resolved = value;
              for (const ref of refs) {
                const fresh = await deps.opRead(ref);
                resolved = resolved.replace(ref, fresh);
              }
              newEnv[key] = resolved;
            } else {
              newEnv[key] = value;
            }
          }
          await deps.writeEnvForServer(serverName, newEnv);
          // SDK respawns subprocess on env update OR caller signals subsequent kill via subprocess-restart
          return {
            kind: "recovered",
            serverName,
            handlerName: "op-refresh",
            durationMs: Date.now() - start,
            note: "op:// references re-resolved",
          };
        } catch (err) {
          return {
            kind: "give-up",
            serverName,
            handlerName: "op-refresh",
            reason: err instanceof Error ? err.message : String(err),
          };
        }
      },
    };
    ```

    ```ts
    import type { RecoveryHandler, RecoveryOutcome, RecoveryDeps } from "./types.js";
    import type { McpServerState } from "../persistent-session-handle.js";

    const FIVE_MIN_MS = 5 * 60_000;

    export const subprocessRestartHandler: RecoveryHandler = {
      name: "subprocess-restart",
      priority: 100,
      matches(error: string, state: McpServerState): boolean {
        const probe = state.capabilityProbe;
        if (!probe || probe.status !== "degraded") return false;
        const lastSuccess = probe.lastSuccessAt ? new Date(probe.lastSuccessAt).getTime() : 0;
        return Date.now() - lastSuccess > FIVE_MIN_MS;
      },
      async recover(serverName, deps): Promise<RecoveryOutcome> {
        const start = Date.now();
        try {

          return {
            kind: "recovered",
            serverName,
            handlerName: "subprocess-restart",
            durationMs: Date.now() - start,
            note: "subprocess killed; SDK will respawn",
          };
        } catch (err) {
          return {
            kind: "give-up",
            serverName,
            handlerName: "subprocess-restart",
            reason: err instanceof Error ? err.message : String(err),
          };
        }
      },
    };
    ```

    ```ts
    import type { McpServerState } from "../persistent-session-handle.js";
    import {
      type RecoveryHandler,
      type RecoveryOutcome,
      type RecoveryDeps,
      type AttemptRecord,
      MAX_ATTEMPTS_PER_HOUR,
      ATTEMPT_WINDOW_MS,
    } from "./types.js";
    import { playwrightChromiumHandler } from "./playwright-chromium.js";
    import { opRefreshHandler } from "./op-refresh.js";
    import { subprocessRestartHandler } from "./subprocess-restart.js";

    export const RECOVERY_REGISTRY: readonly RecoveryHandler[] = Object.freeze(
      [playwrightChromiumHandler, opRefreshHandler, subprocessRestartHandler]
        .sort((a, b) => a.priority - b.priority),
    );

    function pruneAndCount(history: AttemptRecord[], now: Date): readonly AttemptRecord[] {
      const cutoff = now.getTime() - ATTEMPT_WINDOW_MS;
      return history.filter((r) => new Date(r.attemptedAt).getTime() >= cutoff);
    }

    export async function runRecoveryForServer(
      serverName: string,
      state: McpServerState,
      history: Map<string, AttemptRecord[]>,
      deps: RecoveryDeps,
    ): Promise<RecoveryOutcome> {
      const now = (deps.now ?? (() => new Date()))();
      const error = state.capabilityProbe?.error ?? "";
      const recent = pruneAndCount(history.get(serverName) ?? [], now);
      history.set(serverName, [...recent]);   // prune in place

      if (recent.length >= MAX_ATTEMPTS_PER_HOUR) {
        const giveUp: RecoveryOutcome = {
          kind: "give-up",
          serverName,
          handlerName: "registry",
          reason: `budget exhausted: ${recent.length}/${MAX_ATTEMPTS_PER_HOUR} attempts in last ${ATTEMPT_WINDOW_MS / 60_000}min`,
        };
        history.set(serverName, [...recent, { serverName, attemptedAt: now.toISOString(), handlerName: "registry", outcomeKind: giveUp.kind }]);
        return giveUp;
      }

      // Find first matching handler (priority order)
      const handler = RECOVERY_REGISTRY.find((h) => h.matches(error, state));
      if (!handler) {
        return { kind: "not-applicable", serverName };
      }

      const outcome = await handler.recover(serverName, deps);
      history.set(serverName, [...recent, { serverName, attemptedAt: now.toISOString(), handlerName: handler.name, outcomeKind: outcome.kind }]);

      // 3rd failure → admin-clawdy alert
      const updatedHistory = history.get(serverName) ?? [];
      const recentFailures = updatedHistory.filter((r) => r.outcomeKind === "give-up" || r.outcomeKind === "retry-later");
      if (recentFailures.length >= MAX_ATTEMPTS_PER_HOUR) {
        const alertText = `[ClawCode recovery] server ${serverName} has failed recovery ${recentFailures.length} times in last hour. Last error: ${error.slice(0, 500)}`;
        await deps.adminAlert(alertText).catch((e) => deps.log.warn({ e }, "adminAlert failed"));
      }

      return outcome;
    }
    ```
