    import type { ResolvedAgentConfig } from "../shared/types.js";

    /**
     * Phase 999.57 — single source of truth for subagent inheritance overrides.
     *
     * The subagent's config is built by spreading the source agent's config
     * first (so it inherits skills, mcpServers, identity, etc.) and then
     * applying these overrides spread-LAST so they win. Without these
     * overrides, the subagent inherits source-agent lifecycle config that
     * surfaces unrelated content during the subagent's short-lived task
     * execution. See CONTEXT.md + RESEARCH.md for the failure modes.
     *
     * NOTE per RESEARCH.md Findings 1-3:
     * - `heartbeat.enabled: false` only fires after the spawner calls
     *   `heartbeatRunner.setAgentConfigs([subagentConfig])` (the runner's
     *   local agentConfigs Map is the source-of-truth for the gate, not
     *   the session-stored config).
     * - `memoryScannerEnabled: false` is a defensive marker — the scanner
     *   registration runs once at daemon boot and never iterates subagent
     *   configs. Kept for symmetry + future-proofing (Phase 100 GSD-07 /
     *   Phase 115 sub-scope 2 NON_RELOADABLE_FIELDS precedent).
     * - The inbox-leak vector (BACKLOG.md, CONTEXT D-05) is transitively
     *   closed by the heartbeat-runner registration above. There is no
     *   `inboxEnabled` field on ResolvedAgentConfig.
     */
    export function buildSubagentOverrides(
      parentConfig: ResolvedAgentConfig,
      sourceConfig: ResolvedAgentConfig,
      isDelegated: boolean,
    ): Partial<ResolvedAgentConfig> {
      return {
        heartbeat: {
          ...sourceConfig.heartbeat,
          enabled: false,
        },
        memoryAutoLoad: false,
        memoryRetrievalTopK: 0,
        memoryRetrievalTokenBudget: 0,
        memoryScannerEnabled: false,
        memoryFlushIntervalMs: 900_000, // schema default; no-op for short-lived subagents

          ...sourceConfig.memory,
          conversation: sourceConfig.memory.conversation
            ? { ...sourceConfig.memory.conversation, flushIntervalMinutes: 0 }
            : undefined,
        },
      };
    }
