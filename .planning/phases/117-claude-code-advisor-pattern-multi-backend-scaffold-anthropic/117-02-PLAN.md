       ```typescript
       import type { AdvisorBackend } from "./backends/types.js";

       export function resolveBackend(
         agentName: string,
         agentConfig: { advisor?: { backend?: BackendId } } | undefined,
         defaults: { advisor?: { backend?: BackendId } } | undefined,
       ): BackendId {
         const requested =
           agentConfig?.advisor?.backend ?? defaults?.advisor?.backend ?? "native";
         if (requested === "portable-fork") {
           // Defensive: schema rejects this in 117-06, but the registry must not select it.
           return "native";
         }
         return requested;
       }

       export class BackendRegistry {
         private readonly backends = new Map<BackendId, AdvisorBackend>();
         register(b: AdvisorBackend): void { this.backends.set(b.id, b); }
         get(id: BackendId): AdvisorBackend {
           const b = this.backends.get(id);
           if (!b) throw new Error(`AdvisorBackend "${id}" not registered`);
           return b;
         }
       }
       ```

  <action>
    1. Create `src/advisor/service.ts`.

    3. Implement:
       ```typescript
       import { ADVISOR_RESPONSE_MAX_LENGTH, AdvisorBudget } from "../usage/advisor-budget.js";
       import type { AdvisorService, AdvisorRequest, AdvisorResponse, BackendId } from "./types.js";
       import type { AdvisorBackend } from "./backends/types.js";

       export interface AdvisorServiceDeps {
         readonly budget: AdvisorBudget;
         readonly resolveBackend: (agent: string) => { backend: AdvisorBackend; id: BackendId };
         readonly resolveSystemPrompt: (agent: string) => string;
         readonly resolveAdvisorModel: (agent: string) => string;
       }

       export class DefaultAdvisorService implements AdvisorService {
         constructor(private readonly deps: AdvisorServiceDeps) {}

         async ask(req: AdvisorRequest): Promise<AdvisorResponse> {
           const { budget } = this.deps;
           if (!budget.canCall(req.agent)) {
             return {
               answer: "Advisor daily budget exhausted for this agent (cap resets at midnight UTC).",
               budgetRemaining: 0,
               backend: this.deps.resolveBackend(req.agent).id,
             };
           }
           const { backend, id } = this.deps.resolveBackend(req.agent);
           const systemPrompt = this.deps.resolveSystemPrompt(req.agent);
           const advisorModel = this.deps.resolveAdvisorModel(req.agent);
           const { answer } = await backend.consult({
             agent: req.agent,
             question: req.question,
             systemPrompt,
             advisorModel,
           });
           const truncated = answer.length > ADVISOR_RESPONSE_MAX_LENGTH
             ? answer.slice(0, ADVISOR_RESPONSE_MAX_LENGTH)
             : answer;
           budget.recordCall(req.agent);
           return { answer: truncated, budgetRemaining: budget.getRemaining(req.agent), backend: id };
         }
       }
       ```
    4. If `AdvisorBudget` does not currently expose `getRemaining`, add it as a small read-only helper in this task (single-line wrapper around the existing `calls_used` query). Verify the file's signature surface first.
    5. Doc-comment per RESEARCH §6 Pitfall 4: native `max_uses` is per-request, `AdvisorBudget` is per-agent-per-day — both layers apply.
  </action>
  <verify>`npm run build` clean; service test (117-02-T08) green.</verify>
</task>
