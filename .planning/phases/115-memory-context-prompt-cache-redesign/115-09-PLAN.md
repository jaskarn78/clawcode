    import { nanoid } from "nanoid";
    import { appendConsolidationRun } from "./consolidation-run-log.js";
    import type { ConsolidationBatch, BatchStatus } from "./cross-agent-coordinator.types.js";
    import type { MemoryStore } from "../memory/store.js";

    export interface Deps {
      getStoreForAgent(agent: string): MemoryStore;
      log: { warn: Function; info: Function; error: Function };
    }

    export class CrossAgentCoordinator {
      constructor(private deps: Deps) {}

      async runBatch(batch: ConsolidationBatch): Promise<BatchStatus> {
        const runId = batch.runId ?? nanoid();
        const startedAt = batch.startedAt ?? new Date().toISOString();
        await appendConsolidationRun({
          run_id: runId,
          target_agents: batch.targetAgents,
          memories_added: 0,
          status: "started",
          errors: [],
          started_at: startedAt,
        });

        const succeeded: string[] = [];
        const failed: Array<{ agent: string; error: string }> = [];
        let totalAdded = 0;

        for (const write of batch.writes) {
          try {
            const store = this.deps.getStoreForAgent(write.agent);
            // Per-agent atomic batch — all-or-nothing for this agent's writes.
            const added = await this.batchInsertAtomic(store, write.memories, runId);
            succeeded.push(write.agent);
            totalAdded += added;
          } catch (err: unknown) {
            const errMsg = (err as Error).message ?? String(err);
            failed.push({ agent: write.agent, error: errMsg });
            this.deps.log.error({ runId, agent: write.agent, errMsg, action: "cross-agent-batch-failed" }, "[diag] cross-agent-batch-failed");
          }
        }

        if (failed.length === 0) {
          await appendConsolidationRun({
            run_id: runId,
            target_agents: batch.targetAgents,
            memories_added: totalAdded,
            status: "completed",
            errors: [],
            started_at: startedAt,
            completed_at: new Date().toISOString(),
          });
          return { kind: "completed", runId, perAgent: Object.fromEntries(batch.writes.map(w => [w.agent, { added: w.memories.length }])) };
        }

        // Partial failure — record + return the partial state.
        // Rollback policy: roll back ONLY succeeded agents whose writes corresponded to a multi-agent invariant.
        // For now, mark partial-failed; operator can run `clawcode consolidation rollback <runId>` manually.
        await appendConsolidationRun({
          run_id: runId,
          target_agents: batch.targetAgents,
          memories_added: totalAdded,
          status: "failed",
          errors: failed.map(f => `${f.agent}: ${f.error}`),
          started_at: startedAt,
          completed_at: new Date().toISOString(),
        });
        return { kind: "partial-failed", runId, succeeded, failed };
      }

    /**
     * Phase 115 sub-scope 6-B — direct-SDK fast-path for short Discord acks.
     *
     * Mirrors callHaikuDirect (Phase 105) — uses raw @anthropic-ai/sdk with OAuth bearer
     * from ~/.claude/.credentials.json — bypassing the Claude Code CLI subprocess.
     *
     * Recovers 1h cache TTL the CLI no longer offers (post-2026-03-06 regression).
     *
     * Routing rule: only short Discord-paced turns with <500-token user msg AND no tool_use needed.
     * Anything that needs MCP tools, file access, or skill discovery falls through to the SDK.
     */

    import Anthropic from "@anthropic-ai/sdk";
    import { promises as fs } from "fs";
    import { homedir } from "os";
    import { join } from "path";

    interface FastPathOpts {
      stablePrefix: string;
      userMessage: string;
      mutableSuffix: string;
      maxTokens: number;
      model: string;
      log: Logger;
    }

    interface FastPathResult {
      assistantText: string;
      cacheReadInputTokens: number;
      cacheCreationInputTokens: number;
      inputTokens: number;
      outputTokens: number;
      ttlBucket: "1h" | "5m";
    }

    export async function dispatchFastPath(opts: FastPathOpts): Promise<FastPathResult> {
      const oauthPath = join(homedir(), ".claude", ".credentials.json");
      const creds = JSON.parse(await fs.readFile(oauthPath, "utf8"));
      const bearer = creds.access_token ?? creds.token;
      if (!bearer) throw new Error("OAuth bearer not found in credentials.json");

      const client = new Anthropic({
        apiKey: "",  // bypassed; using bearer
        defaultHeaders: { Authorization: `Bearer ${bearer}` },
      });

      // Build the system block with explicit 1h cache_control.
      const system = [
        {
          type: "text" as const,
          text: opts.stablePrefix,
          cache_control: { type: "ephemeral" as const, ttl: "1h" as const },
        },
      ];

      const messages = [
        {
          role: "user" as const,
          content: opts.mutableSuffix + "\n\n" + opts.userMessage,
        },
      ];

      const response = await client.messages.create({
        model: opts.model,
        max_tokens: opts.maxTokens,
        system,
        messages,
      });

      const assistantText = response.content.find(b => b.type === "text")?.text ?? "";
      return {
        assistantText,
        cacheReadInputTokens: response.usage?.cache_read_input_tokens ?? 0,
        cacheCreationInputTokens: response.usage?.cache_creation_input_tokens ?? 0,
        inputTokens: response.usage?.input_tokens ?? 0,
        outputTokens: response.usage?.output_tokens ?? 0,
        ttlBucket: "1h",
      };
    }

    /**
     * Routing decision — should this turn use the fast-path?
     */
    export function shouldUseFastPath(opts: {
      userMessage: string;
      historicalToolUseRate: number;
      configEnabled: boolean;
    }): boolean {
      if (!opts.configEnabled) return false;

      // <500-token user message (rough estimate via char count).
      if (opts.userMessage.length > 2000) return false;

      // Skip when historical tool_use rate is high — the cache write is wasted.
      if (opts.historicalToolUseRate >= 0.30) return false;

      // Heuristic: refuse if message looks tool-flavored.
      const toolFlavoredKeywords = /\b(grep|ls|cat|search|find|read|write|edit|run|execute|sql|query|api|fetch)\b/i;
      if (toolFlavoredKeywords.test(opts.userMessage)) return false;

      return true;
    }
