    ```ts
    function redactSecrets<T>(value: T): T {
      const seen = new WeakSet<object>();
      const SECRET_KEYS = /^(ANTHROPIC_API_KEY|OPENAI_API_KEY|DISCORD_TOKEN|DISCORD_BOT_TOKEN|GITHUB_TOKEN|.*_TOKEN|.*_KEY|.*_SECRET|password|credentials)$/i;
      const SECRET_VALUES = [
        "sk-ant-",            // Anthropic key prefix
        "sk-",                // OpenAI / generic key prefix
        "ghp_", "ghs_",       // GitHub
        "Bearer ",            // OAuth bearer literal
      ];

      function recurse(node: unknown): unknown {
        if (node === null || node === undefined) return node;
        if (typeof node === "string") {
          for (const prefix of SECRET_VALUES) {
            if (node.startsWith(prefix)) return "<REDACTED>";
          }
          return node;
        }
        if (typeof node !== "object") return node;
        if (seen.has(node as object)) return "<CIRCULAR>";
        seen.add(node as object);
        if (Array.isArray(node)) return node.map(recurse);
        const out: Record<string, unknown> = {};
        for (const [k, v] of Object.entries(node)) {
          if (SECRET_KEYS.test(k)) {
            out[k] = "<REDACTED>";
          } else {
            out[k] = recurse(v);
          }
        }
        return out;
      }

      return recurse(value) as T;
    }
    ```

| Threat | Vector | Severity | Mitigation |
|---|---|---|---|
| Diagnostic dump leaks `ANTHROPIC_API_KEY` to disk | `baseOptions.env.ANTHROPIC_API_KEY` serialized to JSON in `~/.clawcode/agents/<agent>/diagnostics/` | HIGH | `redactSecrets` helper redacts known secret keys (regex match) AND known secret value prefixes (`sk-ant-`, `sk-`, `ghp_`, `Bearer `); test suite asserts redaction |

| Diagnostic dump leaks Discord bot token | Discord token may appear in env or extra args | HIGH | Regex match on `*_TOKEN` key names; explicit test case |
| Consolidation run-log writes secrets via error string | Error from a failing consolidation might contain DB row content | MEDIUM | Run-log is operator-readable but local-only; error strings limited to `errMsg.slice(0, 200)`; no full DB row dumps |
| `[diag] likely-prompt-bloat` log line floods at 100% misclassification on legitimate errors | Many invalid_request errors are NOT prompt-bloat (e.g., bad tool args) | LOW | Threshold at 20K chars suppresses most false positives; classifier emits LOG only, not user-facing alert |
