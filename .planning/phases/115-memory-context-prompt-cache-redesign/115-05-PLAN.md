# Plan 115-05: Lazy-load memory tools + Phase 95 dreaming as Tier 1 consolidation engine

## Goal

Land the agent-facing surface that converts the model from "always-injected" to "tool-mediated lazy recall," PLUS rewire Phase 95 dreaming as the active Tier 1 consolidation engine using the locked D-10 three-tier hybrid policy.

The four MCP tools (`clawcode_memory_search`, `clawcode_memory_recall`, `clawcode_memory_edit`, `clawcode_memory_archive`) eliminate the "Apr 26 was last solid memory" failure mode — old context is always one tool-call away.     import { z } from "zod";
    ```ts
    import type { MemoryStore } from "../store.js";
    import type { EmbeddingService } from "../embedder.js";
    import { retrieveMemoryChunks } from "../memory-retrieval.js";

    export const SEARCH_INPUT_SCHEMA = z.object({
      query: z.string().min(1).max(2000),
      k: z.number().int().min(1).max(50).optional().default(10),
      includeTags: z.array(z.string()).optional(),
      excludeTags: z.array(z.string()).optional(),
    });

    export interface SearchHit {
      memoryId: string;
      heading?: string;
      snippet: string;
      score: number;
      tags: string[];
      source: "memories" | "memory_chunks";
      path?: string;
    }

    export interface SearchDeps {
      store: MemoryStore;
      embedder: EmbeddingService;
      agentName: string;  // from auth/session — NEVER from tool arg
      log?: { warn: Function; debug: Function };
    }

    export async function clawcodeMemorySearch(
      input: z.infer<typeof SEARCH_INPUT_SCHEMA>,
      deps: SearchDeps,
    ): Promise<{ hits: SearchHit[]; agentName: string }> {
      const parsed = SEARCH_INPUT_SCHEMA.parse(input);
      const chunks = await retrieveMemoryChunks({
        query: parsed.query,
        store: deps.store,
        embed: (q) => deps.embedder.embed(q),
        topK: parsed.k,
        timeWindowDays: 14,
        tokenBudget: 4000,
        excludeTags: parsed.excludeTags,
      });

      // Snippet to 500 chars per hit (Pain Point #2 lessons).
      const hits: SearchHit[] = chunks.map((c) => ({
        memoryId: c.memoryId ?? c.chunkId ?? "",
        heading: c.heading,
        snippet: (c.body ?? "").slice(0, 500),
        score: c.score ?? 0,
        tags: c.tags ?? [],
        source: c.source as "memories" | "memory_chunks",
        path: c.path,
      })).filter((h) => h.memoryId.length > 0);

      // Optional includeTags filter (post-retrieval, not pushed into RRF).
      const filtered = parsed.includeTags?.length
        ? hits.filter((h) => h.tags.some((t) => parsed.includeTags!.includes(t)))
        : hits;

      return { hits: filtered, agentName: deps.agentName };
    }
    ```

    ```ts
    import { z } from "zod";
    import type { MemoryStore } from "../store.js";

    export const RECALL_INPUT_SCHEMA = z.object({
      memoryId: z.string().min(1).max(200),
    });

    export interface RecallDeps {
      store: MemoryStore;
      agentName: string;
    }

    export async function clawcodeMemoryRecall(
      input: z.infer<typeof RECALL_INPUT_SCHEMA>,
      deps: RecallDeps,
    ): Promise<{
      ok: boolean;
      memoryId: string;
      content?: string;
      tags?: string[];
      source?: string;
      importance?: number;
      created_at?: string;
      error?: string;
    }> {
      const parsed = RECALL_INPUT_SCHEMA.parse(input);
      // Per-agent isolation: deps.store is the calling agent's per-agent DB.
      // Try memories table first, then memory_chunks.
      const mem = deps.store.getById?.(parsed.memoryId);
      if (mem) {
        return {
          ok: true,
          memoryId: parsed.memoryId,
          content: mem.content,
          tags: mem.tags,
          source: mem.source,
          importance: mem.importance,
          created_at: mem.created_at,
        };
      }
      const chunk = deps.store.getMemoryChunk?.(parsed.memoryId);
      if (chunk) {
        return {
          ok: true,
          memoryId: parsed.memoryId,
          content: chunk.body,
          tags: [],
          source: "memory_chunks",
          created_at: chunk.created_at,
        };
      }
      return { ok: false, memoryId: parsed.memoryId, error: "memory not found in this agent's store" };
    }
    ```

    ```ts
    import { z } from "zod";
    import { promises as fs } from "fs";
    import { resolve, normalize, relative } from "path";

    export const EDIT_INPUT_SCHEMA = z.object({
      path: z.enum(["MEMORY.md", "USER.md"]),
      mode: z.enum(["view", "create", "str_replace", "append"]),
      oldStr: z.string().optional(),  // for str_replace
      newStr: z.string().optional(),  // for str_replace
      content: z.string().optional(), // for create / append
    });

    export interface EditDeps {
      memoryRoot: string;  // absolute path of agent's memory root <workspace>/memory/
      agentName: string;
      log?: { warn: Function; error: Function };
    }

    /**
     * Phase 115 sub-scope 7 — clawcode_memory_edit (Anthropic memory_20250818 contract).
     *
     * SECURITY: jailed to <memoryRoot>. Path traversal attempts are rejected and logged.
     * Allowed file names: MEMORY.md, USER.md (zod enum enforces).
     * Path is RESOLVED inside <memoryRoot>; symlinks are NOT followed (lstat check).
     */
    export async function clawcodeMemoryEdit(
      input: z.infer<typeof EDIT_INPUT_SCHEMA>,
      deps: EditDeps,
    ): Promise<{ ok: boolean; after?: string; error?: string }> {
      const parsed = EDIT_INPUT_SCHEMA.parse(input);

      // Jail enforcement.
      const memRoot = normalize(resolve(deps.memoryRoot));
      const candidate = normalize(resolve(memRoot, parsed.path));
      const rel = relative(memRoot, candidate);
      if (rel.startsWith("..") || rel.startsWith("/") || rel.includes("\0")) {
        deps.log?.error({
          agent: deps.agentName,
          path: parsed.path,
          resolved: candidate,
          action: "memory-edit-jail-escape",
        }, "[security] clawcode_memory_edit jail escape attempt blocked");
        return { ok: false, error: "path outside memory root not allowed" };
      }

      // Symlink check — the file (if exists) MUST not be a symlink.
      try {
        const st = await fs.lstat(candidate);
        if (st.isSymbolicLink()) {
          deps.log?.error({
            agent: deps.agentName,
            path: parsed.path,
            action: "memory-edit-symlink-blocked",
          }, "[security] clawcode_memory_edit symlink blocked");
          return { ok: false, error: "symlinks not allowed in memory root" };
        }
      } catch (err: unknown) {
        if ((err as { code?: string })?.code !== "ENOENT") throw err;
      }

      switch (parsed.mode) {
        case "view": {
          try {
            const text = await fs.readFile(candidate, "utf8");
            return { ok: true, after: text };
          } catch (err: unknown) {
            if ((err as { code?: string })?.code === "ENOENT") return { ok: true, after: "" };
            throw err;
          }
        }
        case "create": {
          if (typeof parsed.content !== "string") return { ok: false, error: "create requires content" };
          await fs.writeFile(candidate, parsed.content, "utf8");
          return { ok: true, after: parsed.content };
        }
        case "append": {
          if (typeof parsed.content !== "string") return { ok: false, error: "append requires content" };
          let existing = "";
          try { existing = await fs.readFile(candidate, "utf8"); } catch (err: unknown) {
            if ((err as { code?: string })?.code !== "ENOENT") throw err;
          }
          const next = existing + parsed.content;
          await fs.writeFile(candidate, next, "utf8");
          return { ok: true, after: next };
        }
        case "str_replace": {
          if (typeof parsed.oldStr !== "string" || typeof parsed.newStr !== "string") {
            return { ok: false, error: "str_replace requires oldStr and newStr" };
          }
          let existing: string;
          try { existing = await fs.readFile(candidate, "utf8"); } catch (err: unknown) {
            if ((err as { code?: string })?.code === "ENOENT") return { ok: false, error: "file not found" };
            throw err;
          }
          if (!existing.includes(parsed.oldStr)) {
            return { ok: false, error: "oldStr not found in file" };
          }
          const next = existing.replace(parsed.oldStr, parsed.newStr);
          await fs.writeFile(candidate, next, "utf8");
          return { ok: true, after: next };
        }
      }
    }
    ```

    ```ts
    import { z } from "zod";
    import type { MemoryStore } from "../store.js";
    import { clawcodeMemoryEdit } from "./clawcode-memory-edit.js";

    export const ARCHIVE_INPUT_SCHEMA = z.object({
      chunkId: z.string().min(1).max(200),
      targetPath: z.enum(["MEMORY.md", "USER.md"]),
      // Optional: agent can supply a curated wrapping prose for the chunk.
      wrappingPrefix: z.string().optional(),
      wrappingSuffix: z.string().optional(),
    });

    export interface ArchiveDeps {
      store: MemoryStore;
      memoryRoot: string;
      agentName: string;
      log?: { warn: Function; info: Function; error: Function };
    }

    export async function clawcodeMemoryArchive(
      input: z.infer<typeof ARCHIVE_INPUT_SCHEMA>,
      deps: ArchiveDeps,
    ): Promise<{ ok: boolean; error?: string }> {
      const parsed = ARCHIVE_INPUT_SCHEMA.parse(input);
      const chunk = deps.store.getMemoryChunk?.(parsed.chunkId);
      if (!chunk) return { ok: false, error: "chunk not found" };

      const sectionContent = `\n\n${parsed.wrappingPrefix ?? ""}${chunk.body}${parsed.wrappingSuffix ?? ""}\n`;

      // Append to the target Tier 1 file.
      const editResult = await clawcodeMemoryEdit(
        { path: parsed.targetPath, mode: "append", content: sectionContent },
        { memoryRoot: deps.memoryRoot, agentName: deps.agentName, log: deps.log },
      );

      if (!editResult.ok) {
        return { ok: false, error: editResult.error };
      }

      deps.log?.info({
        agent: deps.agentName,
        chunkId: parsed.chunkId,
        targetPath: parsed.targetPath,
        action: "agent-curated-archive",
      }, "[diag] clawcode-memory-archive: agent-curated promotion");
      return { ok: true };
    }
    ```

    ```ts
    /**
     * Phase 115 D-10 hybrid policy (replaces Phase 95 D-04).
     *
     * Five rows of decision:
     *  1. newWikilinks                                 → auto-apply (additive, idempotent)  [unchanged from D-04]
     *  2. promotionCandidates (additive, score >= 80)  → auto-apply with 30-min Discord veto window
     *  3. promotionCandidates (mutating)               → operator-required (NO auto-apply)
     *  4. suggestedConsolidations                      → operator-required [unchanged from D-04]
     *  5. priority pass (D-05)                          → ALL promotion ALLOWED to mutate; auto-apply with 30-min veto
     */

    export interface ApplyDreamArgs {
      agentName: string;
      outcome: DreamPassOutcome;
      isPriorityPass: boolean;          // Phase 115 D-05 — set true when scheduled by tier-1 truncation trigger
      memoryRoot: string;
      log: Logger;
      discordSummary?: PostFn;          // for D-11 Discord summary
      vetoStore: VetoStore;             // tracks pending veto state for 30-min window
    }

    export async function applyDreamResult(args: ApplyDreamArgs): Promise<ApplyDreamReport> {
      // Row 1 — newWikilinks (always auto-apply, additive)
      await applyAutoLinks(args.agentName, args.outcome.result.newWikilinks);

      // Rows 2 + 5 — promotion candidates
      const candidates = args.outcome.result.promotionCandidates ?? [];
      const eligible = args.isPriorityPass
        ? candidates  // Row 5: priority pass — ALL allowed (override priorityScore floor)
        : candidates.filter(c => !isMutating(c) && (c.priorityScore ?? 0) >= 80);  // Row 2

      const operatorRequired = candidates.filter(c => !eligible.includes(c));  // Rows 3 + 4 candidates

      const runId = nanoid();
      // Schedule auto-apply with 30-min veto window (Row 2 + Row 5).
      if (eligible.length > 0) {
        await args.vetoStore.scheduleAutoApply({
          runId,
          agentName: args.agentName,
          candidates: eligible,
          deadline: Date.now() + 30 * 60 * 1000,
          isPriorityPass: args.isPriorityPass,
        });
      }

      // Row 4 — suggestedConsolidations always operator-required
      const consolidations = args.outcome.result.suggestedConsolidations ?? [];

      // D-11 — post structured Discord summary
      if ((eligible.length > 0 || operatorRequired.length > 0 || consolidations.length > 0) && args.discordSummary) {
        await args.discordSummary(buildPromotionSummary({
          runId,
          agentName: args.agentName,
          autoApplyCandidates: eligible,
          operatorRequiredCandidates: operatorRequired,
          consolidations,
          isPriorityPass: args.isPriorityPass,
        }));
      }

      return {
        runId,
        autoApplyScheduled: eligible.length,
        operatorRequiredCount: operatorRequired.length + consolidations.length,
        isPriorityPass: args.isPriorityPass,
      };
    }

    function isMutating(c: PromotionCandidate): boolean {
      // A candidate is mutating if it intends to overwrite an existing entry.
      return c.action === "edit" || c.action === "merge" || c.targetMode === "overwrite";
    }
    ```
