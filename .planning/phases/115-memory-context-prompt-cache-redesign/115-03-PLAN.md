    ```ts
    function enforceDropLowestImportance(
      sources: {
        identitySoulFingerprint: string;
        identityFile: string;
        identityCapabilityManifest: string;
        identityMemoryAutoload: string;
      },
      budgetTokens: number,
      warn: BudgetWarningCallback,
    ): {
      identitySoulFingerprint: string;
      identityFile: string;
      identityCapabilityManifest: string;
      identityMemoryAutoload: string;
      droppedTokens: number;
    } {
      let total = countTokens([
        sources.identitySoulFingerprint,
        sources.identityFile,
        sources.identityCapabilityManifest,
        sources.identityMemoryAutoload,
      ].join("\n\n"));

      if (total <= budgetTokens) {
        return { ...sources, droppedTokens: 0 };
      }

      // Step A: drop MEMORY.md oldest sections first (lowest importance).
      // Step B: truncate-bullets capability manifest.
      // Step C: head-tail truncate IDENTITY.md (70/20).
      // SOUL fingerprint is NEVER dropped (highest importance).
      let working = { ...sources };
      let dropped = 0;

      // Step A
      while (total > budgetTokens && countTokens(working.identityMemoryAutoload) > 100) {
        const halved = headTailTruncate(working.identityMemoryAutoload, Math.max(1, Math.floor(working.identityMemoryAutoload.length * 0.7)));
        const newTokens = countTokens(halved);
        const oldTokens = countTokens(working.identityMemoryAutoload);
        dropped += oldTokens - newTokens;
        total -= oldTokens - newTokens;
        working.identityMemoryAutoload = halved;
      }
      // Step B
      if (total > budgetTokens) {
        const truncated = truncateBullets(working.identityCapabilityManifest, Math.max(0, budgetTokens - total + countTokens(working.identityCapabilityManifest)) * 4);
        const oldTokens = countTokens(working.identityCapabilityManifest);
        const newTokens = countTokens(truncated);
        dropped += oldTokens - newTokens;
        total -= oldTokens - newTokens;
        working.identityCapabilityManifest = truncated;
      }
      // Step C
      if (total > budgetTokens) {
        const targetChars = Math.max(0, (budgetTokens - countTokens(working.identityCapabilityManifest) - countTokens(working.identitySoulFingerprint)) * 4);
        const truncated = headTailTruncate(working.identityFile, targetChars);
        const oldTokens = countTokens(working.identityFile);
        const newTokens = countTokens(truncated);
        dropped += oldTokens - newTokens;
        total -= oldTokens - newTokens;
        working.identityFile = truncated;
      }

      warn?.({ section: "identity", beforeTokens: total + dropped, budgetTokens, strategy: "drop-lowest-importance", droppedTokens: dropped });
      return { ...working, droppedTokens: dropped };
    }
    ```

    ```ts
    function headTailTruncate(text: string, targetChars: number): string {
      if (text.length <= targetChars) return text;
      const headLen = Math.floor(targetChars * 0.7);
      const tailLen = Math.floor(targetChars * 0.2);
      const dropped = text.length - headLen - tailLen;
      const head = text.slice(0, headLen);
      const tail = text.slice(-tailLen);
      return `${head}\n\n[TRUNCATED — ${dropped} chars dropped]\n\n${tail}`;
    }
    ```

    ```ts
    function truncateBullets(text: string, targetChars: number): string {
      if (text.length <= targetChars) return text;
      // Drop trailing bullet lines until under target.
      const lines = text.split("\n");
      let acc = "";
      const accLines: string[] = [];
      for (const line of lines) {
        if (acc.length + line.length + 1 > targetChars) break;
        accLines.push(line);
        acc += line + "\n";
      }
      return accLines.join("\n") + (accLines.length < lines.length ? `\n[TRUNCATED — ${lines.length - accLines.length} bullets dropped]` : "");
    }
    ```

    ```ts
    /**
     * Phase 115 sub-scope 9 Phase 1 — no-LLM tool-output pruning.
     *
     * Replace old tool outputs in conversation history with 1-line summaries:
     * "[<tool_name> output redacted by tier-1 compaction; <N> chars saved]"
     *
     * Trigger: when conversation_history token estimate > threshold (default 50% context fill).
     * Anti-thrash: skip if last 2 prunes saved < 10% each.
     * NEVER touch the most-recent K turns (K=5 default).
     *
     * Phases 2 (LLM mid-summarization) and 3 (drop oldest) are DEFERRED to follow-on phases.
     */

    export interface ToolOutputPruneInput {
      turns: ConversationTurn[];        // most-recent-last
      maxTokenSpend: number;            // budget hint; abort early when reached
      preserveTailTurns?: number;       // default 5
      thresholdSavingsPct?: number;     // anti-thrash: skip if < this; default 10
      lastTwoSavingsPct?: number[];     // for anti-thrash check
    }

    export interface ToolOutputPruneOutput {
      prunedTurns: ConversationTurn[];
      charsSaved: number;
      summariesInserted: number;
      skipped: boolean;
      skipReason?: string;
    }

    const TOOL_OUTPUT_REGEX = /<tool_use_result.*?>([\s\S]*?)<\/tool_use_result>/g;

    export function pruneOldToolOutputs(input: ToolOutputPruneInput): ToolOutputPruneOutput {
      const preserveTail = input.preserveTailTurns ?? 5;
      const thresholdPct = input.thresholdSavingsPct ?? 10;

      // Anti-thrash gate
      if (input.lastTwoSavingsPct?.length === 2 && input.lastTwoSavingsPct.every(p => p < thresholdPct)) {
        return {
          prunedTurns: input.turns,
          charsSaved: 0,
          summariesInserted: 0,
          skipped: true,
          skipReason: "anti-thrash: last 2 prunes saved < threshold",
        };
      }

      const turns = [...input.turns];  // copy for immutability
      const cutoffIdx = Math.max(0, turns.length - preserveTail);

      let charsSaved = 0;
      let summariesInserted = 0;
      const totalBefore = turns.reduce((acc, t) => acc + (t.content?.length ?? 0), 0);

      for (let i = 0; i < cutoffIdx; i++) {
        const turn = turns[i];
        if (!turn?.content) continue;

        const oldContent = turn.content;
        const newContent = oldContent.replace(TOOL_OUTPUT_REGEX, (_match, body) => {
          const saved = body.length;
          if (saved < 200) return _match;  // don't bother for tiny outputs
          charsSaved += saved;
          summariesInserted += 1;
          return `[tool output redacted by tier-1 compaction; ${saved} chars saved]`;
        });

        turns[i] = { ...turn, content: newContent };

        if (charsSaved >= input.maxTokenSpend * 4) break;  // budget hit
      }

      const totalAfter = turns.reduce((acc, t) => acc + (t.content?.length ?? 0), 0);
      const savingsPct = totalBefore > 0 ? ((totalBefore - totalAfter) / totalBefore) * 100 : 0;

      return {
        prunedTurns: turns,
        charsSaved,
        summariesInserted,
        skipped: false,
      };
    }
    ```
