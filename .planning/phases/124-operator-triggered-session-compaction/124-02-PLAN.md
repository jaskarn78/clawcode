       import { readFileSync } from 'fs';
       import yaml from 'js-yaml';

       it('no agent heartbeat block conflates AUTO-RESET: DISABLED with auto-compact', () => {
         const config = yaml.load(readFileSync('clawcode.yaml', 'utf8')) as any;
         const violations: string[] = [];
         for (const agent of config.agents ?? []) {
           const heartbeat = String(agent.heartbeat ?? agent.systemPrompt ?? '');
           const blocks = heartbeat.split(/^##\s/m); // each block starts at '## '
           for (const block of blocks) {
             const hasReset = /AUTO-RESET:\s*DISABLED/.test(block);
             const hasCompact = /auto-compact/i.test(block);
             if (hasReset && hasCompact) {
               violations.push(`${agent.name}: ${block.slice(0, 80)}...`);
             }
           }
         }
         expect(violations).toEqual([]);
       });
