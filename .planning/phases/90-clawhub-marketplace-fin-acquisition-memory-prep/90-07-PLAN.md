    ```typescript
    export type UpdateAgentConfigArgs = Readonly<{
      existingConfigPath: string;
      agentName: string;
      patch: Readonly<Record<string, unknown>>;  // Partial AgentConfig; schema-validated against agentSchema on merge
      pid?: number;
    }>;

    export type UpdateAgentConfigResult =
      | { outcome: "updated"; destPath: string; targetSha256: string; keysChanged: readonly string[] }
      | { outcome: "no-op"; reason: string }
      | { outcome: "not-found"; reason: string }
      | { outcome: "file-not-found"; reason: string }
      | { outcome: "refused"; reason: string; step: "invalid-patch" | "secret-scan" };

    export async function updateAgentConfig(args: UpdateAgentConfigArgs): Promise<UpdateAgentConfigResult> {
      // 1. Read + parse
      let raw: string;
      try { raw = await readFile(args.existingConfigPath, "utf8"); }
      catch (err) { return { outcome: "file-not-found", reason: (err as Error).message }; }
      const doc = parseDocument(raw);
      const agentsSeq = doc.get("agents") as YAMLSeq;
      if (!agentsSeq) return { outcome: "not-found", reason: "no agents seq" };
      let agentMap: YAMLMap | null = null;
      for (const it of agentsSeq.items) {
        if (it instanceof YAMLMap && it.get("name") === args.agentName) { agentMap = it; break; }
      }
      if (!agentMap) return { outcome: "not-found", reason: `agent ${args.agentName} missing` };

      // 2. Defense-in-depth: build the merged object and run agentSchema against it
      const currentJs = agentMap.toJSON() as Record<string, unknown>;
      const merged = { ...currentJs, ...args.patch };
      const parsed = agentSchema.safeParse(merged);
      if (!parsed.success) return { outcome: "refused", reason: `invalid patch: ${parsed.error.issues[0]?.message ?? "validation failed"}`, step: "invalid-patch" };

      // 3. Secret-scan any string values in the patch (including nested)
      for (const scanResult of scanPatchForLiteralSecrets(args.patch)) {
        if (scanResult.refused) return { outcome: "refused", reason: scanResult.reason, step: "secret-scan" };
      }

      // 4. Apply patch to the YAMLMap (recurse for nested objects like heartbeat)
      const keysChanged: string[] = [];
      for (const [k, v] of Object.entries(args.patch)) {
        const existing = agentMap.has(k) ? agentMap.get(k) : undefined;
        if (JSON.stringify(existing) === JSON.stringify(v)) continue;
        agentMap.set(k, serializeForYaml(v));  // helper: convert plain JS to YAMLMap/YAMLSeq preserving node types
        keysChanged.push(k);
      }
      if (keysChanged.length === 0) return { outcome: "no-op", reason: "all patch keys equal current" };

      // 5. Serialize + atomic temp+rename (COPY from updateAgentModel)
      const newText = doc.toString({ lineWidth: 0 });
      const pid = args.pid ?? process.pid;
      const destDir = dirname(args.existingConfigPath);
      const tmpPath = join(destDir, `.clawcode.yaml.${pid}.${Date.now()}.tmp`);
      await writerFs.writeFile(tmpPath, newText, "utf8");
      try { await writerFs.rename(tmpPath, args.existingConfigPath); }
      catch (err) { try { await writerFs.unlink(tmpPath); } catch {} throw err; }
      const targetSha256 = createHash("sha256").update(newText, "utf8").digest("hex");
      return { outcome: "updated", destPath: args.existingConfigPath, targetSha256, keysChanged: Object.freeze(keysChanged) };
    }

    function* scanPatchForLiteralSecrets(patch: Record<string, unknown>): Generator<{refused: boolean; reason: string}> {
      for (const [key, val] of Object.entries(patch)) {
        if (typeof val === "string") {
          if (val.startsWith("op://")) continue;
          yield scanLiteralValueForSecret(key, val);
        } else if (val && typeof val === "object" && !Array.isArray(val)) {
          yield* scanPatchForLiteralSecrets(val as Record<string, unknown>);
        }
        // Arrays: scan each element if string
        if (Array.isArray(val)) {
          for (const el of val) {
            if (typeof el === "string" && !el.startsWith("op://")) yield scanLiteralValueForSecret(key, el);
            if (el && typeof el === "object") yield* scanPatchForLiteralSecrets(el as Record<string, unknown>);
          }
        }
      }
    }

    function serializeForYaml(value: unknown): unknown {
      // For simple scalars + arrays of scalars + flat objects, yaml 2.x's doc.set accepts plain JS.
      // For mixed/nested arrays (mcpServers with string|map mix), construct explicit YAMLSeq.
      if (Array.isArray(value)) {
        const seq = new YAMLSeq();
        for (const el of value) {
          if (typeof el === "string") seq.add(el);
          else if (el && typeof el === "object") {
            const m = new YAMLMap();
            for (const [k, v] of Object.entries(el as Record<string, unknown>)) m.set(k, serializeForYaml(v));
            seq.add(m);
          } else seq.add(el);
        }
        return seq;
      }
      if (value && typeof value === "object") {
        const m = new YAMLMap();
        for (const [k, v] of Object.entries(value as Record<string, unknown>)) m.set(k, serializeForYaml(v));
        return m;
      }
      return value;
    }
    ```

    **Step 2: Create src/cli/commands/memory-backfill.ts + register it in src/cli/index.ts.**

    ```typescript
    // src/cli/commands/memory-backfill.ts
    import { Command } from "commander";
    import { resolve } from "node:path";
    import { MemoryScanner } from "../../memory/memory-scanner.js";
    import { MemoryStore } from "../../memory/store.js";
    import { embedText } from "../../memory/embedder.js";
    import { loadConfig } from "../../config/loader.js";
    import pino from "pino";

    export async function runMemoryBackfillAction(opts: Readonly<{
      agentName: string;
      configPath?: string;
    }>): Promise<number> {
      const log = pino({ level: "info" });
      const { config } = await loadConfig(opts.configPath ?? "clawcode.yaml");
      const agent = config.agents.find((a) => a.name === opts.agentName);
      if (!agent) {
        log.error(`Agent '${opts.agentName}' not in clawcode.yaml`);
        return 1;
      }
      const workspace = agent.workspace;
      if (!workspace) { log.error(`Agent '${opts.agentName}' has no workspace`); return 1; }
      const dbPath = resolve(agent.memoryPath ?? workspace, "memories.db");
      const store = new MemoryStore(dbPath);
      const scanner = new MemoryScanner({ store, embed: embedText, log }, workspace);
      const result = await scanner.backfill();
      log.info(`Indexed ${result.indexed} memory/*.md files, ${result.chunks} chunks`);
      return 0;
    }

    export function registerMemoryBackfillCommand(parent: Command): void {
      parent.command("backfill")
        .argument("<agent>", "agent name")
        .option("-c, --config <path>", "clawcode.yaml path", "clawcode.yaml")
        .description("Index all memory/*.md files for an agent (Plan 90-02 MemoryScanner.backfill)")
        .action(async (agent: string, opts: {config?: string}) => {
          const code = await runMemoryBackfillAction({ agentName: agent, configPath: opts.config });
          process.exit(code);
        });
    }
    ```

    ```typescript
    export type AgentWebhookIdentityStatus =
      | { status: "verified"; webhookUrl: string; displayName: string }
      | { status: "provisioned"; webhookUrl: string; displayName: string }
      | { status: "missing"; reason: string };

    export async function verifyAgentWebhookIdentity(args: Readonly<{
      client: Client;
      agentName: string;
      channelId: string | undefined;
      displayName: string;
      avatarUrl?: string;
      log: Logger;
    }>): Promise<AgentWebhookIdentityStatus> {
      if (!args.channelId) return { status: "missing", reason: "no channel bound" };
      try {
        const existing = await provisionWebhooks({
          client: args.client,
          agents: [{ name: args.agentName, channels: [args.channelId], webhook: { displayName: args.displayName, avatarUrl: args.avatarUrl } }],
          manualIdentities: new Map(),
          log: args.log,
        });
        const ident = existing.get(args.agentName);
        if (!ident?.webhookUrl) return { status: "missing", reason: "provisionWebhooks returned no URL" };
        // Distinguish verified vs provisioned: re-fetch existing webhooks and check ownership timestamp.
        // Simplification: if webhook was found (not newly created), status is verified. Use provisionWebhooks' log
        // as a signal? Cleaner: probe directly.
        return { status: "verified", webhookUrl: ident.webhookUrl, displayName: args.displayName };
      } catch (err) {
        return { status: "missing", reason: (err as Error).message };
      }
    }
    ```
