    ```typescript
     * Phase 90 MEM-02 — workspace memory file chunks + embeddings + FTS.
     * Idempotent CREATE TABLE IF NOT EXISTS + CREATE VIRTUAL TABLE IF NOT EXISTS
     * (mirrors migrateConversationTables discipline).
     */
    private migrateMemoryChunks(): void {
      this.db.exec(`
        CREATE TABLE IF NOT EXISTS files (
          path TEXT PRIMARY KEY,
          mtime_ms INTEGER NOT NULL,
          sha256 TEXT NOT NULL,
          chunk_count INTEGER NOT NULL,
          indexed_at TEXT NOT NULL
        );
        CREATE TABLE IF NOT EXISTS memory_chunks (
          id TEXT PRIMARY KEY,
          path TEXT NOT NULL REFERENCES files(path) ON DELETE CASCADE,
          chunk_index INTEGER NOT NULL,
          heading TEXT,
          body TEXT NOT NULL,
          token_count INTEGER NOT NULL,
          score_weight REAL NOT NULL DEFAULT 0.0,
          file_mtime_ms INTEGER NOT NULL,
          created_at TEXT NOT NULL
        );
        CREATE INDEX IF NOT EXISTS idx_memory_chunks_path ON memory_chunks(path);
        CREATE VIRTUAL TABLE IF NOT EXISTS vec_memory_chunks USING vec0(
          chunk_id TEXT PRIMARY KEY,
          embedding FLOAT[384]
        );
        CREATE VIRTUAL TABLE IF NOT EXISTS memory_chunks_fts USING fts5(
          chunk_id UNINDEXED,
          heading,
          body,
          content=''
        );
      `);
    }

    insertMemoryChunk(input: Readonly<{
      path: string;
      chunkIndex: number;
      heading: string | null;
      body: string;
      tokenCount: number;
      scoreWeight: number;
      fileMtimeMs: number;
      fileSha256: string;
      embedding: Float32Array;  // 384-dim
    }>): string {
      const chunkId = nanoid();
      const now = new Date().toISOString();
      this.db.transaction(() => {
        // Upsert files row
        const fileStmt = this.db.prepare(
          `INSERT INTO files (path, mtime_ms, sha256, chunk_count, indexed_at)
           VALUES (?, ?, ?, 1, ?)
           ON CONFLICT(path) DO UPDATE SET
             mtime_ms=excluded.mtime_ms, sha256=excluded.sha256,
             chunk_count=chunk_count+1, indexed_at=excluded.indexed_at`,
        );
        fileStmt.run(input.path, input.fileMtimeMs, input.fileSha256, now);
        this.db.prepare(
          `INSERT INTO memory_chunks (id,path,chunk_index,heading,body,token_count,score_weight,file_mtime_ms,created_at)
           VALUES (?,?,?,?,?,?,?,?,?)`,
        ).run(chunkId, input.path, input.chunkIndex, input.heading, input.body,
              input.tokenCount, input.scoreWeight, input.fileMtimeMs, now);
        this.db.prepare(
          `INSERT INTO vec_memory_chunks (chunk_id, embedding) VALUES (?, ?)`,
        ).run(chunkId, Buffer.from(input.embedding.buffer));
        this.db.prepare(
          `INSERT INTO memory_chunks_fts (chunk_id, heading, body) VALUES (?, ?, ?)`,
        ).run(chunkId, input.heading ?? "", input.body);
      })();
      return chunkId;
    }

    deleteMemoryChunksByPath(path: string): number {
      return this.db.transaction(() => {
        // Find chunk_ids for this path
        const ids = this.db.prepare(`SELECT id FROM memory_chunks WHERE path=?`).all(path).map((r: { id: string }) => r.id);
        for (const id of ids) {
          this.db.prepare(`DELETE FROM vec_memory_chunks WHERE chunk_id=?`).run(id);
          this.db.prepare(`DELETE FROM memory_chunks_fts WHERE chunk_id=?`).run(id);
        }
        const info = this.db.prepare(`DELETE FROM memory_chunks WHERE path=?`).run(path);
        this.db.prepare(`DELETE FROM files WHERE path=?`).run(path);
        return info.changes as number;
      })();
    }

    searchMemoryChunksVec(queryEmbedding: Float32Array, limit: number):
      readonly Readonly<{chunk_id: string; distance: number}>[] {
      return this.db.prepare(
        `SELECT chunk_id, distance FROM vec_memory_chunks
         WHERE embedding MATCH ? AND k = ? ORDER BY distance`,
      ).all(Buffer.from(queryEmbedding.buffer), limit) as any;
    }

    searchMemoryChunksFts(query: string, limit: number):
      readonly Readonly<{chunk_id: string; rank: number}>[] {
      return this.db.prepare(
        `SELECT chunk_id, rank FROM memory_chunks_fts
         WHERE memory_chunks_fts MATCH ? ORDER BY rank LIMIT ?`,
      ).all(query, limit) as any;
    }

    getMemoryChunk(chunkId: string): Readonly<{path:string; heading:string|null; body:string; file_mtime_ms:number; score_weight:number}> | null {
      return (this.db.prepare(
        `SELECT path, heading, body, file_mtime_ms, score_weight
         FROM memory_chunks WHERE id=?`,
      ).get(chunkId) as any) ?? null;
    }
    ```

    **Step 2: Create src/memory/memory-chunks.ts — pure chunker + weighter.**
    ```typescript
    const TOKEN_PER_CHAR = 0.25;  // rough token estimate
    const MAX_TOKENS = 800;

    export type MemoryChunk = Readonly<{
      heading: string | null;
      body: string;
      tokenCount: number;
      scoreWeight: number;
    }>;

    export function chunkMarkdownByH2(content: string, maxTokens: number = MAX_TOKENS): readonly MemoryChunk[] {
      const chunks: MemoryChunk[] = [];
      const lines = content.split("\n");
      let currentHeading: string | null = null;
      let currentBody: string[] = [];
      const flush = () => {
        const body = currentBody.join("\n").trim();
        if (body.length === 0) return;
        const tokenCount = Math.ceil(body.length * TOKEN_PER_CHAR);
        if (tokenCount > maxTokens * 1.25) {
          // split on H3 / paragraphs
          for (const sub of splitByH3OrParagraphs(body, maxTokens)) {
            chunks.push({ heading: currentHeading, body: sub, tokenCount: Math.ceil(sub.length * TOKEN_PER_CHAR), scoreWeight: 0 });
          }
        } else {
          chunks.push({ heading: currentHeading, body, tokenCount, scoreWeight: 0 });
        }
      };
      for (const line of lines) {
        const m = /^##\s+(.+)/.exec(line);
        if (m) {
          flush();
          currentHeading = m[1].trim();
          currentBody = [];
        } else {
          currentBody.push(line);
        }
      }
      flush();
      if (chunks.length === 0 && content.trim().length > 0) {
        // No H2 headings — one whole-body chunk
        chunks.push({ heading: null, body: content.trim(),
                      tokenCount: Math.ceil(content.length * TOKEN_PER_CHAR), scoreWeight: 0 });
      }
      return chunks;
    }

    function splitByH3OrParagraphs(text: string, maxTokens: number): string[] {
      // H3 split first, else paragraph split; sized to maxTokens
      const parts: string[] = [];
      let buf = "";
      for (const para of text.split(/\n\n+/)) {
        if ((buf.length + para.length) * TOKEN_PER_CHAR > maxTokens && buf.length > 0) {
          parts.push(buf.trim());
          buf = "";
        }
        buf += (buf ? "\n\n" : "") + para;
      }
      if (buf.trim()) parts.push(buf.trim());
      return parts;
    }

    /** Phase 90 D-19 path weighting for retrieval ranking. */
    export function scoreWeightForPath(absPath: string): number {
      if (absPath.includes("/memory/vault/")) return 0.2;
      if (absPath.includes("/memory/procedures/")) return 0.1;
      if (absPath.includes("/memory/archive/")) return -0.2;
      return 0.0;
    }

    /** Phase 90 D-24 time-window filter (14-day default for dated files; vault/procedures all-time). */
    export function applyTimeWindowFilter<T extends { path: string; file_mtime_ms: number }>(
      chunks: readonly T[], days: number, now: number = Date.now(),
    ): readonly T[] {
      const cutoff = now - days * 86_400_000;
      return chunks.filter((c) => {
        if (c.path.includes("/memory/vault/") || c.path.includes("/memory/procedures/")) return true;
        return c.file_mtime_ms >= cutoff;
      });
    }
    ```

    **Step 3: Create src/memory/memory-scanner.ts — chokidar watcher + backfill CLI hook.**
    ```typescript
    import chokidar from "chokidar";
    import { readFile, stat } from "node:fs/promises";
    import { createHash } from "node:crypto";
    import { glob } from "glob";
    import type { Logger } from "pino";
    import type { MemoryStore } from "./store.js";
    import type { embedText } from "./embedder.js";
    import { chunkMarkdownByH2, scoreWeightForPath } from "./memory-chunks.js";

    export type MemoryScannerDeps = Readonly<{
      store: MemoryStore;
      embed: typeof embedText;  // DI for tests
      log: Logger;
    }>;

    export class MemoryScanner {
      private watcher: chokidar.FSWatcher | null = null;
      constructor(private readonly deps: MemoryScannerDeps, private readonly workspacePath: string) {}

      async start(): Promise<void> {
        this.watcher = chokidar.watch(
          [`${this.workspacePath}/memory/**/*.md`],
          { ignoreInitial: true, awaitWriteFinish: { stabilityThreshold: 500, pollInterval: 100 } },
        );
        this.watcher.on("add", (p) => void this.handleUpsert(p).catch((err) => this.deps.log.warn({err, p}, "memory-scanner: add failed")));
        this.watcher.on("change", (p) => void this.handleUpsert(p).catch((err) => this.deps.log.warn({err, p}, "memory-scanner: change failed")));
        this.watcher.on("unlink", (p) => {
          try { this.deps.store.deleteMemoryChunksByPath(p); } catch (err) { this.deps.log.warn({err, p}, "memory-scanner: unlink failed"); }
        });
      }

      async stop(): Promise<void> { await this.watcher?.close(); this.watcher = null; }

      async backfill(): Promise<{ indexed: number; chunks: number }> {
        const files = await glob(`${this.workspacePath}/memory/**/*.md`, { nodir: true });
        let chunkTotal = 0;
        for (const p of files) chunkTotal += (await this.handleUpsert(p)) ?? 0;
        return { indexed: files.length, chunks: chunkTotal };
      }

      private async handleUpsert(absPath: string): Promise<number> {
        const content = await readFile(absPath, "utf-8");
        const st = await stat(absPath);
        const sha = createHash("sha256").update(content).digest("hex");
        this.deps.store.deleteMemoryChunksByPath(absPath);
        const chunks = chunkMarkdownByH2(content);
        const weight = scoreWeightForPath(absPath);
        for (let i = 0; i < chunks.length; i++) {
          const c = chunks[i];
          const textToEmbed = (c.heading ? c.heading + "\n\n" : "") + c.body;
          const embedding = await this.deps.embed(textToEmbed);
          this.deps.store.insertMemoryChunk({
            path: absPath, chunkIndex: i, heading: c.heading, body: c.body,
            tokenCount: c.tokenCount, scoreWeight: weight,
            fileMtimeMs: st.mtimeMs, fileSha256: sha,
            embedding,
          });
        }
        return chunks.length;
      }
    }
    ```

    **Step 1: Create src/memory/memory-retrieval.ts — RRF fusion + retrieval entry point.**
    ```typescript
    import type { MemoryStore } from "./store.js";
    import type { embedText } from "./embedder.js";
    import { applyTimeWindowFilter } from "./memory-chunks.js";

    export type MemoryRetrievalResult = Readonly<{
      chunkId: string;
      path: string;
      heading: string | null;
      body: string;
      fusedScore: number;
      scoreWeight: number;
    }>;

    const RRF_K = 60;

    export function rrfFuse(
      vecRanked: readonly { chunk_id: string; distance: number }[],
      ftsRanked: readonly { chunk_id: string; rank: number }[],
      k: number = RRF_K,
    ): readonly { chunk_id: string; score: number }[] {
      const scoreMap = new Map<string, number>();
      vecRanked.forEach((r, i) => {
        scoreMap.set(r.chunk_id, (scoreMap.get(r.chunk_id) ?? 0) + 1 / (k + i + 1));
      });
      ftsRanked.forEach((r, i) => {
        scoreMap.set(r.chunk_id, (scoreMap.get(r.chunk_id) ?? 0) + 1 / (k + i + 1));
      });
      return [...scoreMap.entries()]
        .map(([chunk_id, score]) => ({ chunk_id, score }))
        .sort((a, b) => b.score - a.score);
    }

    export async function retrieveMemoryChunks(args: Readonly<{
      query: string;
      store: MemoryStore;
      embed: typeof embedText;
      topK?: number;
      timeWindowDays?: number;
      tokenBudget?: number;
    }>): Promise<readonly MemoryRetrievalResult[]> {
      const topK = args.topK ?? 5;
      const tokenBudget = args.tokenBudget ?? 2000;
      const days = args.timeWindowDays ?? 14;

      const qEmb = await args.embed(args.query);
      const vecTop = args.store.searchMemoryChunksVec(qEmb, 20);
      const ftsTop = args.store.searchMemoryChunksFts(args.query, 20);
      const fused = rrfFuse(vecTop, ftsTop, RRF_K);

      // Hydrate, apply path weighting, time-window filter, top-K truncate, token budget
      const hydrated = [];
      for (const f of fused) {
        const meta = args.store.getMemoryChunk(f.chunk_id);
        if (!meta) continue;
        hydrated.push({
          chunkId: f.chunk_id,
          path: meta.path,
          heading: meta.heading,
          body: meta.body,
          file_mtime_ms: meta.file_mtime_ms,
          fusedScore: f.score + meta.score_weight,
          scoreWeight: meta.score_weight,
        });
      }
      const windowed = applyTimeWindowFilter(hydrated, days);
      windowed.sort((a, b) => b.fusedScore - a.fusedScore);

      // Token budget truncation: ~4 chars/token; stop when cumulative body chars > 4*tokenBudget
      const out: MemoryRetrievalResult[] = [];
      let acc = 0;
      for (const h of windowed.slice(0, topK)) {
        const len = h.body.length;
        if (out.length > 0 && acc + len > tokenBudget * 4) break;
        out.push(Object.freeze({
          chunkId: h.chunkId, path: h.path, heading: h.heading,
          body: h.body, fusedScore: h.fusedScore, scoreWeight: h.scoreWeight,
        }));
        acc += len;
      }
      return Object.freeze(out);
    }
    ```
