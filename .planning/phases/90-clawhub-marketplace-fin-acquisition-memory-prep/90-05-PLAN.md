    ```typescript
      command: string;

      env: Record<string, Readonly<{

    export async function fetchClawhubPlugins(args: Readonly<{
      baseUrl: string; query?: string; cursor?: string; authToken?: string;
      deps?: ClawhubClientDeps;
    }>): Promise<ClawhubPluginsResponse> {
      const f = args.deps?.fetch ?? globalThis.fetch;
      const url = new URL("/api/v1/plugins", args.baseUrl);
      if (args.query) url.searchParams.set("q", args.query);
      if (args.cursor) url.searchParams.set("cursor", args.cursor);
      const headers: Record<string, string> = { "User-Agent": USER_AGENT, "Accept": "application/json" };
      if (args.authToken) headers["Authorization"] = `Bearer ${args.authToken}`;
      const res = await f(url.toString(), { headers });
      if (res.status === 429) { /* ... same as fetchClawhubSkills ... */ }
      if (res.status === 401 || res.status === 403) throw new ClawhubAuthRequiredError(`clawhub plugins: auth required (${res.status})`);
      if (!res.ok) throw new Error(`clawhub plugins: ${res.status} ${res.statusText}`);
      const body = await res.json() as ClawhubPluginsResponse;
      if (!Array.isArray(body.items)) throw new ClawhubManifestInvalidError("malformed plugins response");
      return body;
    }

    export async function downloadClawhubPluginManifest(args: Readonly<{
      manifestUrl: string; authToken?: string; deps?: ClawhubClientDeps;
    }>): Promise<ClawhubPluginManifest> {
      const f = args.deps?.fetch ?? globalThis.fetch;
      const headers: Record<string, string> = { "User-Agent": USER_AGENT, "Accept": "application/json" };
      if (args.authToken) headers["Authorization"] = `Bearer ${args.authToken}`;
      const res = await f(args.manifestUrl, { headers });
      if (res.status === 429) throw new ClawhubRateLimitedError(60_000, "manifest: rate-limited");
      if (res.status === 401 || res.status === 403) throw new ClawhubAuthRequiredError("manifest: auth required");
      if (!res.ok) throw new Error(`manifest: ${res.status}`);
      const m = await res.json() as ClawhubPluginManifest;
      if (typeof m.name !== "string" || typeof m.command !== "string") {
        throw new ClawhubManifestInvalidError("manifest missing required name/command");
      }
      return m;
    }
    ```

    ```typescript
    import { ClawhubRateLimitedError, ClawhubAuthRequiredError, ClawhubManifestInvalidError, type ClawhubPluginManifest } from "./clawhub-client.js";
    import { updateAgentMcpServers } from "../migration/yaml-writer.js";
    import { scanLiteralValueForSecret } from "../migration/skills-secret-scan.js";

    export type PluginInstallOutcome =
      | { kind: "installed"; plugin: string; pluginVersion: string; }
      | { kind: "installed-persist-failed"; plugin: string; pluginVersion: string; persist_error: string; }
      | { kind: "already-installed"; plugin: string; reason: string; }
      | { kind: "blocked-secret-scan"; plugin: string; field: string; reason: string; }
      | { kind: "manifest-invalid"; plugin: string; reason: string; }
      | { kind: "config-missing"; plugin: string; missing_field: string; }
      | { kind: "auth-required"; plugin: string; reason: string; }
      | { kind: "rate-limited"; plugin: string; retryAfterMs: number; }
      | { kind: "not-in-catalog"; plugin: string; };

    export type NormalizedMcpServerEntry = Readonly<{
      name: string;
      command: string;
      args: readonly string[];
      env: Readonly<Record<string, string>>;
      optional?: boolean;
    }>;

    export function normalizePluginManifest(
      manifest: ClawhubPluginManifest,
      configInputs: Readonly<Record<string, string>>,
    ): { ok: true; entry: NormalizedMcpServerEntry } | { ok: false; reason: string; missing_field?: string } {
      // Required-field gate
      for (const [envName, spec] of Object.entries(manifest.env ?? {})) {
        if (spec.required && !(envName in configInputs) && (spec.default === null || spec.default === undefined)) {
          return { ok: false, reason: `missing required env: ${envName}`, missing_field: envName };
        }
      }
      const env: Record<string, string> = {};
      for (const [envName, spec] of Object.entries(manifest.env ?? {})) {
        env[envName] = configInputs[envName] ?? spec.default ?? "";
      }
      return {
        ok: true,
        entry: Object.freeze({
          name: manifest.name,
          command: manifest.command,
          args: Object.freeze([...manifest.args]),
          env: Object.freeze(env),
          optional: false,
        }),
      };
    }

    export async function installClawhubPlugin(args: Readonly<{
      manifest: ClawhubPluginManifest;
      agentName: string;
      configPath: string;
      configInputs: Readonly<Record<string, string>>;
    }>): Promise<PluginInstallOutcome> {
      const norm = normalizePluginManifest(args.manifest, args.configInputs);
      if (!norm.ok) {
        if (norm.missing_field) return { kind: "config-missing", plugin: args.manifest.name, missing_field: norm.missing_field };
        return { kind: "manifest-invalid", plugin: args.manifest.name, reason: norm.reason };
      }
      // Secret-scan env values (D-08 / Phase 84 classifier)
      for (const [k, v] of Object.entries(norm.entry.env)) {
        if (v.startsWith("op://")) continue;  // 1Password refs pass
        const scan = scanLiteralValueForSecret(k, v);
        if (scan.refused) {
          return { kind: "blocked-secret-scan", plugin: args.manifest.name, field: k, reason: scan.reason };
        }
      }
      // Atomic YAML persist
      try {
        const result = await updateAgentMcpServers({
          existingConfigPath: args.configPath,
          agentName: args.agentName,
          entry: norm.entry,
          op: "add",
        });
        if (result.outcome === "updated" || result.outcome === "no-op") {
          return { kind: "installed", plugin: args.manifest.name, pluginVersion: args.manifest.version };
        }
        if (result.outcome === "not-found") return { kind: "not-in-catalog", plugin: args.manifest.name };
        return { kind: "installed-persist-failed", plugin: args.manifest.name, pluginVersion: args.manifest.version, persist_error: (result as any).reason };
      } catch (err) {
        return { kind: "installed-persist-failed", plugin: args.manifest.name, pluginVersion: args.manifest.version, persist_error: err instanceof Error ? err.message : String(err) };
      }
    }
    ```

    ```typescript
    export type UpdateAgentMcpServersArgs = Readonly<{
      existingConfigPath: string;
      agentName: string;
      entry: Readonly<{
        name: string;
        command: string;
        args: readonly string[];
        env: Readonly<Record<string, string>>;
        optional?: boolean;
      }>;
      op: "add" | "remove";
      pid?: number;
    }>;

    export type UpdateAgentMcpServersResult =
      | { outcome: "updated"; destPath: string; targetSha256: string }
      | { outcome: "no-op"; reason: string }
      | { outcome: "not-found"; reason: string }
      | { outcome: "file-not-found"; reason: string }
      | { outcome: "refused"; reason: string; step: "secret-scan" | "invalid-entry" };

    export async function updateAgentMcpServers(args: UpdateAgentMcpServersArgs): Promise<UpdateAgentMcpServersResult> {
      // 1. Validate entry shape via mcpServerSchema
      const parsed = mcpServerSchema.safeParse({
        name: args.entry.name, command: args.entry.command,
        args: [...args.entry.args], env: { ...args.entry.env },
        optional: args.entry.optional ?? false,
      });
      if (!parsed.success) {
        return { outcome: "refused", reason: `invalid entry: ${parsed.error.message.slice(0,200)}`, step: "invalid-entry" };
      }
      // 2. Defense-in-depth secret scan on env values (matches Phase 84 classifier)
      for (const [k, v] of Object.entries(args.entry.env)) {
        if (v.startsWith("op://")) continue;
        const s = scanLiteralValueForSecret(k, v);
        if (s.refused) return { outcome: "refused", reason: `env.${k}: ${s.reason}`, step: "secret-scan" };
      }
      // 3. Read file — file-not-found gate
      let raw: string;
      try { raw = await readFile(args.existingConfigPath, "utf8"); }
      catch (err) { return { outcome: "file-not-found", reason: (err as Error).message }; }
      // 4. parseDocument, find agent, mutate mcpServers seq
      const doc = parseDocument(raw);
      const agentsSeq = doc.get("agents") as YAMLSeq | undefined;
      if (!agentsSeq) return { outcome: "not-found", reason: "no agents seq" };
      let agentMap: YAMLMap | null = null;
      for (const item of agentsSeq.items) {
        if (item instanceof YAMLMap && item.get("name") === args.agentName) { agentMap = item; break; }
      }
      if (!agentMap) return { outcome: "not-found", reason: `agent ${args.agentName} missing` };
      let mcpSeq = agentMap.get("mcpServers") as YAMLSeq | undefined;
      if (!mcpSeq) {
        mcpSeq = new YAMLSeq();
        agentMap.set("mcpServers", mcpSeq);
      }
      // 5. Find existing element by .name (YAMLMap) or string scalar
      let existingIdx = -1;
      for (let i = 0; i < mcpSeq.items.length; i++) {
        const el = mcpSeq.items[i];
        if (el instanceof YAMLMap && el.get("name") === args.entry.name) { existingIdx = i; break; }
        if (typeof el === "string" && el === args.entry.name) { existingIdx = i; break; }
      }
      if (args.op === "remove") {
        if (existingIdx === -1) return { outcome: "no-op", reason: "entry not present" };
        mcpSeq.items.splice(existingIdx, 1);
      } else {  // add
        const newMap = new YAMLMap();
        newMap.set("name", args.entry.name);
        newMap.set("command", args.entry.command);
        const argsSeq = new YAMLSeq(); for (const a of args.entry.args) argsSeq.add(a); newMap.set("args", argsSeq);
        const envMap = new YAMLMap(); for (const [k, v] of Object.entries(args.entry.env)) envMap.set(k, v); newMap.set("env", envMap);
        if (args.entry.optional) newMap.set("optional", true);
        if (existingIdx >= 0) {
          // Idempotency: compare serialized bytes before replace
          const existing = mcpSeq.items[existingIdx];
          const oldBytes = existing instanceof YAMLMap ? existing.toString() : String(existing);
          if (oldBytes === newMap.toString()) return { outcome: "no-op", reason: "entry byte-identical" };
          mcpSeq.items[existingIdx] = newMap;
        } else {
          mcpSeq.add(newMap);
        }
      }
      // 6. Serialize + atomic tmp+rename + sha witness (COPY from updateAgentModel)
      const newText = doc.toString({ lineWidth: 0 });
      const pid = args.pid ?? process.pid;
      const destDir = dirname(args.existingConfigPath);
      const tmpPath = join(destDir, `.clawcode.yaml.${pid}.${Date.now()}.tmp`);
      await writerFs.writeFile(tmpPath, newText, "utf8");
    ```

    ```typescript
      configs: ResolvedAgentConfig[];

    export async function handleMarketplaceListPluginsIpc(
      deps: MarketplacePluginsIpcDeps, params: Record<string, unknown>,
    ): Promise<Readonly<{ agent: string; installed: readonly string[]; available: readonly ClawhubPluginListItem[] }>> {
      const agentName = validateStringParam(params, "agent");
      const cfg = deps.configs.find((c) => c.name === agentName);
      if (!cfg) throw new ManagerError(`Agent '${agentName}' not found`);

      const cacheKey = { endpoint: "plugins", query: "", cursor: "" };
      let response: ClawhubPluginsResponse;
      const hit = deps.cache.get(cacheKey);
      if (hit.kind === "hit") response = hit.value;
      else if (hit.kind === "rate-limited") return Object.freeze({ agent: agentName, installed: [], available: [] });
      else {
        response = await fetchClawhubPlugins({
          baseUrl: deps.clawhubBaseUrl, authToken: deps.clawhubAuthToken,
        });
        deps.cache.set(cacheKey, response);
      }
      const installed = cfg.mcpServers.map((m: any) => m.name ?? m);
      const available = response.items.filter((p) => !installed.includes(p.name));
      return Object.freeze({ agent: agentName, installed: Object.freeze(installed), available: Object.freeze(available) });
    }

    export async function handleMarketplaceInstallPluginIpc(
      deps: MarketplacePluginsIpcDeps, params: Record<string, unknown>,
    ): Promise<PluginInstallOutcome> {
      const agentName = validateStringParam(params, "agent");
      const pluginName = validateStringParam(params, "plugin");
      const configInputs = (params.configInputs ?? {}) as Record<string, string>;
      const cfg = deps.configs.find((c) => c.name === agentName);
      if (!cfg) throw new ManagerError(`Agent '${agentName}' not found`);
      try {
        const resp = await fetchClawhubPlugins({ baseUrl: deps.clawhubBaseUrl, authToken: deps.clawhubAuthToken });
        const item = resp.items.find((p) => p.name === pluginName);
        if (!item) return { kind: "not-in-catalog", plugin: pluginName };
        const manifest = await downloadClawhubPluginManifest({ manifestUrl: item.manifestUrl, authToken: deps.clawhubAuthToken });
        return await installClawhubPlugin({ manifest, agentName, configPath: deps.configPath, configInputs });
      } catch (err) {
        if (err instanceof ClawhubRateLimitedError) return { kind: "rate-limited", plugin: pluginName, retryAfterMs: err.retryAfterMs };
        if (err instanceof ClawhubAuthRequiredError) return { kind: "auth-required", plugin: pluginName, reason: err.message };
        if (err instanceof ClawhubManifestInvalidError) return { kind: "manifest-invalid", plugin: pluginName, reason: err.message };
        throw err;
      }
    }
    ```
