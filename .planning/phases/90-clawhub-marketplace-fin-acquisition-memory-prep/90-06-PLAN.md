    import { execa } from "execa";
    import type { Logger } from "pino";

    export type OpItemCandidate = Readonly<{
      uuid: string; title: string; category: string; tags?: readonly string[];
    }>;

    export type OpRewriteProposal = Readonly<{
      uri: string;
      confidence: "substring" | "levenshtein";
      itemTitle: string;
      distance?: number;
    }>;

    export function levenshtein(a: string, b: string): number {
      if (a.length === 0) return b.length;
      if (b.length === 0) return a.length;
      const m: number[][] = Array.from({ length: a.length + 1 }, () => new Array(b.length + 1).fill(0));
      for (let i = 0; i <= a.length; i++) m[i][0] = i;
      for (let j = 0; j <= b.length; j++) m[0][j] = j;
      for (let i = 1; i <= a.length; i++) {
        for (let j = 1; j <= b.length; j++) {
          const cost = a[i - 1] === b[j - 1] ? 0 : 1;
          m[i][j] = Math.min(m[i - 1][j] + 1, m[i][j - 1] + 1, m[i - 1][j - 1] + cost);
        }
      }
      return m[a.length][b.length];
    }

    export async function listOpItems(deps?: Readonly<{ run?: typeof execa; log?: Logger }>): Promise<readonly OpItemCandidate[]> {
      const run = deps?.run ?? execa;
      try {
        const { stdout } = await run("op", ["item", "list", "--categories=Credential,API", "--format=json"]);
        const arr = JSON.parse(stdout) as unknown[];
        return Object.freeze(
          arr
            .filter((e: any) => typeof e?.id === "string" && typeof e?.title === "string")
            .map((e: any) => Object.freeze({
              uuid: e.id, title: e.title, category: e.category ?? "",
              tags: Array.isArray(e.tags) ? Object.freeze([...e.tags]) : undefined,
            })),
        );
      } catch (err) {
        deps?.log?.debug({ err }, "op item list failed (1Password CLI unavailable); returning empty candidate list");
        return Object.freeze([]);
      }
    }

    /**
     * Field-name to 1Password field default. Maps ClawCode-common env var names to
     * typical 1Password Credential item field names.
     */
    function opFieldFor(envName: string): string {
      const lower = envName.toLowerCase();
      if (lower.endsWith("_password") || lower === "password") return "password";
      if (lower.endsWith("_user") || lower === "user" || lower.endsWith("_username")) return "username";
      if (lower.endsWith("_host") || lower === "host") return "hostname";
      if (lower.endsWith("_port") || lower === "port") return "port";
      if (lower.endsWith("_token") || lower === "token") return "credential";
      if (lower.endsWith("_key") || lower === "key") return "credential";
      return "credential";
    }

    export function proposeOpUri(
      fieldName: string,
      fieldLabel: string,
      items: readonly OpItemCandidate[],
      maxDistance: number = 3,
    ): OpRewriteProposal | null {
      const target = fieldLabel.toLowerCase();
      // Pass 1: substring match (title contains or is contained by label)
      for (const item of items) {
        const t = item.title.toLowerCase();
        if (t.includes(target) || target.includes(t)) {
          return Object.freeze({
            uri: `op://clawdbot/${item.title}/${opFieldFor(fieldName)}`,
            confidence: "substring" as const,
            itemTitle: item.title,
          });
        }
      }
      // Pass 2: Levenshtein ≤ maxDistance
      let best: { item: OpItemCandidate; distance: number } | null = null;
      for (const item of items) {
        const d = levenshtein(item.title.toLowerCase(), target);
        if (d <= maxDistance && (!best || d < best.distance)) {
          best = { item, distance: d };
        }
      }
      if (best) {
        return Object.freeze({
          uri: `op://clawdbot/${best.item.title}/${opFieldFor(fieldName)}`,
          confidence: "levenshtein" as const,
          itemTitle: best.item.title,
          distance: best.distance,
        });
      }
      return null;
    }

    ```typescript
    import { execa } from "execa";
    import type { Logger } from "pino";

    /** ClawHub's registered GitHub App device-code client_id (hardcoded per D-02; public by design). */
    const CLAWHUB_GITHUB_CLIENT_ID = "Iv1.clawhub-public-client";  // placeholder — replace with real when known
    const DEVICE_CODE_URL = "https://github.com/login/device/code";
    const TOKEN_URL = "https://github.com/login/oauth/access_token";

    export type DeviceCodeInit = Readonly<{
      user_code: string;
      verification_uri: string;
      device_code: string;
      interval: number;  // seconds
      expires_at: number;  // ms epoch
    }>;

    export class OAuthExpiredError extends Error {
      constructor() { super("device-code expired"); this.name = "OAuthExpiredError"; }
    }
    export class OAuthAccessDeniedError extends Error {
      constructor() { super("user denied access"); this.name = "OAuthAccessDeniedError"; }
    }

    export type GithubOauthDeps = Readonly<{ fetch?: typeof globalThis.fetch; now?: () => number; sleep?: (ms: number) => Promise<void>; run?: typeof execa; }>;

    export async function initiateDeviceCodeFlow(deps?: GithubOauthDeps): Promise<DeviceCodeInit> {
      const f = deps?.fetch ?? globalThis.fetch;
      const now = (deps?.now ?? Date.now)();
      const res = await f(DEVICE_CODE_URL, {
        method: "POST",
        headers: { "Accept": "application/json", "Content-Type": "application/x-www-form-urlencoded" },
        body: `client_id=${encodeURIComponent(CLAWHUB_GITHUB_CLIENT_ID)}&scope=read:user`,
      });
      if (!res.ok) throw new Error(`github device-code init failed: ${res.status}`);
      const body = await res.json() as any;
      return Object.freeze({
        user_code: body.user_code, verification_uri: body.verification_uri,
        device_code: body.device_code, interval: Number(body.interval ?? 5),
        expires_at: now + Number(body.expires_in ?? 900) * 1000,
      });
    }

    export async function pollForAccessToken(
      init: DeviceCodeInit, deps?: GithubOauthDeps,
    ): Promise<string> {
      const f = deps?.fetch ?? globalThis.fetch;
      const sleep = deps?.sleep ?? ((ms: number) => new Promise((r) => setTimeout(r, ms)));
      const now = () => (deps?.now ?? Date.now)();
      let interval = init.interval;
      while (now() < init.expires_at) {
        await sleep(interval * 1000);
        const res = await f(TOKEN_URL, {
          method: "POST",
          headers: { "Accept": "application/json", "Content-Type": "application/x-www-form-urlencoded" },
          body: `client_id=${encodeURIComponent(CLAWHUB_GITHUB_CLIENT_ID)}&device_code=${encodeURIComponent(init.device_code)}&grant_type=urn:ietf:params:oauth:grant-type:device_code`,
        });
        const body = await res.json() as any;
        if (body.access_token) return body.access_token;
        if (body.error === "authorization_pending") continue;
        if (body.error === "slow_down") { interval += 5; continue; }
        if (body.error === "access_denied") throw new OAuthAccessDeniedError();
        if (body.error === "expired_token") throw new OAuthExpiredError();
        // Unknown error — log and continue until expiry
      }
      throw new OAuthExpiredError();
    }

    export async function storeTokenTo1Password(
      token: string,
      label: string = "ClawHub Token",
      deps?: GithubOauthDeps,
    ): Promise<void> {
      const run = deps?.run ?? execa;
      await run("op", [
        "item", "create",
        "--category=Credential",
        `--title=${label}`,
        `credential=${token}`,
      ]);
    }
    ```

          .setRequired(true);

        else if (field.sensitive) input.setPlaceholder("op:// reference preferred");

    export function parseModalSubmit(

    ): Record<string, string> {
      const out: Record<string, string> = {};

        try { out[f.name] = submit.fields.getTextInputValue(`field:${f.name}`); }

      }

    import { listOpItems, proposeOpUri, type OpRewriteProposal } from "./op-rewrite.js";
    import type { ClawhubPluginManifest } from "./clawhub-client.js";

    export type OpRewriteCandidate = Readonly<{
      fieldName: string; fieldLabel: string;
      typedValue: string;
      proposal: OpRewriteProposal;
    }>;

    /**
     * Probe 1Password and, for any sensitive field where (1) the typed value isn't
     * already an op:// ref AND (2) a fuzzy match exists, return a candidate op://
     * proposal. Operator confirms via Discord button.
     */
    export async function buildOpRewriteCandidates(
      manifest: ClawhubPluginManifest, configInputs: Readonly<Record<string,string>>,
    ): Promise<readonly OpRewriteCandidate[]> {
      const items = await listOpItems();
      if (items.length === 0) return Object.freeze([]);
      const out: OpRewriteCandidate[] = [];
      for (const f of manifest.config?.fields ?? []) {
        if (!f.sensitive) continue;
        const v = configInputs[f.name] ?? "";
        if (v.startsWith("op://")) continue;
        const prop = proposeOpUri(f.name, f.label, items);
        if (prop) out.push(Object.freeze({ fieldName: f.name, fieldLabel: f.label, typedValue: v, proposal: prop }));
      }
      return Object.freeze(out);
    }

    export async function handleClawhubOauthPollIpc(
      deps: {log: Logger}, params: Record<string, unknown>,
    ): Promise<Readonly<{stored:boolean; message:string}>> {
      const device_code = validateStringParam(params, "device_code");
      const poll_interval_s = Number(params.poll_interval_s ?? 5);
