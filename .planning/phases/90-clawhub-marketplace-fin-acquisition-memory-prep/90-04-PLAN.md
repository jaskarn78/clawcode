    ```typescript
    import { createReadStream } from "node:fs";
    import { mkdir, writeFile } from "node:fs/promises";
    import { join } from "node:path";
    import { execa } from "execa";
    import { readFileSync } from "node:fs";

    const PKG = JSON.parse(

    ) as { version: string };
    const USER_AGENT = `ClawCode/${PKG.version} (clawcode-marketplace)`;

    export type ClawhubSkillListItem = Readonly<{
      id: string; name: string; description: string; version: string;
      author: string; downloadUrl: string; manifestUrl?: string;
      rating?: number; downloadCount?: number;
      category?: string; tags?: readonly string[];
      createdAt?: string; updatedAt?: string;
    }>;

    export type ClawhubSkillsResponse = Readonly<{
      items: readonly ClawhubSkillListItem[];
      nextCursor: string | null;
    }>;

    export class ClawhubRateLimitedError extends Error {
      constructor(
        public readonly retryAfterMs: number,
        message: string,
      ) { super(message); this.name = "ClawhubRateLimitedError"; }
    }
    export class ClawhubAuthRequiredError extends Error {
      constructor(message: string) { super(message); this.name = "ClawhubAuthRequiredError"; }
    }
    export class ClawhubManifestInvalidError extends Error {
      constructor(message: string) { super(message); this.name = "ClawhubManifestInvalidError"; }
    }

    export type ClawhubClientDeps = Readonly<{
      fetch: typeof globalThis.fetch;
      // DI hooks for tests; production uses the globals
    }>;

    export async function fetchClawhubSkills(
      args: Readonly<{
        baseUrl: string;
        query?: string;
        cursor?: string;
        authToken?: string;
        deps?: ClawhubClientDeps;
      }>,
    ): Promise<ClawhubSkillsResponse> {
      const f = args.deps?.fetch ?? globalThis.fetch;
      const url = new URL("/api/v1/skills", args.baseUrl);
      if (args.query) url.searchParams.set("q", args.query);
      if (args.cursor) url.searchParams.set("cursor", args.cursor);

      const headers: Record<string, string> = { "User-Agent": USER_AGENT, "Accept": "application/json" };
      if (args.authToken) headers["Authorization"] = `Bearer ${args.authToken}`;

      const res = await f(url.toString(), { headers });
      if (res.status === 429) {
        const ra = res.headers.get("Retry-After");
        const retryAfterMs = ra ? Number(ra) * 1000 : 60_000;
        throw new ClawhubRateLimitedError(retryAfterMs, `clawhub: rate-limited (retry in ${retryAfterMs}ms)`);
      }
      if (res.status === 401 || res.status === 403) {
        throw new ClawhubAuthRequiredError(`clawhub: auth required (status ${res.status})`);
      }
      if (!res.ok) {
        throw new Error(`clawhub: ${res.status} ${res.statusText}`);
      }
      const body = (await res.json()) as ClawhubSkillsResponse;
      if (!Array.isArray(body.items)) {
        throw new ClawhubManifestInvalidError(`clawhub: malformed response (no items[])`);
      }
      return body;
    }

    export async function downloadClawhubSkill(
      args: Readonly<{
        downloadUrl: string;
        stagingDir: string;
        authToken?: string;
        deps?: ClawhubClientDeps;
      }>,
    ): Promise<{ extractedDir: string; files: readonly string[] }> {
      const f = args.deps?.fetch ?? globalThis.fetch;
      const headers: Record<string, string> = { "User-Agent": USER_AGENT };
      if (args.authToken) headers["Authorization"] = `Bearer ${args.authToken}`;

      const res = await f(args.downloadUrl, { headers });
      if (res.status === 429) {
        const ra = res.headers.get("Retry-After");
        throw new ClawhubRateLimitedError(ra ? Number(ra)*1000 : 60_000, `clawhub download: rate-limited`);
      }
      if (res.status === 401 || res.status === 403) {
        throw new ClawhubAuthRequiredError(`clawhub download: auth required (${res.status})`);
      }
      if (!res.ok) throw new Error(`clawhub download: ${res.status} ${res.statusText}`);

      await mkdir(args.stagingDir, { recursive: true });
      const tarPath = join(args.stagingDir, "skill.tar.gz");
      const bytes = Buffer.from(await res.arrayBuffer());
      await writeFile(tarPath, bytes);

      const extractedDir = join(args.stagingDir, "extracted");
      await mkdir(extractedDir, { recursive: true });
      await execa("tar", ["-xzf", tarPath, "-C", extractedDir]);

      const { stdout } = await execa("find", [extractedDir, "-type", "f"]);
      const files = stdout.split("\n").filter(Boolean);
      return { extractedDir, files };
    }
    ```

    **Step 2: Create src/marketplace/clawhub-cache.ts — TTL cache.**
    ```typescript
    export type ClawhubCacheKey = Readonly<{
      endpoint: string; query?: string; cursor?: string;
    }>;

    export type ClawhubCacheHit<T> =
      | { kind: "hit"; value: T }
      | { kind: "rate-limited"; retryAfterMs: number }
      | { kind: "miss" };

    export type ClawhubCache<T> = Readonly<{
      get(key: ClawhubCacheKey): ClawhubCacheHit<T>;
      set(key: ClawhubCacheKey, value: T): void;
      setNegative(key: ClawhubCacheKey, retryAfterMs: number): void;
    }>;

    export function createClawhubCache<T>(
      ttlMs: number,
      now: () => number = () => Date.now(),
    ): ClawhubCache<T> {
      type Entry = { expiresAt: number; value: T | null; negativeUntil: number | null };
      const map = new Map<string, Entry>();
      const keyOf = (k: ClawhubCacheKey) => `${k.endpoint}|${k.query ?? ""}|${k.cursor ?? ""}`;

      return Object.freeze({
        get(k: ClawhubCacheKey): ClawhubCacheHit<T> {
          const e = map.get(keyOf(k));
          const t = now();
          if (!e) return { kind: "miss" };
          if (e.negativeUntil !== null && t < e.negativeUntil) {
            return { kind: "rate-limited", retryAfterMs: e.negativeUntil - t };
          }
          if (e.value !== null && t < e.expiresAt) {
            return { kind: "hit", value: e.value };
          }
          map.delete(keyOf(k));
          return { kind: "miss" };
        },
        set(k: ClawhubCacheKey, value: T): void {
          map.set(keyOf(k), { expiresAt: now() + ttlMs, value, negativeUntil: null });
        },
        setNegative(k: ClawhubCacheKey, retryAfterMs: number): void {
          map.set(keyOf(k), { expiresAt: 0, value: null, negativeUntil: now() + retryAfterMs });
        },
      });
    }
    ```

    ```typescript
    async function installClawhubSkill(args: {
      entry: MarketplaceEntry;  // source.kind === "clawhub"
      agentName: string;
      skillsTargetDir: string;
      clawcodeYamlPath: string;
      ledgerPath: string;
      force: boolean;
    }): Promise<SkillInstallOutcome> {
      const stagingRoot = expandHome("~/.clawcode/manager/clawhub-staging");
      const stagingDir = join(stagingRoot, nanoid());
      const src = args.entry.source as { kind:"clawhub"; downloadUrl:string; authToken?:string };
      try {
        // Step 1: download + extract
        let extracted;
        try {
          extracted = await downloadClawhubSkill({
            downloadUrl: src.downloadUrl,
            stagingDir,
            authToken: src.authToken,
          });
        } catch (err) {
          if (err instanceof ClawhubRateLimitedError) {
            return { kind: "rate-limited", skill: args.entry.name, retryAfterMs: err.retryAfterMs };
          }
          if (err instanceof ClawhubAuthRequiredError) {
            return { kind: "auth-required", skill: args.entry.name, reason: err.message };
          }
          return { kind: "copy-failed", skill: args.entry.name, reason: err instanceof Error ? err.message : String(err) };
        }
        // Step 2: manifest validation — SKILL.md must exist
        const skillMd = join(extracted.extractedDir, "SKILL.md");
        try { await stat(skillMd); } catch {
          return { kind: "manifest-invalid", skill: args.entry.name, reason: "SKILL.md missing from archive" };
        }
        // Step 3: secret scan
        const scan = await scanSkillSecrets(extracted.extractedDir);
        if (!scan.pass) {
          return { kind: "blocked-secret-scan", skill: args.entry.name,
                   offender: `${scan.offender?.path}:${scan.offender?.line}`,
                   reason: scan.offender?.reason ?? "" };
        }
        // Step 4: scope gate (ClawHub entries default to fleet — canLinkSkillToAgent handles the rest)
        const canLink = canLinkSkillToAgent({ name: args.entry.name, scope: args.entry.category }, args.agentName, { force: args.force });
        if (!canLink) {
          return { kind: "rejected-scope", skill: args.entry.name, skillScope: args.entry.category, agentScope: "(per-agent)" };
        }
        // Step 5: normalize frontmatter
        await normalizeSkillFrontmatter(skillMd);
        // Step 6: idempotency via content hash + ledger
        const sourceHash = await computeSkillContentHash(extracted.extractedDir);
        const latest = latestStatusBySkill(await readSkillRows(args.ledgerPath))[args.entry.name];
        if (latest?.status === "migrated" && latest.source_hash === sourceHash) {
          return { kind: "already-installed", skill: args.entry.name, reason: `hash match: ${sourceHash.slice(0,8)}...` };
        }
        // Step 7: copy + persist + ledger
        const targetDir = join(args.skillsTargetDir, args.entry.name);
        try {
          await copySkillDirectory(extracted.extractedDir, targetDir, {});
        } catch (err) {
          return { kind: "copy-failed", skill: args.entry.name, reason: err instanceof Error ? err.message : String(err) };
        }
        const targetHash = await computeSkillContentHash(targetDir);
        try {
          await updateAgentSkills({
            existingConfigPath: args.clawcodeYamlPath,
            agentName: args.agentName,
            skillName: args.entry.name,
            op: "add",
          });

            source_hash: sourceHash, target_hash: targetHash,

            notes: `persisted=false; ${err instanceof Error ? err.message : String(err)}`,
          });
          return { kind: "installed-persist-failed", skill: args.entry.name, targetPath: targetDir, targetHash, persist_error: err instanceof Error ? err.message : String(err) };

          source_hash: sourceHash, target_hash: targetHash,

        });
        return { kind: "installed", skill: args.entry.name, targetPath: targetDir, targetHash };

        try { await rm(stagingDir, { recursive: true, force: true }); } catch { /* best effort */ }
      }
    ```
