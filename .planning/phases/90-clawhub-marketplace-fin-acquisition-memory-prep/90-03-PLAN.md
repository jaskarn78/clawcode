    import { writeFile, rename, mkdir, unlink } from "node:fs/promises";
    import { dirname, join } from "node:path";
    import { nanoid } from "nanoid";
    import type { Logger } from "pino";
    import type { summarizeWithHaiku } from "../manager/summarize-with-haiku.js";
    import type { Turn } from "./conversation-types.js";

    const DEFAULT_FLUSH_INTERVAL_MS = 15 * 60 * 1000;
    const FLUSH_PROMPT = "Summarize the most important decisions, tasks in progress, and standing rules from this session segment. Under 300 words, markdown sections, no meta-commentary.";

    export type MemoryFlushDeps = Readonly<{
      workspacePath: string;
      agentName: string;
      intervalMs?: number;
      getTurnsSince: (sinceTs: number) => readonly Turn[];
      summarize: typeof summarizeWithHaiku;
      log: Logger;
      now?: () => number;
    }>;

    export function meaningfulTurnsSince(turns: readonly Turn[]): boolean {
      let hasUser = false, hasAssistantSignal = false;
      for (const t of turns) {
        if (t.role === "user") hasUser = true;
        if (t.role === "assistant") {
          const hasText = typeof (t as any).text === "string" && (t as any).text.length >= 200;
          const hasToolCall = Array.isArray((t as any).toolCalls) && (t as any).toolCalls.length > 0;
          if (hasText || hasToolCall) hasAssistantSignal = true;
        }
        if (hasUser && hasAssistantSignal) return true;
      }
      return false;
    }

    async function atomicWriteFile(dstPath: string, content: string): Promise<void> {
      await mkdir(dirname(dstPath), { recursive: true });
      const tmp = `${dstPath}.${process.pid}.${Date.now()}.${nanoid(4)}.tmp`;
      await writeFile(tmp, content, "utf8");
      try { await rename(tmp, dstPath); }
      catch (err) { try { await unlink(tmp); } catch {} throw err; }
    }

    export class MemoryFlushTimer {
      private handle: NodeJS.Timeout | null = null;
      private lastFlushAt: number;
      private inFlight: Promise<string | null> | null = null;
      private readonly intervalMs: number;

      constructor(private readonly deps: MemoryFlushDeps) {
        this.intervalMs = deps.intervalMs ?? DEFAULT_FLUSH_INTERVAL_MS;
        this.lastFlushAt = (deps.now ?? Date.now)();
      }

      start(): void {
        if (this.handle) return;
        this.handle = setInterval(() => {
          void this.flushNow().catch((err) =>
            this.deps.log.warn({ err, agent: this.deps.agentName }, "periodic flush failed"),
          );
        }, this.intervalMs);
      }

      stop(): void {
        if (this.handle) { clearInterval(this.handle); this.handle = null; }
      }

      async flushNow(): Promise<string | null> {
        if (this.inFlight) return this.inFlight;
        this.inFlight = (async () => {
          try {
            const now = (this.deps.now ?? Date.now)();
            const turns = this.deps.getTurnsSince(this.lastFlushAt);
            if (!meaningfulTurnsSince(turns)) {
              this.deps.log.debug({ agent: this.deps.agentName, turns: turns.length }, "flush skipped (not meaningful)");
              this.lastFlushAt = now;
              return null;
            }
            const turnText = turns.map((t) => `[${t.role}] ${(t as any).text ?? ""}`).join("\n").slice(0, 20_000);
            const summary = await this.deps.summarize(`${FLUSH_PROMPT}\n\nSession segment:\n${turnText}`, { timeoutMs: 10_000 });
            const iso = new Date(now).toISOString();
            const date = iso.slice(0, 10);
            const hhmm = iso.slice(11, 16).replace(":", "");
            const path = join(this.deps.workspacePath, "memory", `${date}-${hhmm}.md`);
            const body = `---\nflushed_at: ${iso}\nagent: ${this.deps.agentName}\nturn_count: ${turns.length}\n---\n\n# Session Flush ${date} ${iso.slice(11,16)}\n\n${summary}\n`;
            await atomicWriteFile(path, body);
            this.lastFlushAt = now;
            this.deps.log.info({ agent: this.deps.agentName, path, turns: turns.length }, "session flush written");
            return path;
          } finally {
            this.inFlight = null;
          }
        })();
        return this.inFlight;
      }
    }

    import { join } from "node:path";
    import { nanoid } from "nanoid";
    import type { Logger } from "pino";
    import { atomicWriteFile } from "./memory-flush.js";  // export from Task 1

    export const MEMORY_CUE_REGEX = /(remember( this)?|keep this (in )?(long[- ]?term )?memory|standing rule|don'?t forget|note for later|save to memory)/i;

    export function detectCue(text: string): { match: boolean; captured: string | null } {
      const m = MEMORY_CUE_REGEX.exec(text);
      return { match: m !== null, captured: m?.[0] ?? null };
    }

    /** Extract the containing paragraph (max 3 sentences) around the cue. */
    export function extractCueContext(fullText: string, maxSentences: number = 3): string {
      // Split into sentences by ., ?, !
      const parts = fullText.split(/(?<=[.?!])\s+/);
      // Find first part matching cue
      const idx = parts.findIndex((p) => MEMORY_CUE_REGEX.test(p));
      if (idx === -1) return fullText.slice(0, 500);
      const start = Math.max(0, idx - 1);
      return parts.slice(start, start + maxSentences).join(" ").trim();
    }

    export async function writeCueMemory(args: Readonly<{
      workspacePath: string;
      cue: string;
      context: string;
      turnIso: string;
      messageLink?: string;
      log: Logger;
    }>): Promise<string> {
      const date = args.turnIso.slice(0, 10);
      const suffix = nanoid(4);
      const path = join(args.workspacePath, "memory", `${date}-remember-${suffix}.md`);
      const body = `---\ntype: cue\ncaptured_at: ${args.turnIso}\ncue: ${JSON.stringify(args.cue)}\n${args.messageLink ? `discord_link: ${args.messageLink}\n` : ""}---\n\n## Standing note\n\n${args.context}\n`;
      await atomicWriteFile(path, body);
      args.log.info({ path, cue: args.cue }, "cue memory written");
      return path;
    }
