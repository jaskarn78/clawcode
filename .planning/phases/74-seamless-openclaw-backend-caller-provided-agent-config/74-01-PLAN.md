    import crypto from "node:crypto";
    import type {
      CallerIdentity,
      ChatCompletionRequest,
      ClaudeToolDef,
      ClaudeToolChoice,
      ClaudeToolResultBlock,
      Tier,
    } from "./types.js";
    import { OPENCLAW_PREFIX } from "./types.js";
    import type { ApiKeyRow } from "./keys.js";

    const SLUG_RE = /^[a-z0-9][a-z0-9_-]{0,63}$/i;
    const VALID_TIERS: ReadonlySet<string> = new Set(["sonnet", "opus", "haiku"]);

    export function sha256Hex(input: string): string {
      return crypto.createHash("sha256").update(input, "utf8").digest("hex");
    }

    export function extractCallerIdentity(
      body: ChatCompletionRequest,
      row: ApiKeyRow,
      knownAgents: ReadonlyArray<string>,
      translatedTools: ClaudeToolDef[] | null,
      translatedToolChoice: ClaudeToolChoice | null,
      translatedToolResults: ClaudeToolResultBlock[],
    ): CallerIdentity | { readonly error: "unknown_model" | "malformed_caller" } {
      // Fast path: literal agent match → native.
      if (knownAgents.includes(body.model)) {
        return { kind: "clawcode-native", agentName: body.model };
      }
      if (!body.model.startsWith(OPENCLAW_PREFIX)) {
        return { error: "unknown_model" };
      }
      // scope='all' is required for the template route — pinned keys cannot
      // impersonate an OpenClaw-side caller (defense-in-depth).
      if (row.scope !== "all") {
        return { error: "malformed_caller" };
      }
      const rest = body.model.slice(OPENCLAW_PREFIX.length);
      if (rest.length === 0) return { error: "malformed_caller" };
      const parts = rest.split(":");
      if (parts.length > 2) return { error: "malformed_caller" };
      const [slugRaw, tierRaw] = parts;
      if (!slugRaw || !SLUG_RE.test(slugRaw)) return { error: "malformed_caller" };
      const tier: Tier = tierRaw === undefined ? "sonnet" : (VALID_TIERS.has(tierRaw) ? (tierRaw as Tier) : (null as never));
      if (tierRaw !== undefined && !VALID_TIERS.has(tierRaw)) return { error: "malformed_caller" };
      // Extract SOUL from first system message (role "system" or "developer"; OpenClaw uses "system")
      const firstSystem = body.messages.find((m) => m.role === "system");
      let soulPrompt = "";
      if (firstSystem) {
        const c = firstSystem.content as unknown;
        if (typeof c === "string") {
          soulPrompt = c;
        } else if (Array.isArray(c)) {
          const parts: string[] = [];
          for (const p of c as Array<{ type?: string; text?: unknown }>) {
            if (p && p.type === "text" && typeof p.text === "string") parts.push(p.text);
          }
          soulPrompt = parts.join("\n\n");
        }
      }
      const soulFp = sha256Hex(soulPrompt).slice(0, 16);
      return {
        kind: "openclaw-template",
        callerSlug: slugRaw,
        tier,
        soulPrompt,
        soulFp,
        tools: translatedTools,
        toolChoice: translatedToolChoice,
        toolResults: translatedToolResults,
      };
    }

    import type { SessionHandle } from "../manager/session-adapter.js";

    export interface TransientCacheKeyParts {
      readonly keyHash: string;
      readonly callerSlug: string;
      readonly soulFp: string;
      readonly tier: string;
    }

    export function makeTransientCacheKey(parts: TransientCacheKeyParts): string {
      return `${parts.keyHash}::${parts.callerSlug}::${parts.soulFp}::${parts.tier}`;
    }

    export interface TransientCacheOptions {
      readonly maxSize: number;                           // LRU cap (default 32)
      readonly ttlMs: number;                             // 30 * 60 * 1000 per D-03
      readonly now?: () => number;                        // dependency-injected clock for tests
      readonly log?: { warn: (obj: Record<string, unknown>, msg?: string) => void };
    }

    interface CacheEntry {
      readonly handle: SessionHandle;
      lastAccessMs: number;
    }

    /**
     * LRU + TTL cache for per-caller persistent session handles. Evicted entries
     * ALWAYS call handle.close() (fire-and-forget — handle owns the SDK subprocess
     * shutdown; errors are caught + logged so cache invariants hold).
     */
    export class TransientSessionCache {
      private readonly map = new Map<string, CacheEntry>(); // insertion-order = LRU order
      private readonly maxSize: number;
      private readonly ttlMs: number;
      private readonly now: () => number;
      private readonly log?: TransientCacheOptions["log"];
      private closed = false;

      constructor(opts: TransientCacheOptions) {
        this.maxSize = Math.max(1, opts.maxSize);
        this.ttlMs = Math.max(0, opts.ttlMs);
        this.now = opts.now ?? (() => Date.now());
        this.log = opts.log;
      }

      size(): number { return this.map.size; }

      get(key: string): SessionHandle | undefined {
        if (this.closed) return undefined;
        const entry = this.map.get(key);
        if (!entry) return undefined;
        if (this.ttlMs > 0 && this.now() - entry.lastAccessMs > this.ttlMs) {
          this.evictEntry(key, entry, "ttl");
          return undefined;
        }
        // LRU touch: delete + re-insert to move to tail.
        this.map.delete(key);
        entry.lastAccessMs = this.now();
        this.map.set(key, entry);
        return entry.handle;
      }

      set(key: string, handle: SessionHandle): void {
        if (this.closed) { handle.close().catch(() => {}); return; }
        // If already present, close the old handle before replacing (SOUL change path).
        const existing = this.map.get(key);
        if (existing) this.evictEntry(key, existing, "replace");
        this.map.set(key, { handle, lastAccessMs: this.now() });
        while (this.map.size > this.maxSize) {
          const oldestKey = this.map.keys().next().value;
          if (oldestKey === undefined) break;
          const oldest = this.map.get(oldestKey)!;
          this.evictEntry(oldestKey, oldest, "lru");
        }
      }

      async closeAll(): Promise<void> {
        if (this.closed) return;
        this.closed = true;
        const entries = Array.from(this.map.values());
        this.map.clear();
        await Promise.allSettled(entries.map((e) => e.handle.close()));
      }

      private evictEntry(key: string, entry: CacheEntry, reason: "lru" | "ttl" | "replace"): void {
        this.map.delete(key);
        try {
