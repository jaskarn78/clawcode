    /**
     * Phase 56 Plan 01 — run READ-ONLY warmup queries on the 3 per-agent
     * SQLite DBs (memories.db, usage.db, traces.db) to prime the page cache
     * and prepared-statement plan cache. NO WRITES. Target: ≤ 200ms per
     * agent total under normal conditions (SSD).
     *
     * Throws if the agent has no MemoryStore registered (initMemory not
     * called yet) or if any warmup query fails.
     */
    async warmSqliteStores(name: string): Promise<{
      readonly memories_ms: number;
      readonly usage_ms: number;
      readonly traces_ms: number;
    }> {
      const store = this.memoryStores.get(name);
      if (!store) {
        throw new Error(`warmSqliteStores: no MemoryStore for agent '${name}'`);
      }
      const usageTracker = this.usageTrackers.get(name);
      const traceStore = this.traceStores.get(name);

      // memories.db — 3 READ queries
      const t0 = performance.now();
      const db = store.getDatabase();
      db.prepare("SELECT COUNT(*) AS n FROM memories").get();
      db.prepare("SELECT id, tier, importance FROM memories ORDER BY accessed_at DESC LIMIT 1").all();
      // sqlite-vec vec0 extension prime — empty-result KNN
      db.prepare("SELECT rowid FROM memory_vec WHERE embedding MATCH ? AND k = 1").all(
        new Uint8Array(new Float32Array(384).buffer),
      );
      const memories_ms = performance.now() - t0;

      // usage.db — 2 READ queries
      const t1 = performance.now();
      if (usageTracker) {
        const udb = usageTracker.getDatabase();
        udb.prepare("SELECT COUNT(*) AS n FROM usage_events").get();
        udb.prepare("SELECT session_id FROM usage_events WHERE timestamp > ? ORDER BY timestamp DESC LIMIT 1").all(
          new Date(Date.now() - 24 * 60 * 60 * 1000).toISOString(),
        );
      }
      const usage_ms = performance.now() - t1;

      // traces.db — 3 READ queries (includes join-path prime for span retention)
      const t2 = performance.now();
      if (traceStore) {
        const tdb = traceStore.getDatabase();
        tdb.prepare("SELECT COUNT(*) AS n FROM traces").get();
        tdb.prepare("SELECT COUNT(*) AS n FROM trace_spans").get();
        tdb.prepare("SELECT t.id FROM traces t LEFT JOIN trace_spans s ON s.trace_id = t.id LIMIT 1").all();
      }
      const traces_ms = performance.now() - t2;

      return Object.freeze({ memories_ms, usage_ms, traces_ms });
    }

    **Part B — Create `src/manager/warm-path-check.ts` (NEW):**

    ```typescript
    import type { EmbeddingService } from "../memory/embedder.js";

    /**
     * Phase 56 Plan 01 — composite warm-path readiness check.
     *
     * Aggregates 3 per-agent signals:
     *   1. SQLite + sqlite-vec warmup (via deps.sqliteWarm)
     *   2. Embedder.isReady() + one probe embed
     *   3. Session plumbing ready (placeholder — Plan 02 threads in real session)
     *
     * Wrapped in a WARM_PATH_TIMEOUT_MS (10s) overall budget. On timeout,
     * returns ready:false with errors=["timeout after 10000ms"].
     *
     * Never throws — always returns a WarmPathResult. Caller (Plan 02)
     * inspects result.ready to decide whether to mark the agent 'running'.
     */
    export const WARM_PATH_TIMEOUT_MS = 10_000;

    export type WarmPathDurations = {
      readonly sqlite: number;
      readonly embedder: number;
      readonly session: number;
    };

    export type WarmPathResult = {
      readonly ready: boolean;
      readonly durations_ms: WarmPathDurations;
      readonly total_ms: number;
      readonly errors: readonly string[];
    };

    export type WarmPathDeps = {
      readonly agent: string;
      readonly sqliteWarm: (agent: string) => Promise<{ memories_ms: number; usage_ms: number; traces_ms: number }>;
      readonly embedder: Pick<EmbeddingService, "isReady" | "embed">;
      /** Optional session probe — Plan 02 wires a real check; Plan 01 defaults to always-ready. */
      readonly sessionProbe?: () => Promise<void>;
      /** Override timeout for tests. Defaults to WARM_PATH_TIMEOUT_MS. */
      readonly timeoutMs?: number;
    };

    export async function runWarmPathCheck(deps: WarmPathDeps): Promise<WarmPathResult> {
      const timeoutMs = deps.timeoutMs ?? WARM_PATH_TIMEOUT_MS;
      const startedAt = performance.now();
      const errors: string[] = [];
      let sqliteMs = 0;
      let embedderMs = 0;
      let sessionMs = 0;

      const work = (async () => {
        // Step 1 — SQLite warmup
        try {
          const sqStart = performance.now();
          const r = await deps.sqliteWarm(deps.agent);
          sqliteMs = r.memories_ms + r.usage_ms + r.traces_ms;
          void (performance.now() - sqStart);
        } catch (e) {
          errors.push(`sqlite: ${(e as Error).message}`);
        }

        // Step 2 — Embedder readiness + single probe
        const embStart = performance.now();
        try {
          if (!deps.embedder.isReady()) {
            errors.push("embedder: not ready");
          } else {
            await deps.embedder.embed("warmup probe");
          }
        } catch (e) {
          errors.push(`embedder: ${(e as Error).message}`);
        }
        embedderMs = performance.now() - embStart;

        // Step 3 — Session probe (Plan 02 wires real value)
        const sessStart = performance.now();
        try {
          if (deps.sessionProbe) await deps.sessionProbe();
        } catch (e) {
          errors.push(`session: ${(e as Error).message}`);
        }
        sessionMs = performance.now() - sessStart;
      })();

      let timedOut = false;
      await Promise.race([
        work,
        new Promise<void>((resolve) =>
          setTimeout(() => {
            timedOut = true;
            resolve();
          }, timeoutMs),
        ),
      ]);

      if (timedOut) errors.push(`timeout after ${timeoutMs}ms`);

      const total_ms = performance.now() - startedAt;
      const ready = errors.length === 0;

      return Object.freeze({
        ready,
        durations_ms: Object.freeze({ sqlite: sqliteMs, embedder: embedderMs, session: sessionMs }),
        total_ms,
        errors: Object.freeze([...errors]) as readonly string[],
      });
    }
    ```
