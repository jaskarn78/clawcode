       * INVESTIGATION 2026-04-13: No pre-existing Phase 40 emitter was found in
       * the repo at the time of authoring this plan. This file creates a minimal
       * emitter scaffold; when Phase 40's emitter lands (or if this IS Phase 40's
       * emitter for a project where Phase 40 was never implemented), the cache
       * line below is the locked CONTEXT D-03 requirement.
       */
      import type { TraceStore } from "../performance/trace-store.js";
      import type { UsageTracker } from "./tracker.js";
      import type { WebhookManager } from "../discord/webhook-manager.js";
      import type { Logger } from "pino";

      export type BuildDailySummaryArgs = {
        readonly agent: string;
        readonly traceStore: TraceStore;
        readonly usageTracker: UsageTracker;
        readonly now: Date;   // injectable for tests
      };

      export type DailySummaryEmbed = {
        readonly title: string;
        readonly description: string;
      };

      export function buildDailySummaryEmbed(args: BuildDailySummaryArgs): DailySummaryEmbed {
        const { agent, traceStore, usageTracker, now } = args;
        const dayStr = now.toISOString().slice(0, 10);
        const usage = usageTracker.getDailyUsage(dayStr);
        const iso24hAgo = new Date(now.getTime() - 24 * 60 * 60 * 1000).toISOString();
        const cache = traceStore.getCacheTelemetry(agent, iso24hAgo);

        const lines: string[] = [];
        lines.push(`📊 Daily summary for ${agent}`);
        lines.push(`💵 Cost: $${(usage.cost_usd ?? 0).toFixed(2)} · ${usage.tokens_in ?? 0} in / ${usage.tokens_out ?? 0} out · ${usage.turns ?? 0} turns`);
        if (cache.totalTurns > 0) {
          lines.push(`💾 Cache: ${(cache.avgHitRate * 100).toFixed(1)}% over ${cache.totalTurns} turns`);
        }
        return Object.freeze({
          title: `Daily summary — ${dayStr}`,
          description: lines.join("\n"),
        });
      }

      export type EmitDailySummaryArgs = BuildDailySummaryArgs & {
        readonly webhookManager: WebhookManager;
        readonly log: Logger;
      };

      export async function emitDailySummary(args: EmitDailySummaryArgs): Promise<void> {
        try {
          const embed = buildDailySummaryEmbed(args);
          if (args.webhookManager.hasWebhook(args.agent)) {
            await args.webhookManager.send(args.agent, `**${embed.title}**\n${embed.description}`);
          } else {
            args.log.info({ agent: args.agent, embed }, "daily summary emitted (no webhook — dropped)");
          }
        } catch (err) {
          args.log.warn({ err, agent: args.agent }, "daily summary emit failed");
        }
      }
