   ```typescript
   import { readdir as fsReaddir, stat as fsStat } from "node:fs/promises";
   import { join, relative } from "node:path";
   import type { ResolvedAgentConfig } from "../shared/types.js";

   /**
    * Phase 100 GSD-06 — pure helper: extract the parent agent's GSD project root.
    * Returns the gsd.projectDir when set; undefined otherwise (which causes
    * relayCompletionToParent to skip artifact discovery entirely).
    */
   export function resolveArtifactRoot(
     parentConfig: ResolvedAgentConfig | undefined,
   ): string | undefined {
     return parentConfig?.gsd?.projectDir;
   }

   /**
    * Phase 100 GSD-06 — pure async helper: enumerate phase directories under
    * <root>/.planning/phases/ filtered by mtime (last 24h) and prioritized
    * by phase-number prefix matching the task hint. Returns up to 5 RELATIVE
    * paths (from <root>) to keep Discord embeds compact (RESEARCH.md Pitfall 8).
    *
    * Failures (root doesn't exist, readdir errors, stat errors per entry) all
    * resolve to []. NEVER throws — caller's relayCompletionToParent is
    * already in a try/catch but this function returns the empty fallback
    * proactively.
    *
    * DI: filesystem operations are injected so tests don't need real fs.
    */
   export async function discoverArtifactPaths(
     deps: {
       readdir: typeof fsReaddir;
       stat: typeof fsStat;
     },
     root: string,
     taskHint?: string,
   ): Promise<readonly string[]> {
     const phasesDir = join(root, ".planning", "phases");
     try {
       await deps.stat(phasesDir);
     } catch {
       return [];
     }
     let entries: string[];
     try {
       const dirents = await deps.readdir(phasesDir, { withFileTypes: true });
       entries = dirents.filter((d) => d.isDirectory()).map((d) => d.name);
     } catch {
       return [];
     }
     // Filter by 24h mtime window
     const now = Date.now();
     const window = 24 * 60 * 60 * 1000;
     const candidates: { name: string; mtimeMs: number }[] = [];
     for (const name of entries) {
       try {
         const st = await deps.stat(join(phasesDir, name));
         if (now - st.mtimeMs <= window) {
           candidates.push({ name, mtimeMs: st.mtimeMs });
         }
       } catch {
         // Skip unreadable entries
       }
     }
     // Phase-prefix priority
     const phaseMatch = taskHint?.match(/\b(\d+)\b/);
     const phaseNum = phaseMatch?.[1];
     candidates.sort((a, b) => {
       if (phaseNum) {
         const aMatches = a.name.startsWith(`${phaseNum}-`) || a.name === phaseNum;
         const bMatches = b.name.startsWith(`${phaseNum}-`) || b.name === phaseNum;
         if (aMatches && !bMatches) return -1;
         if (!aMatches && bMatches) return 1;
       }
       return b.mtimeMs - a.mtimeMs; // most recent first
     });
     return candidates
       .slice(0, 5)
       .map((c) => `.planning/phases/${c.name}/`);
   }
   ```
