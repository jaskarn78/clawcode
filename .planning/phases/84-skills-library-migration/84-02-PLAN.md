### Scope tags (canonical source — encode as exported const in skills-scope-tags.ts)

```typescript
// Scope strings — ONLY these four values. Map is NOT user-extensible in v2.2
// (deferred to future milestone per SKILL-F1 in REQUIREMENTS.md).
export const SCOPE_TAGS: ReadonlyMap<string, "finmentum" | "personal" | "fleet"> = new Map([
  // P1 skills
  ["finmentum-crm", "finmentum"],
  ["new-reel", "finmentum"],
  ["frontend-design", "fleet"],
  ["self-improving-agent", "fleet"],
  ["tuya-ac", "personal"],
  // Future: P2 skills added here when they migrate (v2.2 only defines P1 scope).
]);

// Agent families — map agent-name-prefix to scope.
// "finmentum" agents have names starting with "fin-" (fin-acquisition, fin-family, fin-ops,
// fin-operations, etc.). "personal" agents are hardcoded by name (jas, clawdy — names that
// bind to Jas's personal channel set). Everyone else gets "fleet".
export function scopeForAgent(agentName: string): "finmentum" | "personal" | "fleet" {
  if (agentName.startsWith("fin-")) return "finmentum";
  if (["clawdy", "jas"].includes(agentName)) return "personal";
  return "fleet";
}

export function canLinkSkillToAgent(
  skillName: string,
  agentName: string,
  opts?: { readonly force?: boolean },
): boolean {
  if (opts?.force) return true;
  const skillScope = SCOPE_TAGS.get(skillName) ?? "fleet";
  const agentScope = scopeForAgent(agentName);
  // Rules:
  //   - fleet skills link to any agent (always allowed)
  //   - finmentum skills link only to finmentum agents
  //   - personal skills link only to personal agents (tuya-ac → clawdy/jas)
  if (skillScope === "fleet") return true;
  return skillScope === agentScope;
}
```
