```ts
// src/cutover/types.ts (EXTEND)

export const CUTOVER_BUTTON_PREFIX = "cutover-";  // EXACT prefix; collision-checked vs model-confirm/skills-picker/plugins-picker/etc.

export const destructiveButtonActionSchema = z.enum(["accept", "reject", "defer"]);
export type DestructiveButtonAction = z.infer<typeof destructiveButtonActionSchema>;

/** customId format: cutover-{agent}-{gapId}:{action} */
export type CutoverButtonCustomId = `cutover-${string}-${string}:${DestructiveButtonAction}`;

/** Parses a customId. Returns null if not a cutover- button (allowing other handlers to pick it up). */
export function parseCutoverButtonCustomId(customId: string): { agent: string; gapId: string; action: DestructiveButtonAction } | null {
  if (!customId.startsWith(CUTOVER_BUTTON_PREFIX)) return null;
  const colonIdx = customId.lastIndexOf(":");
  if (colonIdx < 0) return null;
  const action = customId.slice(colonIdx + 1);
  const parsed = destructiveButtonActionSchema.safeParse(action);
  if (!parsed.success) return null;
  const body = customId.slice(CUTOVER_BUTTON_PREFIX.length, colonIdx);
  // Body shape: agent-gapId — split on LAST hyphen to allow agent names with hyphens
  const lastHyphen = body.lastIndexOf("-");
  if (lastHyphen < 0) return null;
  return { agent: body.slice(0, lastHyphen), gapId: body.slice(lastHyphen + 1), action: parsed.data };
}

export type DestructiveButtonOutcome =
  | { kind: "accepted-applied";        agent: string; gapKind: string; identifier: string; ledgerRow: CutoverLedgerRow }
  | { kind: "accepted-apply-failed";   agent: string; gapKind: string; identifier: string; error: string }
  | { kind: "rejected";                agent: string; gapKind: string; identifier: string; ledgerRow: CutoverLedgerRow }
  | { kind: "deferred";                agent: string; gapKind: string; identifier: string }
  | { kind: "expired";                 customId: string }              // collector timeout — no row written
  | { kind: "invalid-customId";        customId: string };
```

    ```ts
    import { readFile } from "node:fs/promises";
    import { gzipSync } from "node:zlib";
    import {
      type DestructiveCutoverGap,
      type CutoverLedgerRow,
      assertNever,
    } from "./types.js";
    import { appendCutoverRow } from "./ledger.js";

    const SNAPSHOT_MAX_BYTES = 64 * 1024;

    export async function applyDestructiveFix(
      deps: DestructiveApplierDeps,
      gap: DestructiveCutoverGap,
    ): Promise<{ kind: "applied"; row: CutoverLedgerRow } | { kind: "failed"; error: string }> {
      const start = (deps.now ?? (() => new Date()))();

      // Capture preChangeSnapshot BEFORE any mutation.
      let snapshotB64: string | null = null;
      let preChangeReason: string | null = null;
      let sourceHashCaptured: string | null = null;

      try {
        switch (gap.kind) {
          case "outdated-memory-file": {
            const targetPath = gap.targetRef.path; // absolute path resolved via deps.memoryRoot upstream
            try {
              const buf = await readFile(targetPath);
              if (buf.byteLength <= SNAPSHOT_MAX_BYTES) {
                snapshotB64 = gzipSync(buf).toString("base64");
              } else {
                preChangeReason = "irreversible-without-backup";
              }
              sourceHashCaptured = createHash("sha256").update(buf).digest("hex");
            } catch { /* missing target — apply will create it; snapshot is null */ }
            // Apply: rsync from OpenClaw
            const rsyncResult = await deps.runRsync([
              "-av",
              "-e", "ssh -o BatchMode=yes -o ConnectTimeout=10",
              `${deps.openClawHost}:${gap.sourceRef.path}`,
              targetPath,
            ]);
            if (rsyncResult.exitCode !== 0) {
              return { kind: "failed", error: rsyncResult.stderr.slice(0, 4000) };
            }
            const targetHash = createHash("sha256").update(await readFile(targetPath)).digest("hex");
            const row: CutoverLedgerRow = {
              timestamp: start.toISOString(),
              agent: deps.agent,
              action: "apply-destructive",
              kind: gap.kind,
              identifier: gap.identifier,
              sourceHash: gap.sourceRef.sourceHash,
              targetHash,
              reversible: snapshotB64 !== null,
              rolledBack: false,
              preChangeSnapshot: snapshotB64,
              reason: preChangeReason,
            };
            await appendCutoverRow(deps.ledgerPath, row, deps.log);
            return { kind: "applied", row };
          }
          case "mcp-credential-drift": {
            // The actual fix is operator-driven (1Password op:// ref refresh); applier records the operator's
            // confirmation but doesn't auto-rewrite credentials (per D-06 — propose-and-confirm). We log
            // the apply-destructive row so the audit trail captures the decision; the operator separately
            // updates the env via /clawcode-plugins-browse → install path.

              timestamp: start.toISOString(),
              agent: deps.agent,
              action: "apply-destructive",
              kind: gap.kind,
              identifier: gap.identifier,
              sourceHash: null,
              targetHash: null,
              reversible: false,
              rolledBack: false,
              preChangeSnapshot: null,
              reason: "operator-confirmed-credential-drift; manual op:// update required via /clawcode-plugins-browse",
            };
            await appendCutoverRow(deps.ledgerPath, row, deps.log);
            return { kind: "applied", row };
          }
          case "tool-permission-gap": {
            // First-pass: log operator's accept; ACL editing is out of scope (target ACL probe was
            // synthetic in Plan 92-02). Future enhancement wires the actual ACL writer.
            const row: CutoverLedgerRow = {

              sourceHash: null,
    ```
