After Plan 92-02's diff engine produces CUTOVER-GAPS.json, the operator runs `clawcode cutover apply-additive --apply` and the additive-only kinds get fixed without further confirmation (per D-07's three-tier safety: dry-run default → --apply-additive auto-applies the 4 additive kinds → destructive ALWAYS gated, NEVER auto). Reuses every existing primitive — zero new mutation surface.

Output: cutover-ledger.jsonl path + reader/writer module, the additive-applier dispatcher with per-kind fix logic, the `clawcode cutover apply-additive` CLI subcommand, and 12+ tests covering one happy-path + one no-op-idempotency + one secret-scan-refusal per additive kind, plus the destructive-gap deferral check.

```ts
// src/cutover/types.ts (EXTEND)

/**
 * D-05 ledger row — append-only JSONL at ~/.clawcode/manager/cutover-ledger.jsonl.
 * Phase 84 skills-ledger.ts shape donor; Phase 82 migration ledger.ts append-only invariants.
 */
export const cutoverLedgerActionSchema = z.enum([
  "apply-additive",
  "apply-destructive",
  "reject-destructive",
  "rollback",
  "skip-verify",
]);
export type CutoverLedgerAction = z.infer<typeof cutoverLedgerActionSchema>;

export const cutoverLedgerRowSchema = z.object({
  timestamp: z.string().refine((v) => !Number.isNaN(Date.parse(v)) && v.includes("T")),
  agent: z.string().min(1),
  action: cutoverLedgerActionSchema,
  kind: z.string(),               // CutoverGap['kind'] OR a meta-action like "skip-verify"
  identifier: z.string(),
  sourceHash: z.string().nullable(),       // sha256 of source content (null when not applicable)
  targetHash: z.string().nullable(),       // sha256 of target content after apply (null on dry-run / refusal)
  reversible: z.boolean(),                  // true for additive; for destructive only when preChangeSnapshot fits in ledger
  rolledBack: z.boolean(),                  // false at apply-time; rollback action appends a NEW row that flips this to true on the original via timestamp pointer
  preChangeSnapshot: z.string().nullable(), // base64-gzipped pre-apply content for files < 64KB (destructive only); null otherwise
  reason: z.string().nullable(),            // populated for skip-verify / reject-destructive / refusal-due-to-secret-scan
});
export type CutoverLedgerRow = z.infer<typeof cutoverLedgerRowSchema>;

/** Outcome of one apply-additive run. */
export type AdditiveApplyOutcome =
  | { kind: "applied";                  agent: string; gapsApplied: number; gapsSkipped: number; destructiveDeferred: number; ledgerPath: string; durationMs: number }
  | { kind: "dry-run";                  agent: string; plannedAdditive: number; destructiveDeferred: number }
  | { kind: "no-gaps-file";             agent: string; gapsPath: string }
  | { kind: "secret-scan-refused";      agent: string; identifier: string; reason: string }
  | { kind: "yaml-write-failed";        agent: string; identifier: string; error: string }
  | { kind: "rsync-failed";             agent: string; identifier: string; error: string }
  | { kind: "destructive-gaps-deferred"; agent: string; destructiveCount: number };
```

    ```ts
     *   3. appendFile (NOT writeFile) — concurrent writers cannot lose rows.
     */
    import { appendFile, mkdir, readFile } from "node:fs/promises";
    import { dirname, join } from "node:path";
    import { homedir } from "node:os";
    import type { Logger } from "pino";
    import {
      cutoverLedgerRowSchema,
      type CutoverLedgerRow,
    } from "./types.js";

    export const DEFAULT_CUTOVER_LEDGER_PATH = join(homedir(), ".clawcode", "manager", "cutover-ledger.jsonl");

    export async function appendCutoverRow(
      filePath: string,
      row: CutoverLedgerRow,
      log?: Logger,
    ): Promise<void> {
      // Validate FIRST — invalid row never reaches the filesystem.
      const validated = cutoverLedgerRowSchema.parse(row);
      await mkdir(dirname(filePath), { recursive: true });
      await appendFile(filePath, JSON.stringify(validated) + "\n", "utf8");
    }

    export async function readCutoverRows(
      filePath: string,
      log?: Logger,
    ): Promise<readonly CutoverLedgerRow[]> {
      let raw: string;
      try { raw = await readFile(filePath, "utf8"); }
      catch { return []; }
      const rows: CutoverLedgerRow[] = [];
      for (const line of raw.split("\n")) {
        if (!line.trim()) continue;
        try {
          const parsed = cutoverLedgerRowSchema.safeParse(JSON.parse(line));
          if (parsed.success) rows.push(parsed.data);
          else log?.warn({ line: line.slice(0, 200) }, "cutover ledger: malformed row skipped");
        } catch {
          log?.warn({ line: line.slice(0, 200) }, "cutover ledger: JSON parse failed");
        }
      }
      return rows;
    }

    export async function queryCutoverRowsByAgent(
      filePath: string,
      agent: string,
      log?: Logger,
    ): Promise<readonly CutoverLedgerRow[]> {
      const all = await readCutoverRows(filePath, log);
      return all.filter((r) => r.agent === agent);
    }
    ```

    ```ts
    export async function applyAdditiveFixes(deps: AdditiveApplierDeps): Promise<AdditiveApplyOutcome> {
      const start = (deps.now ?? (() => new Date()))();
      const additive = deps.gaps.filter((g) => g.severity === "additive");
      const destructiveCount = deps.gaps.length - additive.length;

      if (!deps.apply) {
        return {
          kind: "dry-run",
          agent: deps.agent,
          plannedAdditive: additive.length,
          destructiveDeferred: destructiveCount,
        };
      }

      let gapsApplied = 0;
      let gapsSkipped = 0;

      // Read current YAML once for idempotency checks.
      const yamlRaw = await readFile(deps.clawcodeYamlPath, "utf8");
      const yamlDoc = parseDocument(yamlRaw);
      const currentSkills: readonly string[] = (yamlDoc.toJS() as { agents?: Record<string, { skills?: string[] }> })
        .agents?.[deps.agent]?.skills ?? [];
      const currentAllowedModels: readonly string[] = (yamlDoc.toJS() as { agents?: Record<string, { allowedModels?: string[] }> })
        .agents?.[deps.agent]?.allowedModels ?? [];

      for (const gap of additive) {
        try {
          if (gap.kind === "missing-skill") {
            // Idempotency: skill already in YAML?
            if (currentSkills.includes(gap.identifier)) { gapsSkipped += 1; continue; }
            // Secret-scan gate — BEFORE any copy.
            const sourceSkillDir = join(deps.openClawSkillsRoot, gap.identifier);
            const scan = await deps.scanSkillForSecrets(sourceSkillDir);
            if (scan.refused) {
              return { kind: "secret-scan-refused", agent: deps.agent, identifier: gap.identifier, reason: scan.reason ?? "secret detected" };
            }
            await deps.normalizeSkillFrontmatter(sourceSkillDir);
            // rsync copy.
            const targetSkillDir = join(deps.skillsTargetDir, gap.identifier);
            const rsyncResult = await deps.runRsync([
              "-av", "--delete",
              "-e", "ssh -o BatchMode=yes -o ConnectTimeout=10",
              `${deps.openClawHost}:${sourceSkillDir}/`,
              `${targetSkillDir}/`,
            ]);
            if (rsyncResult.exitCode !== 0) {
              return { kind: "rsync-failed", agent: deps.agent, identifier: gap.identifier, error: rsyncResult.stderr.slice(0, 4000) };
            }
            // YAML update.
            const nextSkills = [...currentSkills, gap.identifier].sort();
            const updateRes = await deps.updateAgentSkills(deps.agent, nextSkills, { clawcodeYamlPath: deps.clawcodeYamlPath });
            if (updateRes.kind !== "updated" && updateRes.kind !== "no-op") {
              return { kind: "yaml-write-failed", agent: deps.agent, identifier: gap.identifier, error: `updateAgentSkills returned ${updateRes.kind}` };
            }
            await appendCutoverRow(deps.ledgerPath, makeAdditiveRow({ agent: deps.agent, kind: gap.kind, identifier: gap.identifier, sourceHash: null, targetHash: null, now: start }));
            gapsApplied += 1;
          } else if (gap.kind === "missing-mcp") {
            // missing-mcp is additive but requires the operator to first ensure MCP credentials exist
            // in 1Password (op:// refs). This applier records the gap as "deferred" via a special
            // secret-scan-refused-shaped outcome, OR we can treat as a no-op here and rely on
            // Plan 92-04 destructive embed for credential setup. SIMPLEST: skip with a ledger row
            // tagged reason="missing-mcp-needs-credentials" so 92-06 report surfaces it.
            // For v1: skip + ledger reason row. Counts as gapsApplied for the operator's report
            // (the "fix" was the documentation entry).
            await appendCutoverRow(deps.ledgerPath, makeAdditiveRow({
              agent: deps.agent, kind: gap.kind, identifier: gap.identifier,
              sourceHash: null, targetHash: null, now: start,
              reason: "missing-mcp-needs-credentials — operator must add op:// refs via /clawcode-plugins-browse",
            }));
            gapsApplied += 1;
          } else if (gap.kind === "missing-memory-file") {
            // Idempotency: already exists at target?
            const targetPath = join(deps.memoryRoot, gap.identifier);
            try { await stat(targetPath); gapsSkipped += 1; continue; } catch { /* not present, proceed */ }
            const sourcePath = join(deps.openClawWorkspace, gap.identifier);
            const rsyncResult = await deps.runRsync([
              "-av",
              "-e", "ssh -o BatchMode=yes -o ConnectTimeout=10",
              `${deps.openClawHost}:${sourcePath}`,
              targetPath,
            ]);
            if (rsyncResult.exitCode !== 0) return { kind: "rsync-failed", agent: deps.agent, identifier: gap.identifier, error: rsyncResult.stderr.slice(0, 4000) };
            const targetHash = createHash("sha256").update(await readFile(targetPath)).digest("hex");
            await appendCutoverRow(deps.ledgerPath, makeAdditiveRow({ agent: deps.agent, kind: gap.kind, identifier: gap.identifier, sourceHash: gap.sourceRef.sourceHash || null, targetHash, now: start }));
            gapsApplied += 1;
          } else if (gap.kind === "missing-upload") {
            const targetPath = join(deps.uploadsTargetDir, gap.identifier);
            try { await stat(targetPath); gapsSkipped += 1; continue; } catch { /* not present */ }
            const sourcePath = join(deps.openClawWorkspace, "uploads", "discord", gap.identifier);
            const rsyncResult = await deps.runRsync([
              "-av",
              "-e", "ssh -o BatchMode=yes -o ConnectTimeout=10",
              `${deps.openClawHost}:${sourcePath}`,
              targetPath,
            ]);
            if (rsyncResult.exitCode !== 0) return { kind: "rsync-failed", agent: deps.agent, identifier: gap.identifier, error: rsyncResult.stderr.slice(0, 4000) };
            await appendCutoverRow(deps.ledgerPath, makeAdditiveRow({ agent: deps.agent, kind: gap.kind, identifier: gap.identifier, sourceHash: null, targetHash: null, now: start }));
            gapsApplied += 1;
          } else if (gap.kind === "model-not-in-allowlist") {
            if (currentAllowedModels.includes(gap.identifier)) { gapsSkipped += 1; continue; }
            const nextAllowed = [...currentAllowedModels, gap.identifier].sort();
            const updateRes = await deps.updateAgentConfig(deps.agent, { allowedModels: nextAllowed }, { clawcodeYamlPath: deps.clawcodeYamlPath });
            if (updateRes.kind !== "updated" && updateRes.kind !== "no-op") {
              return { kind: "yaml-write-failed", agent: deps.agent, identifier: gap.identifier, error: `updateAgentConfig returned ${updateRes.kind}` };
            }
            await appendCutoverRow(deps.ledgerPath, makeAdditiveRow({ agent: deps.agent, kind: gap.kind, identifier: gap.identifier, sourceHash: null, targetHash: null, now: start }));
            gapsApplied += 1;
          } else {
            // Compile-time: this branch is unreachable for additive gaps. The other kinds in CutoverGap
            // are destructive and were filtered out above. assertNever pins exhaustiveness.
            const _exhaustive: never = gap as never;
            void _exhaustive;
          }
        } catch (err) {
          deps.log.warn({ err, gap }, "cutover additive applier: per-gap error");
          // Continue with remaining gaps; don't abort the batch on a single failure
          // (operator can re-run; idempotency check-then-act ensures no duplicates).
        }
      }

      return {
        kind: "applied",
        agent: deps.agent,
        gapsApplied,
        gapsSkipped,
        destructiveDeferred: destructiveCount,
        ledgerPath: deps.ledgerPath,
        durationMs: Date.now() - start.getTime(),
      };
    }

    function makeAdditiveRow(args: { agent: string; kind: string; identifier: string; sourceHash: string | null; targetHash: string | null; now: Date; reason?: string }): CutoverLedgerRow {
      return {

        agent: args.agent,
        action: "apply-additive",
        kind: args.kind,
        identifier: args.identifier,
        sourceHash: args.sourceHash,
        targetHash: args.targetHash,
        reversible: true,
        rolledBack: false,
        preChangeSnapshot: null,
        reason: args.reason ?? null,
      };
    }
    ```
