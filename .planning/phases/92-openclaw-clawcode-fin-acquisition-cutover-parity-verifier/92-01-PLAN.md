```ts
// src/cutover/types.ts (NEW — defined in this plan)
import { z } from "zod/v4";

/** D-Claude's-Discretion: chunking threshold pinned in source for grep-regression. */
export const PROFILER_CHUNK_THRESHOLD_MSGS = 50000;

export const topIntentSchema = z.object({
  intent: z.string().min(1),
  count: z.number().int().nonnegative(),
});

export const agentProfileSchema = z.object({
  tools: z.array(z.string()),                // ["Bash", "Read", "mcp__1password__read", ...]
  skills: z.array(z.string()),               // ["content-engine", "market-research", ...]
  mcpServers: z.array(z.string()),           // ["browser", "search", "1password", ...]
  memoryRefs: z.array(z.string()),           // ["vault/icapital-2026-04-03.md", ...]
  models: z.array(z.string()),               // ["anthropic-api/claude-sonnet-4-6"]
  uploads: z.array(z.string()),              // ["analysis.pdf", "chart.png", ...]
  topIntents: z.array(topIntentSchema),      // sorted by count DESC, top 20
});
export type AgentProfile = z.infer<typeof agentProfileSchema>;

/** One Discord message after JSONL flattening. */
export const discordHistoryEntrySchema = z.object({
  message_id: z.string(),
  channel_id: z.string(),
  author_id: z.string(),
  author_name: z.string().optional(),
  ts: z.string(),                            // ISO 8601
  content: z.string(),
  attachments: z.array(z.object({
    name: z.string(),
    url: z.string().optional(),
    type: z.string().optional(),
    size: z.number().optional(),
  })).default([]),
  is_bot: z.boolean().default(false),
});
export type DiscordHistoryEntry = z.infer<typeof discordHistoryEntrySchema>;

/** Outcome of an ingest cycle (for CLI exit-code derivation + downstream report writers). */
export type IngestOutcome =
  | { kind: "ingested";            agent: string; channelsProcessed: number; newMessages: number; totalMessages: number; durationMs: number; jsonlPath: string }
  | { kind: "no-changes";          agent: string; totalMessages: number; durationMs: number; jsonlPath: string }
  | { kind: "discord-fetch-failed"; agent: string; channelId: string; error: string; durationMs: number }
  | { kind: "no-channels";         agent: string };

/** Outcome of a profile cycle. */
export type ProfileOutcome =
  | { kind: "profiled";            agent: string; chunksProcessed: number; messagesProcessed: number; profilePath: string; durationMs: number }
  | { kind: "no-history";          agent: string; jsonlPath: string }
  | { kind: "dispatcher-failed";   agent: string; error: string; durationMs: number }
  | { kind: "schema-validation-failed"; agent: string; error: string; rawResponse: string };
```

    ```ts
    import { mkdir, readFile, appendFile, writeFile, rename } from "node:fs/promises";
    import { join, dirname } from "node:path";
    import { randomBytes } from "node:crypto";
    import {
      discordHistoryEntrySchema,
      type DiscordHistoryEntry,
      type IngestOutcome,
    } from "./types.js";

    // ... (full export type IngestDeps + DiscordFetchMessagesFn from <interfaces> block)

    const DISCORD_INTER_REQUEST_SLEEP_MS = 500;
    const DISCORD_PAGE_SIZE = 100;
    const DEFAULT_DEPTH_MSGS = 10000;
    const DEFAULT_DEPTH_DAYS = 90;

    export async function ingestDiscordHistory(deps: IngestDeps): Promise<IngestOutcome> {
      const start = (deps.now ?? (() => new Date()))();
      if (deps.channels.length === 0) {
        return { kind: "no-channels", agent: deps.agent };
      }

      await mkdir(deps.stagingDir, { recursive: true });
      const jsonlPath = join(deps.stagingDir, "discord-history.jsonl");
      const sleep = deps.sleep ?? ((ms: number) => new Promise<void>((r) => setTimeout(r, ms)));

      // Read existing JSONL into Set<message_id> for dedup (Phase 80 origin_id pattern).
      const existingIds = new Set<string>();
      try {
        const existing = await readFile(jsonlPath, "utf8");
        for (const line of existing.split("\n")) {
          if (!line.trim()) continue;
          try {
            const entry = JSON.parse(line);
            if (typeof entry?.message_id === "string") existingIds.add(entry.message_id);
          } catch { /* skip malformed lines */ }
        }
      } catch { /* ENOENT — first run */ }

      const depthMsgs = deps.depthMsgs ?? DEFAULT_DEPTH_MSGS;
      // depthDays not yet enforced (D-01 supports it but kept simple — depthMsgs is the binding cap for first pass)

      let totalNew = 0;
      let totalSeen = existingIds.size;
      let channelsProcessed = 0;
      const newEntries: DiscordHistoryEntry[] = [];

      for (const channelId of deps.channels) {
        let before: string | undefined = undefined;
        let fetchedThisChannel = 0;
        let firstRequest = true;

        // eslint-disable-next-line no-constant-condition
        while (true) {
          if (!firstRequest) await sleep(DISCORD_INTER_REQUEST_SLEEP_MS);
          firstRequest = false;

          const remaining = Math.max(0, depthMsgs - fetchedThisChannel);
          if (remaining === 0) break;
          const limit = Math.min(DISCORD_PAGE_SIZE, remaining);

          let page;
          try {
            page = await deps.fetchMessages({ chat_id: channelId, before, limit });
          } catch (err) {
            const msg = err instanceof Error ? err.message : String(err);
            return {
              kind: "discord-fetch-failed",
              agent: deps.agent,
              channelId,
              error: msg,
              durationMs: Date.now() - start.getTime(),
            };
          }

          for (const raw of page.messages) {
            const parsed = discordHistoryEntrySchema.safeParse(raw);
            if (!parsed.success) continue;
            if (existingIds.has(parsed.data.message_id)) continue;
            existingIds.add(parsed.data.message_id);
            newEntries.push(parsed.data);
            totalNew += 1;
            totalSeen += 1;
          }

          fetchedThisChannel += page.messages.length;
          if (!page.hasMore || page.messages.length < limit) break;
          before = page.messages[page.messages.length - 1]?.message_id;
        }

        channelsProcessed += 1;
      }

      if (totalNew === 0) {
        return {
          kind: "no-changes",
          agent: deps.agent,
          totalMessages: totalSeen,
          durationMs: Date.now() - start.getTime(),
          jsonlPath,
        };
      }

      // Sort newEntries by ts ASC (oldest→newest) before append for deterministic ordering.
      const sorted = [...newEntries].sort((a, b) => a.ts.localeCompare(b.ts));
      const jsonlLines = sorted.map((e) => JSON.stringify(e)).join("\n") + "\n";
      await appendFile(jsonlPath, jsonlLines, "utf8");

      return {
        kind: "ingested",
        agent: deps.agent,
        channelsProcessed,
        newMessages: totalNew,
        totalMessages: totalSeen,
        durationMs: Date.now() - start.getTime(),
        jsonlPath,
      };
    }
    ```
