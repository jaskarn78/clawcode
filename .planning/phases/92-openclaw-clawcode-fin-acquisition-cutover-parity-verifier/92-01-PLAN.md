```ts
// src/cutover/types.ts (NEW — defined in this plan)
import { z } from "zod/v4";

/** D-Claude's-Discretion: chunking threshold pinned in source for grep-regression. */
export const PROFILER_CHUNK_THRESHOLD_MSGS = 50000;

/** D-11: Mission Control default base URL. Override via env MC_API_BASE or --mc-base. */
export const MC_DEFAULT_BASE_URL = "http://100.71.14.96:4000";

export const topIntentSchema = z.object({
  intent: z.string().min(1),         // may be prefixed "cron:" for cron-kind sessions (D-11)
  count: z.number().int().nonnegative(),
});

export const agentProfileSchema = z.object({
  tools: z.array(z.string()),                // ["Bash", "Read", "mcp__1password__read", ...]
  skills: z.array(z.string()),               // ["content-engine", "market-research", ...]
  mcpServers: z.array(z.string()),           // ["browser", "search", "1password", ...]
  memoryRefs: z.array(z.string()),           // ["vault/icapital-2026-04-03.md", ...]
  models: z.array(z.string()),               // ["anthropic-api/claude-sonnet-4-6"]
  uploads: z.array(z.string()),              // ["analysis.pdf", "chart.png", ...]
  topIntents: z.array(topIntentSchema),      // sorted by count DESC, top 20; cron-kind clustered separately with "cron:" prefix
});
export type AgentProfile = z.infer<typeof agentProfileSchema>;

/** D-11: One Mission Control session-history entry, post JSONL flattening. */
export const mcHistoryEntrySchema = z.object({
  origin: z.literal("mc"),
  sessionId: z.string(),
  sequenceIndex: z.number().int().nonnegative(),
  role: z.enum(["user", "assistant", "system", "tool"]),
  content: z.string(),
  model: z.string().optional(),
  ts: z.string(),                            // ISO 8601
  kind: z.enum(["direct", "cron", "orchestra", "scheduled", "unknown"]),
  label: z.string().optional(),
});
export type McHistoryEntry = z.infer<typeof mcHistoryEntrySchema>;

/** One Discord message after JSONL flattening. */
export const discordHistoryEntrySchema = z.object({
  origin: z.literal("discord"),
  message_id: z.string(),
  channel_id: z.string(),
  author_id: z.string(),
  author_name: z.string().optional(),
  ts: z.string(),                            // ISO 8601
  content: z.string(),
  attachments: z.array(z.object({
    name: z.string(),
    url: z.string().optional(),
    type: z.string().optional(),
    size: z.number().optional(),
  })).default([]),
  is_bot: z.boolean().default(false),
});
export type DiscordHistoryEntry = z.infer<typeof discordHistoryEntrySchema>;

/** D-11: discriminated union — profiler accepts entries from EITHER origin. */
export const historyEntrySchema = z.discriminatedUnion("origin", [
  mcHistoryEntrySchema,
  discordHistoryEntrySchema,
]);
export type HistoryEntry = z.infer<typeof historyEntrySchema>;

/** Outcome of an MC ingest cycle (D-11). */
export type McIngestOutcome =
  | { kind: "ingested";              agent: string; sessionsProcessed: number; newEntries: number; totalEntries: number; durationMs: number; jsonlPath: string }
  | { kind: "no-changes";            agent: string; totalEntries: number; durationMs: number; jsonlPath: string }
  | { kind: "agent-not-found-in-mc"; agent: string; gatewayAgentId: string }
  | { kind: "missing-bearer-token";  agent: string }                                          // env MC_API_TOKEN unset
  | { kind: "mc-gateway-503";        agent: string; error: string; durationMs: number }       // "Failed to connect to OpenClaw Gateway"
  | { kind: "mc-fetch-failed";       agent: string; phase: "agents" | "sessions" | "history"; error: string; durationMs: number };

/** Outcome of a Discord ingest cycle. */
export type DiscordIngestOutcome =
  | { kind: "ingested";             agent: string; channelsProcessed: number; newMessages: number; totalMessages: number; durationMs: number; jsonlPath: string }
  | { kind: "no-changes";           agent: string; totalMessages: number; durationMs: number; jsonlPath: string }
  | { kind: "discord-fetch-failed"; agent: string; channelId: string; error: string; durationMs: number }
  | { kind: "no-channels";          agent: string };

/** Combined CLI-level outcome. */
export type IngestOutcome =
  | { kind: "ingested-both";       agent: string; mc: McIngestOutcome; discord: DiscordIngestOutcome }
  | { kind: "ingested-mc-only";    agent: string; mc: McIngestOutcome }
  | { kind: "ingested-discord-only"; agent: string; discord: DiscordIngestOutcome };

/** Outcome of a profile cycle. */
export type ProfileOutcome =
  | { kind: "profiled";              agent: string; chunksProcessed: number; messagesProcessed: number; profilePath: string; durationMs: number }
  | { kind: "no-history";            agent: string; jsonlPaths: readonly string[] }
  | { kind: "dispatcher-failed";     agent: string; error: string; durationMs: number }
  | { kind: "schema-validation-failed"; agent: string; error: string; rawResponse: string };
```

    ```ts
            model: (raw?.model as string | undefined) ?? session.defaults?.model,
            ts: (raw?.ts as string | undefined) ?? session.updatedAt,
            kind: (session.kind as McHistoryEntry["kind"] | undefined) ?? "unknown",
            label: session.label,
          };
          const parsed = mcHistoryEntrySchema.safeParse(candidate);
          if (!parsed.success) continue;
          const key = `${parsed.data.sessionId}:${parsed.data.sequenceIndex}`;
          if (existingKeys.has(key)) continue;
          existingKeys.add(key);
          newEntries.push(parsed.data);
        }

          totalEntries: existingKeys.size,
          durationMs: Date.now() - start.getTime(),

        };

      await appendFile(jsonlPath, lines, "utf8");

        kind: "ingested", agent: deps.agent,
        sessionsProcessed: targetSessions.length,
        newEntries: newEntries.length,
        totalEntries: existingKeys.size,
        durationMs: Date.now() - start.getTime(),
        jsonlPath,
      };
    ```

    ```ts
    function chunkBy30DayWindows(entries: readonly HistoryEntry[]): HistoryEntry[][] {

          windowStart = t;
        }
        current.push(e);
      }
      if (current.length > 0) chunks.push(current);
      return chunks;
    }

    function buildChunkPrompt(chunk: readonly HistoryEntry[]): string {
      // Render each entry as a single line: [ts] origin sessionId/channelId role: content
      // Include kind for mc entries so the model can distinguish cron from direct intents.
      const lines = chunk.map((e) => {
        if (e.origin === "mc") {
          return `[${e.ts}] mc session=${e.sessionId} kind=${e.kind} role=${e.role}: ${e.content}`;
        }
        return `[${e.ts}] discord channel=${e.channel_id} author=${e.author_name ?? e.author_id}: ${e.content}`;
      });
      return "Messages:\n" + lines.join("\n");
    }

    function extractJsonObject(text: string): string {
      const fence = text.match(/```(?:json)?\s*\n([\s\S]+?)\n```/);
      return (fence?.[1] ?? text).trim();
    }

    function mergeProfiles(parts: readonly AgentProfile[]): AgentProfile {
      const u = (k: keyof AgentProfile) => [...new Set(parts.flatMap((p) => p[k] as string[]))].sort();
      const intentCounts = new Map<string, number>();
      for (const p of parts) {
        for (const ti of p.topIntents) {
          intentCounts.set(ti.intent, (intentCounts.get(ti.intent) ?? 0) + ti.count);
        }
      }
      return {
        tools: u("tools"),
        skills: u("skills"),
        mcpServers: u("mcpServers"),
        memoryRefs: u("memoryRefs"),
        models: u("models"),
        uploads: u("uploads"),
        topIntents: [...intentCounts.entries()]
          .map(([intent, count]) => ({ intent, count }))
          .sort((a, b) => b.count - a.count || a.intent.localeCompare(b.intent))
          .slice(0, 20),
      };
    }

    function sortedKeysReplacer(): (k: string, v: unknown) => unknown {
      return (_key, value) => {
        if (value && typeof value === "object" && !Array.isArray(value)) {
          const sorted: Record<string, unknown> = {};
          for (const k of Object.keys(value as object).sort()) {
            sorted[k] = (value as Record<string, unknown>)[k];
          }
          return sorted;
        }
        return value;
      };
    }
    ```

    ```ts
      // Refuse-to-start: missing bearer when source includes mc

        return 1;

      const agentCfg = config.agents[args.agent];

      const stagingDir = args.stagingDir
        ?? join(homedir(), ".clawcode", "manager", "cutover-staging", args.agent);
      const mcBaseUrl = args.mcBase ?? process.env.MC_API_BASE ?? MC_DEFAULT_BASE_URL;

      let discordOutcome = null as Awaited<ReturnType<typeof ingestDiscordHistory>> | null;

          agent: args.agent,
          gatewayAgentId: args.agent,                    // typically same; future: read from agentCfg.openclawGatewayAgentId
          mcBaseUrl,
          bearerToken,
          stagingDir,
          fetchFn: args.fetchFn,
          log,
        });

        if (mcOutcome.kind === "mc-gateway-503" && source === "mc") {
          cliError(`MC gateway 503: ${mcOutcome.error}`);
          return 1;
        }
        if (mcOutcome.kind === "missing-bearer-token") {
          cliError("MC_API_TOKEN env var required for --source mc");
          return 1;
        }
        if (mcOutcome.kind === "mc-fetch-failed" && source === "mc") {
          cliError(`MC fetch failed (${mcOutcome.phase}): ${mcOutcome.error}`);
          return 1;
        }
        if (mcOutcome.kind === "agent-not-found-in-mc" && source === "mc") {
          cliError(`Agent ${mcOutcome.gatewayAgentId} not found in Mission Control's /api/agents response`);
          return 1;
        }

      if (wantDiscord) {
        const channels: string[] = [];
        const ch = (agentCfg.channels as { discord?: string | string[] | undefined }).discord;

      cliLog(JSON.stringify(combined, null, 2));

      // Exit code: 0 if at least one source succeeded; 1 otherwise.
      const mcOk = !mcOutcome || mcOutcome.kind === "ingested" || mcOutcome.kind === "no-changes";
      const discordOk = !discordOutcome || discordOutcome.kind === "ingested" || discordOutcome.kind === "no-changes";
      return (mcOk && discordOk) ? 0 : 1;
    }

            depthMsgs: opts.depthMsgs,
            depthDays: opts.depthDays,
            stagingDir: opts.stagingDir,
          });
          process.exit(code);
        });
    ```
