```ts
// src/cutover/types.ts (EXTEND)

export const REPORT_FRESHNESS_MS = 24 * 60 * 60 * 1000;            // 24h per D-09
export const DEFAULT_CUTOVER_REPORT_PATH = (agent: string) =>
  join(homedir(), ".clawcode", "manager", "cutover-reports", agent, "latest.md");

export const cutoverReportFrontmatterSchema = z.object({
  agent: z.string(),
  cutover_ready: z.boolean(),
  report_generated_at: z.string().refine((v) => !Number.isNaN(Date.parse(v))),
  gap_count: z.number().int().nonnegative(),
  additive_gap_count: z.number().int().nonnegative(),
  destructive_gap_count: z.number().int().nonnegative(),
  canary_pass_rate: z.number().nonnegative(),
  canary_total_invocations: z.number().int().nonnegative(),
});
export type CutoverReportFrontmatter = z.infer<typeof cutoverReportFrontmatterSchema>;

export type VerifyOutcome =
  | { kind: "verified-ready";       agent: string; reportPath: string; gapCount: 0; canaryPassRate: number }
  | { kind: "verified-not-ready";   agent: string; reportPath: string; gapCount: number; destructiveCount: number; canaryPassRate: number; reason: string }
  | { kind: "ingest-failed";        agent: string; error: string }
  | { kind: "profile-failed";       agent: string; error: string }
  | { kind: "probe-failed";         agent: string; error: string }
  | { kind: "diff-failed";          agent: string; error: string }
  | { kind: "additive-apply-failed"; agent: string; error: string; identifier: string }
  | { kind: "canary-failed";        agent: string; error: string }
  | { kind: "report-write-failed"; agent: string; error: string };

export type RollbackOutcome =
  | { kind: "rolled-back";          agent: string; rowsReverted: number; rowsSkippedIrreversible: number; rowsAlreadyRolledBack: number }
  | { kind: "no-rows-newer-than";   agent: string; targetTs: string }
  | { kind: "ledger-not-found";     agent: string }
  | { kind: "rollback-failed";      agent: string; error: string; partialRowsReverted: number };

export type SetAuthoritativePreconditionResult =
  | { kind: "precondition-passed";  reportPath: string; reportAge: number }
  | { kind: "precondition-skipped-by-flag"; reason: string }
  | { kind: "report-missing";       reportPath: string }
  | { kind: "report-stale";         reportPath: string; ageMs: number }
  | { kind: "report-not-ready";     reportPath: string; reason: string }
  | { kind: "report-invalid";       reportPath: string; error: string };
```

    ```ts
    export async function writeCutoverReport(
      deps: CutoverReportDeps,
    ): Promise<{ kind: "written"; reportPath: string; cutoverReady: boolean } | { kind: "write-failed"; error: string }> {
      try {
        const generatedAt = (deps.now ?? (() => new Date()))().toISOString();
        const additiveCount = deps.gaps.filter((g) => g.severity === "additive").length;
        const destructiveCount = deps.gaps.filter((g) => g.severity === "destructive").length;
        const total = deps.canaryResults?.length ?? 0;
        const passed = deps.canaryResults?.filter((r) => r.status === "passed").length ?? 0;
        const passRate = total === 0 ? (deps.canaryResults === null ? 0 : 100) : Math.round((passed / total) * 1000) / 10;

        // cutover_ready true iff: zero gaps remaining (operator addressed all destructive via 92-04 + applied all additive via 92-03)
        // AND canary ran AND passRate === 100. If canaryResults is null (canary skipped), cutover_ready=false.
        const cutoverReady = deps.gaps.length === 0 && deps.canaryResults !== null && passRate === 100;

        const fm: CutoverReportFrontmatter = {
          agent: deps.agent,
          cutover_ready: cutoverReady,
          report_generated_at: generatedAt,
          gap_count: deps.gaps.length,
          additive_gap_count: additiveCount,
          destructive_gap_count: destructiveCount,
          canary_pass_rate: passRate,
          canary_total_invocations: total,
        };

        const fmYaml = yamlStringify(fm).trim();

        const reasonLines: string[] = [];
        if (destructiveCount > 0) reasonLines.push(`- ${destructiveCount} destructive gap(s) — address via /clawcode-cutover-verify in admin-clawdy`);
        if (additiveCount > 0) reasonLines.push(`- ${additiveCount} additive gap(s) — run \`clawcode cutover apply-additive --apply --agent ${deps.agent}\``);
        if (deps.canaryResults === null) reasonLines.push(`- Canary not run — re-run after addressing gaps`);
        else if (passRate < 100) reasonLines.push(`- Canary pass rate ${passRate}% (${passed}/${total}) — investigate failures in CANARY-REPORT.md`);

        const body = [
          "---",
          fmYaml,
          "---",
          "",
          `# Cutover Report — ${deps.agent}`,
          "",
          `Generated: ${generatedAt}`,
          "",
          "## Summary",
          "",
          `- **Gaps remaining:** ${deps.gaps.length} (additive: ${additiveCount}, destructive: ${destructiveCount})`,
          `- **Canary:** ${deps.canaryResults === null ? "not run" : `${passed}/${total} passed (${passRate}%)`}`,
          deps.additiveOutcome ? `- **Additive outcome:** \`${deps.additiveOutcome.kind}\`` : "",
          "",
          ...(reasonLines.length === 0
            ? ["## Status\n\nAll checks pass. Cutover is ready to proceed.\n"]
            : ["## Action Items\n", ...reasonLines, ""]),
          "## Gap Detail",
          "",
          deps.gaps.length === 0
            ? "(none)"
            : deps.gaps.map((g) => `- **${g.kind}** \`${g.identifier}\` (${g.severity})`).join("\n"),
          "",
          // The literal end-of-doc line — pinned by E-LITERAL test
          `Cutover ready: ${cutoverReady}`,
          "",
        ].filter((l) => l !== "").join("\n");

        await mkdir(deps.outputDir, { recursive: true });
        const outPath = join(deps.outputDir, "CUTOVER-REPORT.md");
        const tmp = `${outPath}.${randomBytes(6).toString("hex")}.tmp`;
        await writeFile(tmp, body, "utf8");
        await rename(tmp, outPath);

        // Also update the agent-level latest.md symlink-equivalent (just write a copy at <DEFAULT_CUTOVER_REPORT_DIR>/<agent>/latest.md)
        // for set-authoritative.ts's default-path consumer.

        return { kind: "written", reportPath: outPath, cutoverReady };
      } catch (err) {
        return { kind: "write-failed", error: err instanceof Error ? err.message : String(err) };
      }

    export async function readCutoverReport(
      filePath: string,
    ): Promise<{ kind: "read"; frontmatter: CutoverReportFrontmatter; raw: string } | { kind: "missing" } | { kind: "invalid"; error: string }> {
      let raw: string;
      try { raw = await readFile(filePath, "utf8"); }
      catch { return { kind: "missing" }; }
      const fmMatch = raw.match(/^---\n([\s\S]+?)\n---\n/);
      if (!fmMatch) return { kind: "invalid", error: "no frontmatter" };
      let parsed;
      try {
        const obj = yamlParse(fmMatch[1]);
        parsed = cutoverReportFrontmatterSchema.safeParse(obj);
      } catch (err) {
        return { kind: "invalid", error: err instanceof Error ? err.message : String(err) };
      }
      if (!parsed.success) return { kind: "invalid", error: parsed.error.message };
      return { kind: "read", frontmatter: parsed.data, raw };
    }
    ```

    ```ts
    /**
     * Phase 92 Plan 06 — verify pipeline: orchestrates Plans 92-01..05.
     *
     * Phases (sequential, halt on first failure):
     *   1. ingestDiscordHistory
     *   2. profileAgentFromDiscordHistory
     *   3. probeTargetCapability
     *   4. diffAgentVsTarget (pure)
     *   5. applyAdditiveFixes (dry-run unless deps.applyAdditive=true)
     *   6. canary (only if zero destructive gaps remaining AND deps.runCanaryOnReady=true)
     *   7. writeCutoverReport (always — even on partial failure)
     */
    export async function runVerifyPipeline(deps: VerifyPipelineDeps): Promise<VerifyOutcome> {
      const ingestRes = await deps.ingestDiscordHistory({ ...deps.ingestDeps, agent: deps.agent });
      if (ingestRes.kind !== "ingested" && ingestRes.kind !== "no-changes") {
        return { kind: "ingest-failed", agent: deps.agent, error: JSON.stringify(ingestRes) };
      }

      const profileRes = await deps.profileAgentFromDiscordHistory({ ...deps.profileDeps, agent: deps.agent });
      if (profileRes.kind !== "profiled") {
        return { kind: "profile-failed", agent: deps.agent, error: JSON.stringify(profileRes) };
      }

      const probeRes = await deps.probeTargetCapability({ ...deps.probeDeps, agent: deps.agent });
      if (probeRes.kind !== "probed") {
        return { kind: "probe-failed", agent: deps.agent, error: JSON.stringify(probeRes) };
      }

      // Read profile + capability JSONs back into typed shapes
      const profile = JSON.parse(await readFile(profileRes.profilePath, "utf8")) as AgentProfile;
      const capability = JSON.parse(await readFile(probeRes.capabilityPath, "utf8")) as TargetCapability;

      const gaps = deps.diffAgentVsTarget(profile, capability);

      const additiveOutcome = await deps.applyAdditiveFixes({
        ...deps.applierDeps,
        agent: deps.agent,
        gaps,
        apply: deps.applyAdditive,
      });

      // Re-read gaps after apply: any additive gap that succeeded is now "fixed". For v1, just track
      // the destructive count from the original diff — the report writer surfaces both.
      const destructiveCount = gaps.filter((g) => g.severity === "destructive").length;

      let canaryResults: CanaryInvocationResult[] | null = null;
      if (destructiveCount === 0 && deps.runCanaryOnReady) {
        // Synthesize prompts from the AgentProfile.topIntents
        const synthRes = await deps.synthesizeCanaryPrompts({
          ...deps.synthesizerDeps,
          agent: deps.agent,
          topIntents: profile.topIntents,
        });
        if (synthRes.kind !== "synthesized") {
          // Log + write report with canary skipped + cutover_ready: false
          const reportRes = await deps.writeCutoverReport({
            agent: deps.agent, gaps, canaryResults: null, additiveOutcome,
            outputDir: deps.outputDir,
          });
          return { kind: "canary-failed", agent: deps.agent, error: `synth: ${JSON.stringify(synthRes)}` };
        }
        const canaryRes = await deps.runCanary({
          ...deps.canaryDeps,
          agent: deps.agent,
          prompts: synthRes.prompts,
        });
        if (canaryRes.kind === "ran") canaryResults = [...canaryRes.results];
        else {
          const reportRes = await deps.writeCutoverReport({
            agent: deps.agent, gaps, canaryResults: null, additiveOutcome,
            outputDir: deps.outputDir,
          });
          return { kind: "canary-failed", agent: deps.agent, error: JSON.stringify(canaryRes) };
        }
      }

      const reportRes = await deps.writeCutoverReport({
        agent: deps.agent, gaps, canaryResults, additiveOutcome, outputDir: deps.outputDir,
      });
      if (reportRes.kind !== "written") {
        return { kind: "report-write-failed", agent: deps.agent, error: reportRes.error };
      }

      const passRate = canaryResults === null ? 0 : (canaryResults.filter((r) => r.status === "passed").length / canaryResults.length) * 100;

      if (reportRes.cutoverReady) {
        return { kind: "verified-ready", agent: deps.agent, reportPath: reportRes.reportPath, gapCount: 0, canaryPassRate: passRate };
      }
      return {
        kind: "verified-not-ready",
        agent: deps.agent,
        reportPath: reportRes.reportPath,
        gapCount: gaps.length,
        destructiveCount,
        canaryPassRate: passRate,
        reason: destructiveCount > 0 ? "destructive-gaps-remaining" : (canaryResults === null ? "canary-not-run" : "canary-pass-rate-below-100"),
      };
    }
    ```
