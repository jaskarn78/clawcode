```typescript
/**
 * POSIX liveness check for a PID. Uses process.kill(pid, 0) — signal 0 only checks existence
 * + permission, does not actually signal. ESRCH (no such process) → false; EPERM (process
 * exists but we lack signal permission) → true (sufficient for our purposes).
 *
 * Validates: returns false immediately for pid <= 0 (no syscall).
 */
export function isPidAlive(pid: number): boolean;

/**
 * Existing function — extended with optional opts. When opts.minAge is set, filters out claude
 * procs younger than that many seconds. bootTimeUnix + clockTicksPerSec MUST be supplied when
 * minAge is — caller already has them in scope; we don't re-read /proc/stat per call.
 */
export interface DiscoverClaudeOpts {
  readonly minAge?: number;          // seconds
  readonly bootTimeUnix?: number;    // from readBootTimeUnix() once at boot
  readonly clockTicksPerSec?: number; // from readClockTicksPerSec() once at boot
}
export async function discoverClaudeSubprocessPid(
  daemonPid: number,
  opts?: DiscoverClaudeOpts,
): Promise<number | null>;
```

    ```typescript
      try {
        process.kill(pid, 0);
        return true;
      } catch (err) {

        if (code === "EPERM") return true;
        throw err;
    ```
