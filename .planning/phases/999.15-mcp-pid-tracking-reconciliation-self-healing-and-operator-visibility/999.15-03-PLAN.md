    case "mcp-tracker-snapshot": {
      if (!mcpTracker) {
        return { agents: [] };  // no tracker configured (Linux-only feature)
      }
      const tracker = mcpTracker;
      const entries = tracker.getRegisteredAgents();
      const agents = [];
      for (const [agent, entry] of entries) {
        const aliveMcps = entry.mcpPids.filter(isPidAlive);
        const cmdlines = tracker.getCmdlinesForAgent(agent); // new getter from Plan 01 cache
        agents.push({
          agent,
          claudePid: entry.claudePid,
          claudeAlive: isPidAlive(entry.claudePid),
          mcpPids: entry.mcpPids,
          aliveCount: aliveMcps.length,
          totalCount: entry.mcpPids.length,
          cmdlines,
          registeredAt: entry.registeredAt,
        });
      }
      return { agents };
    }
