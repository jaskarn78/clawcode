    - Import: mkdir, writeFile, access, lstat from "node:fs/promises"; join from "node:path"
    - Import: ResolvedAgentConfig, WorkspaceResult from "../shared/types.js"

    - export async function createWorkspace(agent: ResolvedAgentConfig): Promise&lt;WorkspaceResult&gt;

      - Track filesWritten: string[] for result
      - SOUL.md logic (per D-12 idempotency):
        - If agent.soul is defined: write resolved content (await resolveContent(agent.soul)) to SOUL.md -- always overwrite (config is source of truth)

      - IDENTITY.md logic (per D-12 idempotency):

        - If agent.identity is undefined AND IDENTITY.md does not exist: write renderIdentity(DEFAULT_IDENTITY_TEMPLATE, agent.name)
        - If agent.identity is undefined AND IDENTITY.md exists: skip (preserve user edits)
      - Return { agentName: agent.name, path: agent.workspace, created: true, filesWritten }
      - Wrap filesystem errors in WorkspaceError with path context

    - export async function createWorkspaces(agents: ResolvedAgentConfig[]): Promise&lt;WorkspaceResult[]&gt;
      - Map each agent through createWorkspace sequentially (not parallel -- clearer error reporting)

      - .description("Initialize agent workspaces from config")
      - .option("-c, --config &lt;path&gt;", "Path to config file", "clawcode.yaml")
      - .option("--dry-run", "Show what would be created without creating it")
