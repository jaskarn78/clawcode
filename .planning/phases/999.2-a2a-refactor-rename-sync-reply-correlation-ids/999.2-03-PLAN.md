import type { Logger } from "pino";
import type { WebhookManager } from "../discord/webhook-manager.js";
import type { AgentConfig } from "../config/types.js";  // (verify exact path during Task 2)
import { buildAgentMessageEmbed } from "../discord/agent-message.js";

export interface AskAgentDeps {
  runningAgents: Set<string> | string[];
  dispatchTurn: (to: string, content: string) => Promise<string>;
  writeInbox: (params: { from: string; to: string; content: string; priority: string }) => Promise<{ messageId: string }>;
  webhookManager: Pick<WebhookManager, "hasWebhook" | "sendAsAgent">;
  configs: AgentConfig[];
  log: Logger;
}

export interface AskAgentParams {
  from: string;
  to: string;
  content: string;
  priority?: string;
  mirror_to_target_channel?: boolean;
}

export interface AskAgentResult {
  ok: true;
  messageId: string;
  response?: string;
}

export function buildAskAgentDeps(overrides: Partial<AskAgentDeps>): AskAgentDeps {
  // For tests; in production daemon constructs AskAgentDeps directly.
  return {
    runningAgents: overrides.runningAgents ?? [],
    dispatchTurn: overrides.dispatchTurn ?? (async () => ""),
    writeInbox: overrides.writeInbox ?? (async () => ({ messageId: "test-msg-id" })),
    webhookManager: overrides.webhookManager ?? { hasWebhook: () => false, sendAsAgent: async () => {} },
    configs: overrides.configs ?? [],
    log: overrides.log ?? ({ info: () => {}, warn: () => {}, error: () => {} } as unknown as Logger),
  };
}

export async function handleAskAgentIpc(
  params: AskAgentParams,
  deps: AskAgentDeps,
): Promise<AskAgentResult> {
  const { from, to, content } = params;
  const priority = params.priority ?? "normal";
  const mirror = params.mirror_to_target_channel === true;

  const isRunning = (Array.isArray(deps.runningAgents)
    ? deps.runningAgents.includes(to)
    : deps.runningAgents.has(to));

  // Always write inbox first (preserves existing behavior — both online and offline path).
  const { messageId } = await deps.writeInbox({ from, to, content, priority });

  if (!isRunning) {
    // D-SYN-03: target offline — no synchronous reply.
    return { ok: true, messageId, response: undefined };
  }

  // D-SYN-04: pre-dispatch mirror — best-effort, never aborts the ask.
  if (mirror && deps.webhookManager.hasWebhook(to)) {
    try {
      const senderConfig = deps.configs.find((c) => c.name === from);
      const senderDisplayName = senderConfig?.webhook?.displayName ?? from;
      const senderAvatarUrl = senderConfig?.webhook?.avatarUrl;
      const embed = buildAgentMessageEmbed(from, senderDisplayName, content);
      await deps.webhookManager.sendAsAgent(to, senderDisplayName, senderAvatarUrl, embed);
    } catch (err) {
      deps.log.warn(
        { from, to, err: (err as Error).message },
        "[ask-agent] mirror prompt webhook failed (best-effort, ask continues)",
      );
    }
  }

  // D-SYN-05: NO try/catch around dispatchTurn — errors propagate.
  const response = await deps.dispatchTurn(to, content);

  // D-SYN-04: post-dispatch mirror — best-effort, never aborts the ask.
  if (mirror && response && deps.webhookManager.hasWebhook(to)) {
    try {
      const targetConfig = deps.configs.find((c) => c.name === to);
      const targetDisplayName = targetConfig?.webhook?.displayName ?? to;
      const targetAvatarUrl = targetConfig?.webhook?.avatarUrl;
      // Reply-threading via Discord blockquote (Pitfall 4 fix).
      const embedBody = `> reply to ${from}:\n\n${response}`;
      const embed = buildAgentMessageEmbed(to, targetDisplayName, embedBody);
      await deps.webhookManager.sendAsAgent(to, targetDisplayName, targetAvatarUrl, embed);
    } catch (err) {
      deps.log.warn(
        { from, to, err: (err as Error).message },
        "[ask-agent] mirror response webhook failed (best-effort, ask continues)",
      );
    }
  }

  return { ok: true, messageId, response };
}

const askAgentHandler = async (input) => {
  try {
    const result = (await sendIpcRequest(SOCKET_PATH, "ask-agent", input)) as {
      ok: boolean;
      messageId: string;
      response?: string;
    };

    if (result.response !== undefined) {
      // D-SYN-02 — surface the reply in the tool-result text. THE BUG-FIX LINE.
      return {
        content: [{
          type: "text" as const,
          text: `Message sent to ${input.to} (id: ${result.messageId})\n\n${input.to} replied:\n${result.response}`,
        }],
      };
    }
    // D-SYN-03 — offline target.
    return {
      content: [{
        type: "text" as const,
        text: `Message queued in ${input.to}'s inbox. ${input.to} is not running — no synchronous reply.`,
      }],
    };
  } catch (err) {
    // D-SYN-05 — error propagation as plain tool-result text (caller's LLM sees the failure).
    const msg = err instanceof Error ? err.message : String(err);
    return {
      content: [{ type: "text" as const, text: `Failed to ask ${input.to}: ${msg}` }],
    };
  }
};
