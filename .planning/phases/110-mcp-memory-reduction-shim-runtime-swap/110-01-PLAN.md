    In `src/manager/daemon.ts`, locate the IPC method dispatch (look for the switch/Map that handles existing methods like `broker-status`, `fleet-stats`, `ask-agent`). Add a handler for `list-mcp-tools`:

    ```typescript
    import { zodToJsonSchema } from "zod-to-json-schema";
    import { TOOL_DEFINITIONS as SEARCH_TOOLS } from "../search/tools";
    import { TOOL_DEFINITIONS as IMAGE_TOOLS } from "../image/tools";
    import { TOOL_DEFINITIONS as BROWSER_TOOLS } from "../browser/tools";
    import { z } from "zod/v4";  // match the version used in tools.ts files
    import { listMcpToolsRequestSchema, type ListMcpToolsResponse } from "../ipc/protocol";

    function handleListMcpTools(rawParams: unknown): ListMcpToolsResponse {
      const parsed = listMcpToolsRequestSchema.safeParse(rawParams);
      if (!parsed.success) {
        // -32602 invalid params per JSON-RPC 2.0 spec
        throw new IpcError("invalid params: " + parsed.error.message, -32602);
      }
      const defs =
        parsed.data.shimType === "search"  ? SEARCH_TOOLS  :
        parsed.data.shimType === "image"   ? IMAGE_TOOLS   :
        BROWSER_TOOLS;
      return {
        tools: defs.map((def) => ({
          name: def.name,
          description: def.description,
          inputSchema: zodToJsonSchema(z.object(def.schemaBuilder(z))) as Record<string, unknown>,
        })),
      };
    }
    ```
