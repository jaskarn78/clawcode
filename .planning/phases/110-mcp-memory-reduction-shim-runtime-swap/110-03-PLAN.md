    ```javascript
    #!/usr/bin/env node
    // Phase 110 Stage 0b — npm postinstall hook.
    // Selects the correct prebuilt clawcode-mcp-shim binary for the host
    // arch and copies it to node_modules/.bin/clawcode-mcp-shim so it's
    // on $PATH for child processes spawned by the daemon.
    //
    // Mirrors the better-sqlite3 prebuild-install pattern (locked
    // decision in CONTEXT.md). Fails LOUD on unsupported arch — no
    // silent skip, per the global fail-loud policy.

    "use strict";
    const fs = require("fs");
    const path = require("path");

    const SUPPORTED = {
      "linux-x64":   "linux-amd64",
      "linux-arm64": "linux-arm64",
    };

    function selectPrebuild(platform, arch) {
      const key = `${platform}-${arch}`;
      const dir = SUPPORTED[key];
      if (!dir) {
        const supportedKeys = Object.keys(SUPPORTED).join(", ");
        throw new Error(
          `clawcode-mcp-shim: no prebuilt binary for ${key}. ` +
          `Supported: ${supportedKeys}. ` +
          `Build from source via 'go build ./cmd/clawcode-mcp-shim' or open an issue.`,
        );
      }
      return path.join("prebuilds", dir, "clawcode-mcp-shim");
    }

    function install({ pkgRoot, target }) {
      const src = path.join(pkgRoot, selectPrebuild(process.platform, process.arch));
      if (!fs.existsSync(src)) {
        throw new Error(`clawcode-mcp-shim: prebuild missing at ${src} — npm package is corrupt or built without Go shim. Re-run npm publish from a clean Go-build CI.`);
      }
      fs.mkdirSync(path.dirname(target), { recursive: true });
      fs.copyFileSync(src, target);
      fs.chmodSync(target, 0o755);
    }

    if (require.main === module) {
      try {
        const pkgRoot = path.resolve(__dirname, "..", "..");
        const target = path.join(pkgRoot, "node_modules", ".bin", "clawcode-mcp-shim");
        // When clawcode is itself the installed package: pkgRoot is the
        // package dir; npm has already created node_modules/.bin nearby.
        // We resolve target relative to the package root.
        install({ pkgRoot, target });
        console.log(`clawcode-mcp-shim: installed binary at ${target}`);
      } catch (err) {
        console.error(`clawcode-mcp-shim postinstall FAILED: ${err.message}`);
        process.exit(1);
      }
    }

    module.exports = { selectPrebuild, install };
    ```

    ```yaml
    name: npm Publish (with Go shim prebuilds)

    on:
      push:
        tags: ['v*']
      workflow_dispatch:
        inputs:
          go_build_run_id:
            description: 'Override go-build.yml run id (default: latest successful on master)'
            required: false
            type: string

    concurrency:
      group: npm-publish
      cancel-in-progress: false

    jobs:
      publish:
        name: Bundle prebuilds and npm publish
        runs-on: ubuntu-latest
        timeout-minutes: 20
        permissions:
          contents: read
          actions: read
          id-token: write
        steps:
          - name: Checkout
            uses: actions/checkout@v4

          - name: Setup Node.js
            uses: actions/setup-node@v4
            with:
              node-version: '22'
              cache: 'npm'
              registry-url: 'https://registry.npmjs.org'

          - name: Install dependencies
            run: npm ci

          - name: Build TypeScript
            run: npm run build

          - name: Resolve go-build run id
            id: resolve_run
            env:
              GH_TOKEN: ${{ secrets.GITHUB_TOKEN }}
              OVERRIDE: ${{ inputs.go_build_run_id }}
            run: |
              if [ -n "$OVERRIDE" ]; then
                echo "Using override run id: $OVERRIDE" >&2
                echo "run_id=$OVERRIDE" >> "$GITHUB_OUTPUT"
              else
                RUN_ID=$(gh run list --workflow=go-build.yml --branch=master --status=success --limit=1 --json databaseId -q '.[0].databaseId')
                if [ -z "$RUN_ID" ] || [ "$RUN_ID" = "null" ]; then
                  echo "ERROR: no successful go-build.yml run on master to download artifacts from" >&2
                  exit 1
                fi
                echo "Resolved go-build run id: $RUN_ID" >&2
                echo "run_id=$RUN_ID" >> "$GITHUB_OUTPUT"
              fi

          - name: Download amd64 prebuild
            uses: actions/download-artifact@v4
            with:
              name: clawcode-mcp-shim-linux-amd64
              path: ./_artifacts/amd64
              github-token: ${{ secrets.GITHUB_TOKEN }}
              run-id: ${{ steps.resolve_run.outputs.run_id }}

          - name: Download arm64 prebuild
            uses: actions/download-artifact@v4
            with:
              name: clawcode-mcp-shim-linux-arm64
              path: ./_artifacts/arm64
              github-token: ${{ secrets.GITHUB_TOKEN }}
              run-id: ${{ steps.resolve_run.outputs.run_id }}

              set -euo pipefail
    ```
