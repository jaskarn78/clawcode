    ```yaml
    name: Go Shim Build

    on:

          - 'internal/shim/**'
          - 'go.mod'
          - 'go.sum'
          - '.github/workflows/go-build.yml'
      workflow_dispatch:

    concurrency:
      group: ${{ github.workflow }}-${{ github.ref }}
      cancel-in-progress: true

    jobs:
      build:
        name: Build clawcode-mcp-shim (${{ matrix.goarch }})
        runs-on: ubuntu-latest
        strategy:
          fail-fast: false
          matrix:
            goarch: [amd64, arm64]
        steps:
          - uses: actions/checkout@v4

          - name: Set up Go
            uses: actions/setup-go@v5
            with:
              go-version: '1.22'
              cache: true
              cache-dependency-path: go.sum

          - name: Verify go.mod tidiness
            run: go mod tidy && git diff --exit-code go.mod go.sum

          - name: Run Go tests (host arch)
            if: matrix.goarch == 'amd64'
            run: go test ./...

          - name: Build static binary (${{ matrix.goarch }})
            env:
              GOOS: linux
              GOARCH: ${{ matrix.goarch }}
              CGO_ENABLED: 0
            run: |
              go build -ldflags="-s -w" -o clawcode-mcp-shim-linux-${{ matrix.goarch }} ./cmd/clawcode-mcp-shim
              # Sanity-check binary size — Go static MCP server should be well under 12 MB
              SIZE=$(stat -c %s clawcode-mcp-shim-linux-${{ matrix.goarch }})
              echo "Binary size: $SIZE bytes"
              if [ "$SIZE" -gt 12582912 ]; then
                echo "ERROR: binary exceeds 12 MB ceiling — investigate dependencies or build flags" >&2
                exit 1
              fi

          - name: Upload artifact
            uses: actions/upload-artifact@v4
            with:
              name: clawcode-mcp-shim-linux-${{ matrix.goarch }}
              path: clawcode-mcp-shim-linux-${{ matrix.goarch }}
              retention-days: 30
              if-no-files-found: error
    ```

    ```javascript
    #!/usr/bin/env node
    // Phase 110 Stage 0b — npm postinstall hook.
    // Selects the correct prebuilt clawcode-mcp-shim binary for the host
    // arch and copies it to node_modules/.bin/clawcode-mcp-shim so it's
    // on $PATH for child processes spawned by the daemon.
    //
    // Mirrors the better-sqlite3 prebuild-install pattern (locked
    // decision in CONTEXT.md). Fails LOUD on unsupported arch — no
    // silent skip, per the global fail-loud policy.

    "use strict";
    const fs = require("fs");
    const path = require("path");

    const SUPPORTED = {
      "linux-x64":   "linux-amd64",
      "linux-arm64": "linux-arm64",
    };

    function selectPrebuild(platform, arch) {
      const key = `${platform}-${arch}`;
      const dir = SUPPORTED[key];
      if (!dir) {
        const supportedKeys = Object.keys(SUPPORTED).join(", ");
        throw new Error(
          `clawcode-mcp-shim: no prebuilt binary for ${key}. ` +
          `Supported: ${supportedKeys}. ` +
          `Build from source via 'go build ./cmd/clawcode-mcp-shim' or open an issue.`,
        );
      }
      return path.join("prebuilds", dir, "clawcode-mcp-shim");
    }

    function install({ pkgRoot, target }) {
      const src = path.join(pkgRoot, selectPrebuild(process.platform, process.arch));
      if (!fs.existsSync(src)) {
        throw new Error(`clawcode-mcp-shim: prebuild missing at ${src} — npm package is corrupt or built without Go shim. Re-run npm publish from a clean Go-build CI.`);
      }
      fs.mkdirSync(path.dirname(target), { recursive: true });
      fs.copyFileSync(src, target);
      fs.chmodSync(target, 0o755);
    }

    if (require.main === module) {
      try {
        const pkgRoot = path.resolve(__dirname, "..", "..");
        const target = path.join(pkgRoot, "node_modules", ".bin", "clawcode-mcp-shim");
        // When clawcode is itself the installed package: pkgRoot is the
        // package dir; npm has already created node_modules/.bin nearby.
        // We resolve target relative to the package root.
        install({ pkgRoot, target });
        console.log(`clawcode-mcp-shim: installed binary at ${target}`);
      } catch (err) {
        console.error(`clawcode-mcp-shim postinstall FAILED: ${err.message}`);
        process.exit(1);
      }
    }

    module.exports = { selectPrebuild, install };
    ```
