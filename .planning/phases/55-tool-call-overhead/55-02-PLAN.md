```typescript
import { canonicalStringify } from "../shared/canonical-stringify.js";

/**
 * Phase 55 — per-Turn idempotent tool result cache.
 *
 * LIFETIME: one instance per Turn. The cache Map is unreachable once the
 * Turn is `end()`ed (the Turn drops its reference — GC cleans up). By
 * construction, there is no cross-turn leak.
 *
 * SCOPE: ONLY whitelisted tools (`perf.tools.idempotent`) are cache-read
 * candidates. Non-whitelisted tools bypass the cache entirely.
 *
 * MUTATION SAFETY: cache hits return `Object.freeze`d structured clones of
 * the stored value. Callers cannot corrupt subsequent hits by mutating
 * their copy.
 */
export class ToolCache {
  private readonly hits: Map<string, unknown> = new Map();

  /** Build the stable cache key for a tool call. */
  static key(toolName: string, args: unknown): string {
    return `${toolName}:${canonicalStringify(args)}`;
  }

  /**
   * Record a cached value. Structured-clones on store so mutations to the
   * stored reference do not bleed into later hits.
   */
  set(toolName: string, args: unknown, value: unknown): void { /* ... */ }

  /** Returns the frozen cached value if present, else undefined. */
  get(toolName: string, args: unknown): unknown | undefined { /* ... */ }

  /** For telemetry — total cache hits this Turn. */
  hitCount(): number { /* ... */ }
}
```

```typescript
 * runWithConcurrencyLimit runs up to `maxConcurrent` handlers in parallel
```

    ```typescript
    import { canonicalStringify } from "../shared/canonical-stringify.js";

    /**
     * Deep-freeze helper — returns a frozen copy at every level. Used on
     * both `set` and `get` so mutations to caller-visible references cannot
     * poison subsequent cache hits.
     */
    function deepFreezeClone<T>(value: T): T {
      if (value === null || typeof value !== "object") return value;
      if (Array.isArray(value)) {
        return Object.freeze(value.map((v) => deepFreezeClone(v))) as unknown as T;
      }
      const clone: Record<string, unknown> = {};
      for (const k of Object.keys(value as Record<string, unknown>)) {
        clone[k] = deepFreezeClone((value as Record<string, unknown>)[k]);
      }
      return Object.freeze(clone) as unknown as T;
    }

    export class ToolCache {
      private readonly hits: Map<string, unknown> = new Map();
      private _hitCount = 0;

      static key(toolName: string, args: unknown): string {
        return `${toolName}:${canonicalStringify(args)}`;
      }

      set(toolName: string, args: unknown, value: unknown): void {
        this.hits.set(ToolCache.key(toolName, args), deepFreezeClone(value));
      }

      get(toolName: string, args: unknown): unknown | undefined {
        const v = this.hits.get(ToolCache.key(toolName, args));
        if (v !== undefined) this._hitCount++;
        return v;
      }

      hitCount(): number { return this._hitCount; }
    }
    ```

    ```typescript
    export async function runWithConcurrencyLimit<T>(
      handlers: ReadonlyArray<() => Promise<T>>,
      maxConcurrent: number,
    ): Promise<ReadonlyArray<PromiseSettledResult<T>>> {
      if (handlers.length === 0) return [];
      if (maxConcurrent <= 0) throw new Error("maxConcurrent must be >= 1");

      const results: PromiseSettledResult<T>[] = new Array(handlers.length);
      let nextIndex = 0;
      const worker = async (): Promise<void> => {
        while (true) {
          const i = nextIndex++;
          if (i >= handlers.length) return;
          try {
            const value = await handlers[i]!();
            results[i] = { status: "fulfilled", value };
          } catch (reason) {
            results[i] = { status: "rejected", reason };
          }
        }
      };
      const workerCount = Math.min(maxConcurrent, handlers.length);
      await Promise.all(Array.from({ length: workerCount }, () => worker()));
      return results;
    }
    ```

    ```typescript
    async function invokeWithCache<R>(
      toolName: string,
      agentName: string,
      args: unknown,
      rawCall: () => Promise<R>,
      deps: McpServerDeps | undefined,
    ): Promise<R> {
    ```
