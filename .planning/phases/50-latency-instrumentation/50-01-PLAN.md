```typescript
import type { Logger } from "pino";
import type { TurnStatus, SpanRecord, TurnRecord } from "./types.js";
import type { TraceStore } from "./trace-store.js";

export class TraceCollector {
  constructor(
    private readonly store: TraceStore,
    private readonly log: Logger,
  ) {}

  startTurn(turnId: string, agent: string, channelId: string | null): Turn {
    return new Turn(turnId, agent, channelId, this.store, this.log.child({ agent, turnId }));
  }
}

export class Turn {
  private readonly spans: SpanRecord[] = [];
  private readonly startedAt: number;
  private committed = false;

  constructor(
    private readonly id: string,
    private readonly agent: string,
    private readonly channelId: string | null,
    private readonly store: TraceStore,
    private readonly log: Logger,
  ) {
    this.startedAt = Date.now();
  }

  startSpan(name: string, metadata: Record<string, unknown> = {}): Span {
    const startedAt = Date.now();
    return new Span(name, startedAt, metadata, (record) => {
      if (this.committed) return;
      this.spans.push(record);
    });
  }

  end(status: TurnStatus): void {
    if (this.committed) return;
    this.committed = true;
    const endedAt = Date.now();
    const record: TurnRecord = Object.freeze({
      id: this.id,
      agent: this.agent,
      channelId: this.channelId,
      startedAt: new Date(this.startedAt).toISOString(),
      endedAt: new Date(endedAt).toISOString(),
      totalMs: endedAt - this.startedAt,
      status,
      spans: Object.freeze([...this.spans]),
    });
    try {
      this.store.writeTurn(record);
      this.log.debug({ spanCount: record.spans.length, totalMs: record.totalMs }, "turn traced");
    } catch (err) {
      const msg = err instanceof Error ? err.message : "unknown";
      this.log.warn({ err: msg }, "trace write failed — dropping turn");
    }
  }
}

export class Span {
  private ended = false;
  constructor(
    private readonly name: string,
    private readonly startedAt: number,
    private readonly metadata: Record<string, unknown>,
    private readonly onEnd: (record: SpanRecord) => void,
  ) {}

  end(): void {
    if (this.ended) return;
    this.ended = true;
    const durationMs = Date.now() - this.startedAt;
    this.onEnd(Object.freeze({
      name: this.name,
      startedAt: new Date(this.startedAt).toISOString(),
      durationMs,
      metadata: Object.freeze({ ...this.metadata }),
    }));
  }
}
```
