4. Modify `sendAndStream` (line 430) per RESEARCH code example. Full replacement body:
```typescript
async sendAndStream(message: string, onChunk: (accumulated: string) => void, turn?: Turn): Promise<string> {
  if (closed) throw new Error(`Session ${sessionId} is closed`);
  const endToEnd = turn?.startSpan("end_to_end");
  const firstToken = turn?.startSpan("first_token");
  let firstTokenEnded = false;
  const activeTools = new Map<string, import("../performance/trace-collector.js").Span>();

  try {
    const q = sdk.query({ prompt: message, options: turnOptions() });
    const textParts: string[] = [];
    for await (const msg of q) {
      if (msg.type === "assistant") {
        // Filter out subagent-generated assistant messages (parent_tool_use_id !== null)
        // See RESEARCH Pitfall 6 — only the PARENT assistant's first text block marks first_token
        const parentToolUseId = (msg as { parent_tool_use_id?: string | null }).parent_tool_use_id ?? null;

        if (parentToolUseId === null) {
          // Inspect nested content blocks from the SDK's BetaMessage
          const contentBlocks = ((msg as { message?: { content?: unknown[] } }).message?.content ?? []) as unknown[];
          for (const raw of contentBlocks) {
            const block = raw as { type?: string; name?: string; id?: string };
            if (block.type === "text" && !firstTokenEnded) {
              firstToken?.end();
              firstTokenEnded = true;
            }
            if (block.type === "tool_use" && block.id && block.name) {
              const span = turn?.startSpan(`tool_call.${block.name}`, { tool_use_id: block.id });
              if (span) activeTools.set(block.id, span);
            }
          }
        }

        // Maintain existing streaming UX: append string content when the SDK emits the narrowed variant
        if (typeof msg.content === "string" && msg.content.length > 0) {
          textParts.push(msg.content);
          onChunk(textParts.join("\n"));
        }
      }

      // End tool_call spans when a user message carries the tool_use_result
      if (msg.type === "user") {
        const toolUseId = (msg as { parent_tool_use_id?: string | null }).parent_tool_use_id;
        if (toolUseId) {
          const span = activeTools.get(toolUseId);
          if (span) {
            span.end();
            activeTools.delete(toolUseId);
          }
        }
      }

      if (msg.type === "result") {
        if (msg.session_id) sessionId = msg.session_id;
        extractUsage(msg, usageCallback);
        // Close any still-open tool_call spans (result arrived before user message for final tool)
        for (const span of activeTools.values()) span.end();
        activeTools.clear();
        if (!firstTokenEnded) firstToken?.end();  // safety: result without text block
        endToEnd?.end();
        if ("result" in msg && typeof msg.result === "string" && msg.result.length > 0) {
          return msg.result;
        }
        if (msg.subtype !== "success") {
          if ("is_error" in msg && msg.is_error) {
            throw new Error(`Agent error: ${msg.subtype}`);
          }
        }
        break;
      }
    }
    if (!firstTokenEnded) firstToken?.end();
    endToEnd?.end();
    return textParts.join("\n");
  } catch (err) {
    for (const span of activeTools.values()) span.end();
    if (!firstTokenEnded) firstToken?.end();
    endToEnd?.end();
    const error = err instanceof Error ? err : new Error(String(err));
    notifyError(error);
    throw error;
  }
},
```

```typescript
/**
 * Heartbeat check that prunes expired traces per agent.
 *
 * Auto-discovered by HeartbeatRunner from the checks directory.
 * Deletes rows from `traces` older than `perf.traceRetentionDays` (default 7).
 * Child spans are removed via ON DELETE CASCADE foreign key (per RESEARCH Pitfall 4).
 */

import { subDays } from "date-fns";
import type { CheckModule, CheckResult } from "../types.js";

const traceRetentionCheck: CheckModule = {
  name: "trace-retention",

  async execute(context): Promise<CheckResult> {
    const { agentName, sessionManager } = context;

    const agentConfig = sessionManager.getAgentConfig(agentName);
    if (!agentConfig) {
      return { status: "healthy", message: "No config available" };
    }

    const store = sessionManager.getTraceStore(agentName);
    if (!store) {
      return { status: "healthy", message: "No trace store" };
    }

    const retentionDays = agentConfig.perf?.traceRetentionDays ?? 7;
    const cutoffDate = subDays(new Date(), retentionDays);
    const cutoffIso = cutoffDate.toISOString();

    const deleted = store.pruneOlderThan(cutoffIso);

    return {
      status: "healthy",
      message: deleted > 0 ? `Pruned ${deleted} expired turn(s)` : "No expired traces",
      metadata: { deleted, cutoff: cutoffIso, retentionDays },
    };
  },
};

export default traceRetentionCheck;
```
