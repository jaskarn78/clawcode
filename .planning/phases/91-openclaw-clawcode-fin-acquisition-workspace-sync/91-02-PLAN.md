    ```ts
    import type { SyncConflict } from "./types.js";

    export type FileHashPair = {
      readonly path: string;
      readonly sourceHash: string;
      readonly destHash: string | null;
    };

    export type ConflictDetectionResult = {
      readonly cleanFiles: readonly string[];
      readonly conflicts: readonly SyncConflict[];
    };

    /**
     * D-11: Conflict if dest sha256 ≠ last-written sha256 AND source sha256
     * has changed since last sync.
     *
     * Per-file decision matrix:
     *   - destHash === null (file missing on dest)                    → CLEAN (fresh copy)
     *   - lastWrittenHashes[path] === undefined (never synced before) → CLEAN (first sync)
     *   - destHash === lastWrittenHashes[path]                        → CLEAN (dest untouched since last write)
     *   - destHash !== lastWrittenHashes[path]:
     *       - AND sourceHash === lastWrittenHashes[path] → CLEAN (only dest changed; leave as-is until D-16 reverse sync path; for Phase 91 initial direction openclaw→clawcode, this is OPERATOR edit — still treat as CONFLICT per D-11's "operator edited destination")
     *       - AND sourceHash !== lastWrittenHashes[path] → CONFLICT (both sides changed)
     *       - D-11 literal reading: "destination hash ≠ last-written hash AND source changed since last sync"
     *         → so operator-only-edit (source unchanged) is NOT a conflict by D-11 text; it flows as clean
     *         → BUT the cleaner-safer interpretation is: ANY dest-drift from last-written IS a conflict,
     *           because sync would overwrite operator's edit. Choose this safer reading for Phase 91.
     *       - Therefore: ANY destHash ≠ lastWrittenHashes[path] triggers conflict (regardless of source).
     *
     * D-13: Source-wins default — conflicts stop syncing until operator
     * runs `clawcode sync resolve <path> --side openclaw|clawcode`.
     */
    export function detectConflicts(
      lastWrittenHashes: Readonly<Record<string, string>>,
      currentCandidates: readonly FileHashPair[],
      now: Date,
    ): ConflictDetectionResult {
      const clean: string[] = [];
      const conflicts: SyncConflict[] = [];
      const nowIso = now.toISOString();

      for (const cand of currentCandidates) {
        const last = lastWrittenHashes[cand.path];

        // First-ever sync for this path — always clean
        if (last === undefined) { clean.push(cand.path); continue; }

        // Dest missing (deleted on ClawCode side OR never existed) — clean; rsync will re-create
        if (cand.destHash === null) { clean.push(cand.path); continue; }

        // Dest untouched since last write — safe
        if (cand.destHash === last) { clean.push(cand.path); continue; }

        // Dest drifted from last-written — operator edit detected. CONFLICT.
        conflicts.push(Object.freeze({
          path: cand.path,
          sourceHash: cand.sourceHash,
          destHash: cand.destHash,
          detectedAt: nowIso,
          resolvedAt: null,
        }));
      }

      return Object.freeze({
        cleanFiles: Object.freeze(clean),
        conflicts: Object.freeze(conflicts),
      });
    }
    ```

    ```ts
    import type { Logger } from "pino";
    import type { SyncConflict } from "./types.js";

    /** admin-clawdy channel from 91-CONTEXT §specifics. */
    export const ADMIN_CLAWDY_CHANNEL_ID = "1494117043367186474";

    export type ConflictAlertDeps = {
      readonly botToken: string;
      readonly channelId: string;
      readonly log: Logger;
      readonly fetchImpl?: typeof fetch;
    };

    export type ConflictAlertResult =
      | { readonly sent: true; readonly messageId: string }
      | { readonly sent: false; readonly reason: "no-conflicts" | "no-bot-token" | "http-error" | "network-error"; readonly detail?: string };

    /**
     * Send one embed summarizing all conflicts in this sync cycle to
     * admin-clawdy. Fire-and-forget — caller uses `void` + `.catch(log.warn)`.
     * D-15: one embed per cycle, not per file.
     */
    export async function sendConflictAlert(
      conflicts: readonly SyncConflict[],
      cycleId: string,
      deps: ConflictAlertDeps,
    ): Promise<ConflictAlertResult> {
      if (conflicts.length === 0) {
        return { sent: false, reason: "no-conflicts" };
      }
      if (!deps.botToken || deps.botToken.trim().length === 0) {
        deps.log.warn({ cycleId }, "conflict-alert: no bot token; skipping");
        return { sent: false, reason: "no-bot-token" };
      }

      const fetchFn = deps.fetchImpl ?? fetch;

      const fields = conflicts.slice(0, 25).map((c) => ({
        name: c.path.slice(0, 256),
        value: `src: \`${c.sourceHash.slice(0, 8)}\` · dest: \`${c.destHash.slice(0, 8)}\``,
        inline: false,
      }));

      const embed = {
        title: `Sync conflicts detected (${conflicts.length} file${conflicts.length === 1 ? "" : "s"})`,
        description:
          "The following files were edited on both sides since the last sync — ClawCode has stopped syncing them to prevent data loss.\n\n" +
          "Run `clawcode sync resolve <path> --side openclaw|clawcode` to resolve each.",
        fields,
        footer: { text: `cycle ${cycleId} · ${new Date().toISOString()}` },
        color: 15158332,   // red
      };

      const url = `https://discord.com/api/v10/channels/${deps.channelId}/messages`;
      let res: Response;
      try {
        res = await fetchFn(url, {
          method: "POST",
          headers: {
            "Authorization": `Bot ${deps.botToken}`,
            "Content-Type": "application/json",
          },
          body: JSON.stringify({ embeds: [embed] }),
        });
      } catch (err) {
        deps.log.warn({ err, cycleId }, "conflict-alert: network error");
        return { sent: false, reason: "network-error", detail: err instanceof Error ? err.message : String(err) };
      }

      if (!res.ok) {
        const body = await res.text().catch(() => "<unreadable>");
        deps.log.warn({ status: res.status, body, cycleId }, "conflict-alert: http error");
        return { sent: false, reason: "http-error", detail: `${res.status}: ${body.slice(0, 200)}` };
      }

      const json = (await res.json().catch(() => ({}))) as { id?: string };
      return { sent: true, messageId: json.id ?? "unknown" };
    }
    ```
