    import { readFile, stat, readdir } from "node:fs/promises";
    import { createHash } from "node:crypto";
    import { basename, join } from "node:path";
    import type { Logger } from "pino";
    import type { ConversationStore } from "../memory/store.js";   // adjust if path differs
    import { readTranslatorCursor, writeTranslatorCursor, type TranslatorCursorFile, type PerFileCursorEntry } from "./translator-cursor-store.js";

    export const MID_WRITE_SKIP_MS = 60_000;   // D-06 — skip files whose mtime < now - 60s

    /** sha256 prefix-16 for origin_id — mirrors Phase 80 memory-translator.sha256Hex. */
    export function sha256Hex(s: string): string {
      return createHash("sha256").update(s, "utf8").digest("hex");
    }

    /** D-09 origin_id shape: openclaw-session-<sha256(sessionId + turnIndex)-prefix16> */
    export function computeTurnOriginId(sessionId: string, turnIndex: number): string {
      const hashInput = `${sessionId}:${turnIndex}`;
      return `openclaw-session-${sha256Hex(hashInput).slice(0, 16)}`;
    }

    export type TranslatorRunOutcome = {
      readonly sessionsScanned: number;
      readonly sessionsSkippedMidWrite: number;
      readonly sessionsSkippedParseError: number;
      readonly turnsInserted: number;
      readonly turnsSkippedDuplicate: number;
      readonly turnsSkippedNonText: number;
      readonly durationMs: number;
    };

    export type TranslatorDeps = {
      readonly sessionsDir: string;                            // ~/.openclaw/agents/fin-acquisition/sessions
      readonly conversationStore: ConversationStore;
      readonly cursorPath: string;
      readonly agentName: string;                              // "fin-acquisition" — for ConversationStore associations
      readonly now?: () => Date;
      readonly log: Logger;
    };

    export async function translateAllSessions(deps: TranslatorDeps): Promise<TranslatorRunOutcome> {
      const start = (deps.now?.() ?? new Date()).getTime();
      const cursor = await readTranslatorCursor(deps.cursorPath, deps.log);
      const nowMs = deps.now?.().getTime() ?? Date.now();

      let sessionsScanned = 0, sessionsSkippedMidWrite = 0, sessionsSkippedParseError = 0;
      let turnsInserted = 0, turnsSkippedDuplicate = 0, turnsSkippedNonText = 0;

      const nextPerFileCursor: Record<string, PerFileCursorEntry> = { ...cursor.perFileCursor };

      let entries: string[];
      try {
        entries = (await readdir(deps.sessionsDir)).filter(n => n.endsWith(".jsonl"));
      } catch (err) {
        deps.log.warn({ err, sessionsDir: deps.sessionsDir }, "translator: sessions dir unreadable");
        return freezeOutcome({ sessionsScanned: 0, sessionsSkippedMidWrite: 0, sessionsSkippedParseError: 0, turnsInserted: 0, turnsSkippedDuplicate: 0, turnsSkippedNonText: 0, durationMs: Date.now() - start });
      }

      for (const name of entries.sort()) {
        const absPath = join(deps.sessionsDir, name);
        let st;
        try { st = await stat(absPath); } catch { continue; }
        if (!st.isFile()) continue;

        // D-06 — mid-write protection
        if (st.mtimeMs > nowMs - MID_WRITE_SKIP_MS) {
          sessionsSkippedMidWrite++;
          continue;
        }

        const existingCursor = cursor.perFileCursor[absPath];
        // If file size hasn't grown and mtime unchanged, nothing new.
        if (existingCursor && existingCursor.fileSize === st.size && existingCursor.mtime === st.mtime.toISOString()) {
          continue;
        }
        sessionsScanned++;

        let raw: string;
        try {
          raw = await readFile(absPath, "utf8");
        } catch (err) {
          deps.log.warn({ err, absPath }, "translator: file read failed; skipping");
          sessionsSkippedParseError++;
          continue;
        }

        const startLine = existingCursor?.lineCount ?? 0;
        const lines = raw.split("\n");
        let turnIndex = startLine;
        let sessionId = basename(name, ".jsonl");   // fallback — override from "session" line below

        for (let i = startLine; i < lines.length; i++) {
          const line = lines[i];
          if (!line || line.trim().length === 0) continue;
          let obj: unknown;
          try { obj = JSON.parse(line); }
          catch (err) {
            deps.log.warn({ err, absPath, lineIndex: i }, "translator: JSONL parse error on single line; skipping line");
            sessionsSkippedParseError++;
            continue;
          }
          const typed = obj as { type?: string; sessionId?: string; message?: { role?: string; content?: unknown } };

          // session-start line — capture sessionId if present
          if (typed.type === "session" && typeof typed.sessionId === "string") {
            sessionId = typed.sessionId;
            continue;
          }

          // D-08 — only role=user|assistant text. Drop tool_use, tool_result, thinking, custom, model-snapshot.
          if (typed.type !== "message" || !typed.message) {
            turnsSkippedNonText++;
            continue;
          }
          const role = typed.message.role;
          if (role !== "user" && role !== "assistant") {
            turnsSkippedNonText++;
            continue;
          }

          const text = extractTextContent(typed.message.content);
          if (text.length === 0) {
            turnsSkippedNonText++;
            continue;
          }

          const originId = computeTurnOriginId(sessionId, turnIndex);
          const inserted = await tryInsertTurn(deps, {
            agentName: deps.agentName,
            sessionId,
            role,
            content: text,
            turnIndex,
            originId,
            timestamp: st.mtime.toISOString(),   // best available — per-turn wall-clock not reliably recoverable from OpenClaw jsonl
          });
          if (inserted) turnsInserted++; else turnsSkippedDuplicate++;
          turnIndex++;
        }

        // Update per-file cursor
        nextPerFileCursor[absPath] = {
          byteOffset: Buffer.byteLength(raw, "utf8"),
          lineCount: lines.length,
          fileSize: st.size,
          mtime: st.mtime.toISOString(),
        };
      }

      const nextCursor: TranslatorCursorFile = {
        version: 1,
        lastScanAt: new Date(nowMs).toISOString(),
        perFileCursor: nextPerFileCursor,
      };
      await writeTranslatorCursor(deps.cursorPath, nextCursor, deps.log);

      return freezeOutcome({
        sessionsScanned, sessionsSkippedMidWrite, sessionsSkippedParseError,
        turnsInserted, turnsSkippedDuplicate, turnsSkippedNonText,
        durationMs: Date.now() - start,
      });
    }

    function freezeOutcome(o: TranslatorRunOutcome): TranslatorRunOutcome { return Object.freeze(o); }

    /** Extract text from user/assistant message content. String → passthrough. Array → concat text blocks only. */
    export function extractTextContent(content: unknown): string {
      if (typeof content === "string") return content;
      if (!Array.isArray(content)) return "";
      const parts: string[] = [];
      for (const block of content) {
        if (block && typeof block === "object" && (block as { type?: string }).type === "text") {
          const text = (block as { text?: unknown }).text;
          if (typeof text === "string") parts.push(text);
        }
        // D-08 — ignore tool_use, tool_result, thinking, etc.
      }
      return parts.join("\n");
    }

    /**
     * Insert turn into ConversationStore; return true if fresh insert, false if origin_id collision.
     * Exact method signature depends on ConversationStore — adjust to match store.ts at implementation time.
     * The CRITICAL invariant: origin_id UNIQUE must gate insertion — duplicate origin_ids result in no-op (skipped).
     */
    async function tryInsertTurn(deps: TranslatorDeps, turn: {
      agentName: string; sessionId: string; role: "user" | "assistant"; content: string;
      turnIndex: number; originId: string; timestamp: string;
    }): Promise<boolean> {
      try {
        // Pseudocode — replace with actual ConversationStore API discovered during implementation:
        //   const result = deps.conversationStore.insertTurn({ ...turn, origin_id: turn.originId });
        //   return result.inserted;
        // If ConversationStore has no dedicated method, check for conversation_turns table access
        // and use `INSERT OR IGNORE` SQL via the store's prepared-statement API.
        const result = (deps.conversationStore as any).insertTurn?.({
          agentName: turn.agentName,
          sessionId: turn.sessionId,
          role: turn.role,
          content: turn.content,
          turnIndex: turn.turnIndex,
          origin_id: turn.originId,
          timestamp: turn.timestamp,
        });
        // Phase 80 convention: returned entry.embedding === null signals collision.
        if (result && typeof result === "object") {
          return !("embedding" in result && result.embedding === null);
        }
        return true;
      } catch (err) {
        deps.log.warn({ err, originId: turn.originId }, "translator: turn insert failed (non-blocking)");
        return false;
      }
    }
