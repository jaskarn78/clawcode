    ```ts
    import { readFile } from "node:fs/promises";
    import pino from "pino";
    import type { Logger } from "pino";
    import { DEFAULT_SYNC_STATE_PATH, readSyncState } from "../../sync/sync-state-store.js";
    import { cliLog, cliError } from "../output.js";

    export const DEFAULT_SYNC_JSONL_PATH = `${process.env.HOME}/.clawcode/manager/sync.jsonl`;

    export type RunSyncStatusArgs = Readonly<{
      syncStatePath?: string;
      syncJsonlPath?: string;
      log?: Logger;
    }>;

    export async function runSyncStatusAction(args: RunSyncStatusArgs): Promise<number> {
      const log = args.log ?? (pino({ level: "info" }) as unknown as Logger);
      const statePath = args.syncStatePath ?? DEFAULT_SYNC_STATE_PATH;
      const jsonlPath = args.syncJsonlPath ?? DEFAULT_SYNC_JSONL_PATH;

      const state = await readSyncState(statePath, log);

      let lastEntry: unknown = null;
      try {
        const raw = await readFile(jsonlPath, "utf8");
        const lines = raw.trim().split("\n").filter(Boolean);
        if (lines.length > 0) {
          lastEntry = JSON.parse(lines[lines.length - 1]!);
        }
      } catch {
        // missing or empty — OK
      }

      cliLog(JSON.stringify({
        authoritativeSide: state.authoritativeSide,
        lastSyncedAt: state.lastSyncedAt,
        conflictCount: state.conflicts.length,
        conflicts: state.conflicts.map(c => ({ path: c.path, detectedAt: c.detectedAt })),
        lastCycle: lastEntry,
      }, null, 2));
      return 0;
    }

    export function registerSyncStatusCommand(parent: Command): void {
      parent.command("status").description("Print sync state + last cycle outcome").action(async () => {
        process.exit(await runSyncStatusAction({}));
      });
    }
    ```

    ```ts
    import type { Command } from "commander";
    import { createHash } from "node:crypto";
    import { readFile } from "node:fs/promises";
    import { execa } from "execa";
    import pino from "pino";
    import type { Logger } from "pino";
    import { homedir } from "node:os";
    import { join } from "node:path";
    import { DEFAULT_SYNC_STATE_PATH, readSyncState, writeSyncState } from "../../sync/sync-state-store.js";
    import { cliLog, cliError } from "../output.js";

    export type RunSyncResolveArgs = Readonly<{
      path: string;                              // relpath under workspace root
      side: "openclaw" | "clawcode";
      syncStatePath?: string;
      log?: Logger;
      runSsh?: (args: readonly string[]) => Promise<{ exitCode: number; stdout: string; stderr: string }>;
      runRsync?: (args: readonly string[]) => Promise<{ exitCode: number; stdout: string; stderr: string }>;
      readFileImpl?: typeof readFile;
    }>;

    export async function runSyncResolveAction(args: RunSyncResolveArgs): Promise<number> {
      const log = args.log ?? (pino({ level: "info" }) as unknown as Logger);
      const statePath = args.syncStatePath ?? DEFAULT_SYNC_STATE_PATH;
      const state = await readSyncState(statePath, log);

      const existing = state.conflicts.find((c) => c.path === args.path && c.resolvedAt === null);
      if (!existing) {
        cliError(`No unresolved conflict for path '${args.path}'`);
        return 1;
      }

      // D-14: copy chosen side → other side
      if (args.side === "openclaw") {
        // pull openclaw/<path> to clawcode/<path>
        const src = `${state.openClawHost}:${state.openClawWorkspace}/${args.path}`;
        const dst = `${state.clawcodeWorkspace}/${args.path}`;
        const result = await (args.runRsync ?? ((a) => execa("rsync", a, { reject: false })))([
          "-av", "--inplace", "--timeout=60",
          "-e", "ssh -o BatchMode=yes -o ConnectTimeout=10",
          src, dst,
        ]);
        if (result.exitCode !== 0) {
          cliError(`rsync pull failed: ${result.stderr}`);
          return 1;
        }
      } else {
        // push clawcode/<path> to openclaw/<path>
        const src = `${state.clawcodeWorkspace}/${args.path}`;
        const dst = `${state.openClawHost}:${state.openClawWorkspace}/${args.path}`;
        const result = await (args.runRsync ?? ((a) => execa("rsync", a, { reject: false })))([
          "-av", "--inplace", "--timeout=60",
          "-e", "ssh -o BatchMode=yes -o ConnectTimeout=10",
          src, dst,
        ]);
        if (result.exitCode !== 0) {
          cliError(`rsync push failed: ${result.stderr}`);
          return 1;
        }
      }

      // Compute new sha256 of the now-identical file (read from clawcode side — local)
      const readImpl = args.readFileImpl ?? readFile;
      const absLocal = join(state.clawcodeWorkspace, args.path);
      const buf = await readImpl(absLocal);
      const newHash = createHash("sha256").update(buf).digest("hex");

      // Update state: perFileHashes[path] = newHash, remove conflict entry
      const nextState = {
        ...state,
        updatedAt: new Date().toISOString(),
        perFileHashes: { ...state.perFileHashes, [args.path]: newHash },
        conflicts: state.conflicts.filter(
          (c) => !(c.path === args.path && c.resolvedAt === null),
        ),
      };
      await writeSyncState(statePath, nextState, log);

      cliLog(`Resolved '${args.path}' using ${args.side} side. Sync resumed for this file.`);
      return 0;
    }

    export function registerSyncResolveCommand(parent: Command): void {
      parent
        .command("resolve <path>")
        .description("Resolve a sync conflict by choosing which side wins")
        .requiredOption("--side <side>", "openclaw | clawcode")
        .action(async (path: string, opts: { side: string }) => {
          if (opts.side !== "openclaw" && opts.side !== "clawcode") {
            cliError(`--side must be 'openclaw' or 'clawcode'`);
            process.exit(1);
          }
          process.exit(await runSyncResolveAction({ path, side: opts.side }));
        });
    }
    ```

    ```ts
    import type { Command } from "commander";
    import { createInterface } from "node:readline/promises";
    import pino from "pino";
    import type { Logger } from "pino";
    import { DEFAULT_SYNC_STATE_PATH, readSyncState, writeSyncState } from "../../sync/sync-state-store.js";
    import { syncOnce, type SyncRunnerDeps } from "../../sync/sync-runner.js";
    import { cliLog, cliError } from "../output.js";

    export const ROLLBACK_WINDOW_MS = 7 * 24 * 60 * 60 * 1000;

    export type RunSyncSetAuthoritativeArgs = Readonly<{
      side: "openclaw" | "clawcode";
      confirmCutover?: boolean;
      revertCutover?: boolean;
      forceRollback?: boolean;
      syncStatePath?: string;
      log?: Logger;
      runSyncOnce?: (deps: SyncRunnerDeps) => Promise<unknown>;
      makeSyncDeps?: () => SyncRunnerDeps;
      promptConfirm?: (q: string) => Promise<boolean>;
      now?: () => Date;
    }>;

    export async function runSyncSetAuthoritativeAction(args: RunSyncSetAuthoritativeArgs): Promise<number> {
      const log = args.log ?? (pino({ level: "info" }) as unknown as Logger);
      const statePath = args.syncStatePath ?? DEFAULT_SYNC_STATE_PATH;
      const state = await readSyncState(statePath, log);
      const now = args.now?.() ?? new Date();

      // D-17 — flip TO clawcode requires --confirm-cutover
      if (args.side === "clawcode") {
        if (state.authoritativeSide === "clawcode") {
          cliError("Already authoritative: clawcode");
          return 1;
        }
        if (!args.confirmCutover) {
          cliError(
            "Flipping to clawcode is destructive — pass --confirm-cutover to proceed. " +
            "This will pause the 5-min timer, drain OpenClaw→ClawCode one last time, and flip the flag.",
          );
          return 1;
        }

        // D-17 step 2 — synchronous drain
        cliLog("Draining OpenClaw → ClawCode...");
        const syncDeps = args.makeSyncDeps?.();
        const runner = args.runSyncOnce ?? (syncDeps ? ((d) => syncOnce(d)) : undefined);
        if (!runner || !syncDeps) {
          cliError("Cannot drain — makeSyncDeps not provided (this is an injection bug; normal CLI path provides defaults)");
          return 1;
        }
        const outcome = await runner(syncDeps);
        cliLog(`Drain outcome: ${JSON.stringify(outcome)}`);

        // D-21 — atomic mid-drain verification: compare perFileHashes before + after; if any source
        // file changed during drain, loop back and drain again (max 3 iterations to prevent infinite loop)
        // Simplified: read state again; if lastSyncedAt advanced (new cycle happened) proceed. If the
        // runner returned 'partial-conflicts' or 'failed-*', abort the flip.
        if (typeof outcome === "object" && outcome !== null && "kind" in outcome) {
          const kind = (outcome as { kind: string }).kind;
          if (kind === "failed-ssh" || kind === "failed-rsync") {
            cliError(`Drain failed (${kind}). Flip aborted. Resolve and retry.`);
            return 1;
          }
          if (kind === "partial-conflicts") {
            cliError("Drain completed with conflicts. Resolve via 'clawcode sync resolve ...' before cutover.");
            return 1;
          }
        }

        // D-17 step 3 — operator confirmation
        const prompt = args.promptConfirm ?? defaultPrompt;
        const ok = await prompt(`Drain complete. Flip authoritative to 'clawcode'? (y/N) `);
        if (!ok) {
          cliLog("Aborted — state unchanged.");
          return 0;
        }

        // D-17 step 4 — flip
        const nextState = { ...state, authoritativeSide: "clawcode" as const, updatedAt: now.toISOString() };
        await writeSyncState(statePath, nextState, log);
        cliLog("authoritativeSide flipped → clawcode. 5-min timer is now no-op (opt in with `clawcode sync start --reverse`).");
        return 0;
      }

      // D-19 — flip back to openclaw (revert)
      if (args.side === "openclaw") {
        if (state.authoritativeSide === "openclaw") {
          cliError("Already authoritative: openclaw");
          return 1;
        }
        if (!args.revertCutover && !args.forceRollback) {
    ```
