    **Create `src/sync/sync-runner.ts`** — pure DI module. Signature:

    ```ts
    import { execa } from "execa";
    import { createHash } from "node:crypto";
    import { readFile, stat, readdir } from "node:fs/promises";
    import { join } from "node:path";
    import type { Logger } from "pino";
    import { nanoid } from "nanoid";
    import type { SyncStateFile, SyncRunOutcome, SyncConflict } from "./types.js";
    import { readSyncState, writeSyncState } from "./sync-state-store.js";

    export type SyncRunnerDeps = {
      readonly syncStatePath: string;
      readonly filterFilePath: string;       // path to clawcode-sync-filter.txt
      readonly syncJsonlPath: string;        // ~/.clawcode/manager/sync.jsonl
      readonly now?: () => Date;             // DI for tests
      readonly runRsync?: (args: readonly string[]) => Promise<{ stdout: string; stderr: string; exitCode: number }>;
      readonly appendJsonl?: (path: string, entry: object) => Promise<void>;
      readonly log: Logger;
    };

    export async function syncOnce(deps: SyncRunnerDeps): Promise<SyncRunOutcome> {
      const cycleId = nanoid();
      const start = deps.now?.() ?? new Date();

      const state = await readSyncState(deps.syncStatePath, deps.log);

      // D-01 + D-18: if authoritativeSide === "clawcode" AND reverse not opted in (future Plan 91-04), return paused
      if (state.authoritativeSide === "clawcode") {
        const outcome: SyncRunOutcome = { kind: "paused", cycleId, reason: "authoritative-is-clawcode-no-reverse-opt-in" };
        await appendOutcomeToJsonl(deps, outcome, start);
        return outcome;
      }

      // Build rsync command — MEMORY.md sync FIRST (91-CONTEXT §specifics: highest priority) then full dir sync.
      // Use rsync -av --filter-file=<filterPath> --delete --partial --inplace --itemize-changes --stats
      //   jjagpal@100.71.14.96:~/.openclaw/workspace-finmentum/ /home/clawcode/.clawcode/agents/finmentum/
      const rsyncArgs: string[] = [
        "-av",
        "--filter", `merge ${deps.filterFilePath}`,
        "--delete",
        "--partial",
        "--inplace",
        "--itemize-changes",
        "--stats",
        "--timeout=120",
        "-e", "ssh -o BatchMode=yes -o ConnectTimeout=10 -o StrictHostKeyChecking=accept-new",
        `${state.openClawHost}:${state.openClawWorkspace}/`,
        `${state.clawcodeWorkspace}/`,
      ];

      const runner = deps.runRsync ?? (async (args) => execa("rsync", args, { reject: false }));
      let result: { stdout: string; stderr: string; exitCode: number };
      try {
        result = await runner(rsyncArgs);
      } catch (err) {
        // SSH/network failure before rsync starts
        const outcome: SyncRunOutcome = {
          kind: "failed-ssh",
          cycleId,
          error: err instanceof Error ? err.message : String(err),
          durationMs: Date.now() - start.getTime(),
        };
        await appendOutcomeToJsonl(deps, outcome, start);
        deps.log.warn({ err }, "sync ssh failed");
        return outcome;
      }

      // Parse rsync --stats output:
      //   "Number of regular files transferred: N"
      //   "Total transferred file size: N bytes"
      // Parse --itemize-changes lines: ">f+++++++++ MEMORY.md" (added), ">f.st...... path" (updated), "*deleting path" (removed)
      const parsed = parseRsyncStats(result.stdout);

      if (result.exitCode !== 0 && result.exitCode !== 23 /* partial — conflict skip acceptable */) {
        const outcome: SyncRunOutcome = {
          kind: "failed-rsync",
          cycleId,
          error: result.stderr.slice(0, 4000),
          durationMs: Date.now() - start.getTime(),
          exitCode: result.exitCode,
        };
        await appendOutcomeToJsonl(deps, outcome, start);
        return outcome;
      }

      if (parsed.filesAdded + parsed.filesUpdated + parsed.filesRemoved === 0) {
        const outcome: SyncRunOutcome = { kind: "skipped-no-changes", cycleId, durationMs: Date.now() - start.getTime() };
        await appendOutcomeToJsonl(deps, outcome, start);
        return outcome;
      }

      // Update perFileHashes for files touched this cycle by computing sha256 of destination copies.
      // (Conflict detection added in Plan 91-02 — this plan only tracks destination hashes as the
      // "last-written" baseline. No skip logic yet; Plan 91-02 layers conflict detection on top.)
      const updatedHashes = { ...state.perFileHashes };
      for (const relpath of parsed.touchedPaths) {
        try {
          const absDest = join(state.clawcodeWorkspace, relpath);
          const buf = await readFile(absDest);
          updatedHashes[relpath] = createHash("sha256").update(buf).digest("hex");
        } catch { /* deleted or unreadable — drop from hash map */ delete updatedHashes[relpath]; }
      }

      const nextState: SyncStateFile = {
        ...state,
        updatedAt: start.toISOString(),
        lastSyncedAt: start.toISOString(),
        perFileHashes: updatedHashes,
      };
      await writeSyncState(deps.syncStatePath, nextState, deps.log);

      const outcome: SyncRunOutcome = {
        kind: "synced",
        cycleId,
        filesAdded: parsed.filesAdded,
        filesUpdated: parsed.filesUpdated,
        filesRemoved: parsed.filesRemoved,
        filesSkippedConflict: 0,   // Plan 91-02 fills this in
        bytesTransferred: parsed.bytesTransferred,
        durationMs: Date.now() - start.getTime(),
      };
      await appendOutcomeToJsonl(deps, outcome, start);
      return outcome;
    }

    export function parseRsyncStats(stdout: string): {
      filesAdded: number;
      filesUpdated: number;
      filesRemoved: number;
      bytesTransferred: number;
      touchedPaths: readonly string[];
    } {
      // Implementation detail — parse --itemize-changes + --stats verbatim. Use regexes:
      //   /^>f\+\+\+\+\+\+\+\+\+ (.+)$/m  → added
      //   /^>f[.c][sc][tT][.p][og][.u][.a][.x]\.* (.+)$/m → updated (">f.st......")
      //   /^\*deleting (.+)$/m  → removed
      //   /Total transferred file size: ([\d,]+) bytes/ → bytes (strip commas)
    }

    async function appendOutcomeToJsonl(deps: SyncRunnerDeps, outcome: SyncRunOutcome, start: Date): Promise<void> {
      const entry = {
        timestamp: start.toISOString(),
        cycleId: outcome.cycleId,
        direction: "openclaw-to-clawcode" as const,
        ...outcome,
      };
      const appender = deps.appendJsonl ?? (async (p, e) => {
        const { appendFile, mkdir } = await import("node:fs/promises");
        const { dirname } = await import("node:path");
        await mkdir(dirname(p), { recursive: true });
        await appendFile(p, JSON.stringify(e) + "\n", "utf8");
      });
      try {
        await appender(deps.syncJsonlPath, entry);
      } catch (err) {
        deps.log.warn({ err }, "sync.jsonl append failed");
      }
    }
    ```
