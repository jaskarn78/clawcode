       ```typescript
       import { CANONICAL_SEGMENTS, type CanonicalSegment, type PercentileRow } from "./types.js";

       /** SLO target metric: which percentile is the watch line. */
       export type SloMetric = "p50" | "p95" | "p99";

       /** Status of a percentile row vs its SLO threshold. */
       export type SloStatus = "healthy" | "breach" | "no_data";

       /** A single SLO target: which segment, which percentile, and the max ms before breach. */
       export type SloEntry = {
         readonly segment: CanonicalSegment;
         readonly metric: SloMetric;
         readonly thresholdMs: number;
       };

       /**
        * Default SLO targets for v1.7. Single source of truth — imported by
        * the daemon's latency IPC handler (for slo_status augmentation) and
        * by the CI regression gate (src/benchmarks/thresholds.ts) so the
        * dashboard and CI never disagree.
        *
        * Frozen at module load. Override per-agent via clawcode.yaml
        * `perf.slos: [...]` which merges through `mergeSloOverrides`.
        */
       export const DEFAULT_SLOS: readonly SloEntry[] = Object.freeze([
         Object.freeze({ segment: "end_to_end",       metric: "p95", thresholdMs: 6000 }),
         Object.freeze({ segment: "first_token",      metric: "p50", thresholdMs: 2000 }),
         Object.freeze({ segment: "context_assemble", metric: "p95", thresholdMs: 300 }),
         Object.freeze({ segment: "tool_call",        metric: "p95", thresholdMs: 1500 }),
       ]);

       export function evaluateSloStatus(
         row: Pick<PercentileRow, "p50" | "p95" | "p99" | "count">,
         thresholdMs: number,
         metric: SloMetric,
       ): SloStatus {
         if (row.count === 0) return "no_data";
         const value = row[metric];
         if (value === null) return "no_data";
         return value <= thresholdMs ? "healthy" : "breach";
       }

       /**
        * Merge user overrides into DEFAULT_SLOS.
        *
        * Per-segment + per-metric override semantics: an override matching
        * BOTH segment and metric replaces the threshold. An override matching
        * the segment but a different metric is APPENDED (a segment may carry
        * multiple SLOs in future, e.g. p50 first_token + p95 first_token).
        */
       export function mergeSloOverrides(
         defaults: readonly SloEntry[],
         overrides: readonly SloEntry[],
       ): readonly SloEntry[] {
         const result: SloEntry[] = [];
         const consumed = new Set<number>();
         for (const def of defaults) {
           const match = overrides.findIndex(
             (o) => o.segment === def.segment && o.metric === def.metric,
           );
           if (match >= 0) {
             const o = overrides[match]!;
             result.push(Object.freeze({ segment: def.segment, metric: def.metric, thresholdMs: o.thresholdMs }));
             consumed.add(match);
           } else {
             result.push(def);
           }
         }
         for (let i = 0; i < overrides.length; i++) {
           if (!consumed.has(i)) result.push(Object.freeze({ ...overrides[i]! }));
         }
         return Object.freeze(result);
       }
       ```

    2. CREATE `src/benchmarks/thresholds.ts`:
       ```typescript
       import { readFileSync } from "node:fs";
       import { parse as parseYaml } from "yaml";
       import { z } from "zod/v4";

       import { BenchmarkConfigError, type Baseline, type BenchReport } from "./types.js";
       import { CANONICAL_SEGMENTS, type CanonicalSegment } from "../performance/types.js";

       const segmentEnum = z.enum(["end_to_end", "first_token", "context_assemble", "tool_call"]);

       /** Per-segment override allowing absolute floor (for noisy segments). */
       const segmentOverrideSchema = z.object({
         segment: segmentEnum,
         p95MaxDeltaPct: z.number().nonnegative().optional(),
         p95MaxDeltaMs: z.number().nonnegative().optional(),
       });

       /** Top-level thresholds.yaml shape. */
       export const thresholdsSchema = z.object({
         defaultP95MaxDeltaPct: z.number().nonnegative().default(20),
         segments: z.array(segmentOverrideSchema).default([]),
       });

       export type ThresholdsConfig = z.infer<typeof thresholdsSchema>;
       export type SegmentOverride = z.infer<typeof segmentOverrideSchema>;

       /**
        * One regression entry. `deltaPct` is computed as
        * (currentMs - baselineMs) / baselineMs * 100.
        */
       export type Regression = {
         readonly segment: CanonicalSegment;
         readonly baselineMs: number;
         readonly currentMs: number;
         readonly deltaPct: number;
         readonly thresholdPct: number;
       };

       export type RegressionResult = {
         readonly regressions: readonly Regression[];
         readonly status: "clean" | "regressed";
       };

       /**
        * Load thresholds.yaml. Throws BenchmarkConfigError on missing file,
        * unparseable YAML, or schema violations.
        */
       export function loadThresholds(path: string): ThresholdsConfig {
         let raw: string;
         try {
           raw = readFileSync(path, "utf-8");
         } catch (err) {
           const msg = err instanceof Error ? err.message : "unknown";
           throw new BenchmarkConfigError(`read failed: ${msg}`, path);
         }
         let parsed: unknown;
         try {
           parsed = parseYaml(raw);
         } catch (err) {
           const msg = err instanceof Error ? err.message : "unknown";
           throw new BenchmarkConfigError(`yaml parse failed: ${msg}`, path);
         }
         const result = thresholdsSchema.safeParse(parsed ?? {});
         if (!result.success) {
           throw new BenchmarkConfigError(
             `schema invalid: ${result.error.issues.map((i) => i.message).join("; ")}`,
             path,
           );
         }
         return Object.freeze({
           defaultP95MaxDeltaPct: result.data.defaultP95MaxDeltaPct,
           segments: Object.freeze(result.data.segments.map((s) => Object.freeze(s))) as readonly SegmentOverride[],
         });
       }

       function findOverride(thresholds: ThresholdsConfig, segment: CanonicalSegment): SegmentOverride | undefined {
         return thresholds.segments.find((s) => s.segment === segment);
       }

       /**
        * Compare a fresh bench report against a baseline using per-segment
        * thresholds. Skips segments where either side has count === 0.
        * Honors per-segment p95MaxDeltaMs absolute floor for noisy segments.
        */
       export function evaluateRegression(
         report: BenchReport,
         baseline: Baseline,
         thresholds: ThresholdsConfig,
       ): RegressionResult {
         const regressions: Regression[] = [];
         const baselineBySeg = new Map(baseline.overall_percentiles.map((r) => [r.segment, r]));
         for (const reportRow of report.overall_percentiles) {
           const baseRow = baselineBySeg.get(reportRow.segment);
           if (!baseRow) continue;
           if (baseRow.count === 0 || reportRow.count === 0) continue;
           if (baseRow.p95 === null || reportRow.p95 === null) continue;
           if (baseRow.p95 === 0) continue; // avoid div-by-zero; treat as no comparable baseline
           const override = findOverride(thresholds, reportRow.segment as CanonicalSegment);
           const thresholdPct = override?.p95MaxDeltaPct ?? thresholds.defaultP95MaxDeltaPct;
           const deltaMs = reportRow.p95 - baseRow.p95;
           const deltaPct = (deltaMs / baseRow.p95) * 100;
           if (deltaPct <= thresholdPct) continue;
           // Per-segment absolute floor escape hatch
       ```
