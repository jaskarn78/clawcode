<task type="auto" name="T-03: UsageRefreshPoller class" tdd="true">
  <files>src/usage/usage-refresh-poller.ts (new)</files>
  <action>
    ```ts
    import type { RateLimitTracker } from "./rate-limit-tracker.js";

    export interface UsageRefreshPollerDeps {
      readonly tracker: RateLimitTracker;
      readonly fetchCurrentUsage: () => Promise<SDKRateLimitInfo[] | null>;
      readonly intervalMs: number;
      readonly agentName: string;
      readonly logger?: { info(obj: unknown): void; warn(obj: unknown): void };
    }

    export class UsageRefreshPoller {
      private timer: NodeJS.Timeout | null = null;
      constructor(private readonly deps: UsageRefreshPollerDeps) {}

      start(): void {
        if (this.timer !== null) return;
        this.timer = setInterval(() => void this.pollNow(), this.deps.intervalMs);
      }

      stop(): void {
        if (this.timer === null) return;
        clearInterval(this.timer);
        this.timer = null;
      }

      async pollNow(): Promise<void> {
        try {
          const fresh = await this.deps.fetchCurrentUsage();
          if (fresh === null) return; // no signal — skip
          for (const info of fresh) {
            this.deps.tracker.recordSnapshot(info); // existing tracker API
          }
          this.deps.logger?.info({ key: "phase128-usage-refresh", agent: this.deps.agentName, count: fresh.length, source: "poll" });
        } catch (err) {
          this.deps.logger?.warn({ key: "phase128-usage-refresh-fail", agent: this.deps.agentName, err: String(err) });
        }
      }
    }
    ```

    `fetchCurrentUsage` is the injection point — Plan adapts based on T-01 SURVEY outcome.
