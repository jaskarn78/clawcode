    ```typescript
    const configWatcher = new ConfigWatcher({
      configPath,
      auditTrailPath,
      onChange: async (diff, newResolvedAgents) => {
        // Phase 104 — walk the diff for op:// URI changes and update the
        // SecretsResolver cache accordingly. Ordering matters: invalidate FIRST
        // (so a stale value can't be returned between invalidate + resolve),
        // then warm-resolve any new op:// URIs.
        for (const change of diff.changes) {
          const oldVal = change.oldValue;
          const newVal = change.newValue;
          const oldIsOp = typeof oldVal === "string" && oldVal.startsWith("op://") && oldVal.length > "op://".length;
          const newIsOp = typeof newVal === "string" && newVal.startsWith("op://") && newVal.length > "op://".length;
          if (oldIsOp && oldVal !== newVal) {
            secretsResolver.invalidate(oldVal as string);
          }
          if (newIsOp) {
            try {
              await secretsResolver.resolve(newVal as string);
            } catch (err) {
              // SecretsResolver already logged the failure with structured fields.
              // Don't crash the reload — the affected MCP will degrade via the
              // same onMcpResolutionError path used at boot.
              const reason = err instanceof Error ? err.message : String(err);
              log.warn({ uri: newVal, reason }, "secrets: warm-resolve after config change failed");
            }
          }
        }
        const summary = await configReloader.applyChanges(diff, newResolvedAgents);
        log.info(
          { subsystems: summary.subsystemsReloaded, agents: summary.agentsAffected },
          "config hot-reloaded",
        );
      },
      log,
      opRefResolver: cachedOpRefResolver,  // unchanged from Wave 2
    });
    ```

        invalidate,

      // position to verify ordering).
