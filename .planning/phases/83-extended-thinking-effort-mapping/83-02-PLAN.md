<objective>
Make the effort override survive `clawcode restart <agent>` (EFFORT-03) and prevent cost-spike propagation into forked Opus advisor sessions (EFFORT-06). Two orthogonal concerns live together because both touch the runtime-vs-config boundary for effort state and both land on top of Plan 01's SDK wiring.

Purpose: The P0 fix alone would regress to the config default on every restart — a lurking UX bug. Fork quarantine prevents a v1.5 escalation at `effort=max` from launching an Opus advisor that inherits the high budget, blowing out the escalation cost cap (PITFALLS §Pitfall 3).
Output: Per-agent runtime effort state persisted to disk; `startAgent` re-applies it; `buildForkConfig` explicitly resets effort to the parent's CONFIG default (not its runtime override).
</objective>

       ```typescript
       import { readFile, writeFile, mkdir, rename } from "node:fs/promises";
       import { dirname, join } from "node:path";
       import { tmpdir } from "node:os";
       import { randomBytes } from "node:crypto";
       import type { Logger } from "pino";
       import { effortSchema, type EffortLevel } from "../config/schema.js";
       import { z } from "zod/v4";

       /**
        * Phase 83 EFFORT-03 — per-agent runtime effort persistence across daemon
        * restart. Persistence file lives alongside the existing registry.json
        * at ~/.clawcode/manager/effort-state.json. Shape is versioned + atomic.
        *
        * Invariants:
        *   - Missing file → null for every agent (no error)
        *   - Corrupt JSON → null + warn (do not crash the daemon)
        *   - Unknown / invalid level for a recorded agent → null for that agent,
        *     other agents still readable
        *   - writeEffortState uses temp+rename for atomicity
        */

       const effortStateFileSchema = z.object({
         version: z.literal(1),
         updatedAt: z.string(),
         agents: z.record(z.string(), effortSchema),
       });

       export type EffortStateFile = z.infer<typeof effortStateFileSchema>;

       export async function readEffortState(
         filePath: string,
         agentName: string,
         log?: Logger,
       ): Promise<EffortLevel | null> {
         try {
           const raw = await readFile(filePath, "utf8");
           const parsed = effortStateFileSchema.safeParse(JSON.parse(raw));
           if (!parsed.success) {
             log?.warn({ filePath, error: parsed.error.message }, "effort-state file invalid, ignoring");
             return null;
           }
           const level = parsed.data.agents[agentName];
           if (!level) return null;
           return level;
         } catch (err) {
           const msg = err instanceof Error ? err.message : String(err);
           // ENOENT is the common / expected path — debug-level only.
           if ((err as NodeJS.ErrnoException).code === "ENOENT") return null;
           log?.warn({ filePath, error: msg }, "effort-state read failed");
           return null;
         }
       }

       export async function writeEffortState(
         filePath: string,
         agentName: string,
         level: EffortLevel,
         log?: Logger,
       ): Promise<void> {
         // Read existing, merge this agent, write back atomically.
         let existing: EffortStateFile = { version: 1, updatedAt: "", agents: {} };
         try {
           const raw = await readFile(filePath, "utf8");
           const parsed = effortStateFileSchema.safeParse(JSON.parse(raw));
           if (parsed.success) existing = parsed.data;
         } catch { /* missing / corrupt — start fresh */ }

         const next: EffortStateFile = {
           version: 1,
           updatedAt: new Date().toISOString(),
           agents: { ...existing.agents, [agentName]: level },
         };
         await mkdir(dirname(filePath), { recursive: true });
         const tmp = `${filePath}.${randomBytes(6).toString("hex")}.tmp`;
         await writeFile(tmp, JSON.stringify(next, null, 2), "utf8");
         await rename(tmp, filePath);
         log?.debug({ agent: agentName, level }, "effort-state persisted");
       }

       export async function clearEffortState(
         filePath: string,
         agentName: string,
       ): Promise<void> {
         try {
           const raw = await readFile(filePath, "utf8");
           const parsed = effortStateFileSchema.safeParse(JSON.parse(raw));
           if (!parsed.success) return;
           const next: EffortStateFile = {
             version: 1,
             updatedAt: new Date().toISOString(),
             agents: Object.fromEntries(
               Object.entries(parsed.data.agents).filter(([k]) => k !== agentName),
             ),
           };
           const tmp = `${filePath}.${randomBytes(6).toString("hex")}.tmp`;
           await writeFile(tmp, JSON.stringify(next, null, 2), "utf8");
           await rename(tmp, filePath);
         } catch { /* nothing to clear */ }
       }

       export const DEFAULT_EFFORT_STATE_PATH =
         join(process.env.HOME ?? tmpdir(), ".clawcode", "manager", "effort-state.json");
       ```
