    ```typescript
    export async function rerankTop<T extends { content: string }>(
      query: string,
      candidates: T[],
      opts: { topK: number; timeoutMs?: number; logger?: Logger }
    ): Promise<T[]> {
      if (candidates.length === 0) return candidates;
      const timeoutMs = opts.timeoutMs ?? 500;
      try {
        const p = await loadReranker();
        const inputs = candidates.map(c => ({ text: query, text_pair: c.content }));
        const scoresPromise = p(inputs);
        const scores = await Promise.race([
          scoresPromise,
          new Promise((_, rej) => setTimeout(() => rej(new Error('reranker-timeout')), timeoutMs)),
        ]) as Array<{ score: number }>;
        const scored = candidates.map((c, i) => ({ c, s: scores[i].score }));
        scored.sort((a, b) => b.s - a.s);
        opts.logger?.info({ phase: 'phase101-ingest', event: 'reranker-applied', n: candidates.length, kept: Math.min(opts.topK, candidates.length) });
        return scored.slice(0, opts.topK).map(x => x.c);
      } catch (err) {
        opts.logger?.warn({ phase: 'phase101-ingest', event: 'reranker-fallback', reason: (err as Error).message });
        return candidates.slice(0, opts.topK);
      }
    }
    ```
