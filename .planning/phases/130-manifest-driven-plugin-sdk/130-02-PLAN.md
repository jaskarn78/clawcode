<task type="auto" name="T-02: skill-loader.ts chokepoint module">

  <action>
    ```ts
    import fs from "node:fs";
    import path from "node:path";
    import matter from "gray-matter"; // verify already in deps; if not, use existing SKILL.md parser
    import { SkillManifestSchema, type SkillManifest } from "../plugin-sdk/index.js";

    export interface LoadSkillManifestResult {
      manifest: SkillManifest | null;       // null when no frontmatter / refused
      status: "loaded" | "refused-mcp-missing" | "manifest-missing" | "parse-error";
      reason?: string;
      missingMcp?: string[];
    }

    export function loadSkillManifest(
      skillDir: string,
      enabledMcpServers: string[],
    ): LoadSkillManifestResult {
      const skillMdPath = path.join(skillDir, "SKILL.md");
      if (!fs.existsSync(skillMdPath)) {
        return { manifest: null, status: "manifest-missing" };
      }

      const raw = fs.readFileSync(skillMdPath, "utf-8");
      const parsed = matter(raw); // or existing parser
      if (!parsed.data || Object.keys(parsed.data).length === 0) {
        console.warn("phase130-skill-manifest-missing", JSON.stringify({ skill: path.basename(skillDir), path: skillMdPath }));
        return { manifest: null, status: "manifest-missing" };
      }

      const result = SkillManifestSchema.safeParse(parsed.data);
      if (!result.success) {
        console.error("phase130-skill-manifest-parse-error", JSON.stringify({
          skill: path.basename(skillDir),
          issues: result.error.issues.map(i => ({ path: i.path.join("."), message: i.message })),
        }));
        return { manifest: null, status: "parse-error", reason: "schema mismatch" };
      }

      const manifest = result.data;
      const enabledSet = new Set(enabledMcpServers);
      const missingMcp = manifest.requiredMcpServers.filter(s => !enabledSet.has(s));
      if (missingMcp.length > 0) {
        console.error("phase130-skill-load-fail", JSON.stringify({
          skill: manifest.name,
          missingMcp,
          enabledMcp: enabledMcpServers,
        }));
        return { manifest: null, status: "refused-mcp-missing", missingMcp };
      }

      console.info("phase130-skill-load-success", JSON.stringify({
        skill: manifest.name,
        capabilities: manifest.capabilities,
        requiredMcpServers: manifest.requiredMcpServers,
      }));
      return { manifest, status: "loaded" };
    }
    ```
