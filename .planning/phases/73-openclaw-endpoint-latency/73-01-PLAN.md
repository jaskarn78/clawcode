```typescript
/**
 * Phase 73 Plan 01 â€” depth-1 serial turn queue + pushable async iterable.
 *
 * Two primitives consumed by persistent-session-handle.ts:
 *
 *   SerialTurnQueue: Discord's TurnDispatcher assumes one in-flight turn
 *     per agent; this enforces that invariant at the SessionHandle boundary.
 *     Depth 1 (one in-flight + one queued). A third concurrent send rejects
 *     with QUEUE_FULL â€” caller (OpenAi server) maps to 429 Retry-After.
 *
 *   AsyncPushQueue<T>: pushable AsyncIterable<T>. The SDK's streaming input
 *     mode wants an AsyncIterable<SDKUserMessage>; we build one here so per-
 *     turn sendAndStream() can push a user message into the shared stream.
 */

export const QUEUE_FULL_ERROR_MESSAGE = "QUEUE_FULL";

export class AsyncPushQueue<T> implements AsyncIterable<T> {
  private queue: T[] = [];
  private waiter: ((v: IteratorResult<T>) => void) | null = null;
  private done = false;

  push(item: T): void {
    if (this.done) return; // reject pushes after end()
    if (this.waiter) {
      const w = this.waiter;
      this.waiter = null;
      w({ value: item, done: false });
      return;
    }
    this.queue.push(item);
  }

  end(): void {
    this.done = true;
    if (this.waiter) {
      this.waiter({ value: undefined as unknown as T, done: true });
      this.waiter = null;
    }
  }

  [Symbol.asyncIterator](): AsyncIterator<T> {
    return {
      next: () => new Promise<IteratorResult<T>>((resolve) => {
        const item = this.queue.shift();
        if (item !== undefined) {
          resolve({ value: item, done: false });
          return;
        }
        if (this.done) {
          resolve({ value: undefined as unknown as T, done: true });
          return;
        }
        this.waiter = resolve;
      }),
    };
  }
}

export class SerialTurnQueue {
  private inFlight: Promise<unknown> | null = null;
  private queued: Promise<unknown> | null = null;

  async run<T>(fn: () => Promise<T>): Promise<T> {
    if (this.inFlight && this.queued) {
      throw new Error(QUEUE_FULL_ERROR_MESSAGE);
    }
    if (this.inFlight) {
      // There IS an inFlight but no queued yet â€” this becomes the queued turn.
      // Swallow the inFlight rejection on the WAIT path only; the inFlight's
      // own caller already sees its rejection directly.
      const waitFor = this.inFlight;
      this.queued = waitFor.then(
        () => {},
        () => {},
      );
      await this.queued;
    }
    const p = fn();
    this.inFlight = p.finally(() => {
      this.inFlight = null;
      this.queued = null;
    });
    return p;
  }
}
```

```typescript
    generatorDead = true;

    //
    // Apply usageCallback on msg.type === "result" (same shape as extractUsage).
    // Apply prefixHashProvider.get()/.persist() same as session-adapter.ts.
    // Apply skillTracking.skillUsageTracker.record mentions same shape.

  const handle: SessionHandle = {
    get sessionId() { return sessionId; },

        inputQueue.push({
          type: "user",
          message: { role: "user", content: promptWithMutable(message) },
          parent_tool_use_id: null,

          type: "user",
          message: { role: "user", content: promptWithMutable(message) },

          type: "user",
          message: { role: "user", content: promptWithMutable(message) },
          parent_tool_use_id: null,
```
