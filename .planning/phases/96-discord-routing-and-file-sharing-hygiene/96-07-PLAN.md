fs-probe.ts heartbeat check (Phase 85 mcp-reconnect.ts mirror):
```ts
import { runFsProbe } from "../../manager/fs-probe.js";
import { writeFsSnapshot } from "../../manager/fs-snapshot-store.js";
import { resolveFileAccess } from "../../config/loader.js";
// ...

export const fsProbeCheck = {
  name: "fs-probe",
  intervalMs: 60_000,                        // D-01: 60s, same as MCP probe
  async tick(deps) {
    const startMs = Date.now();
    for (const agent of deps.listAgents()) {
      const handle = deps.getHandle(agent);
      if (!handle) continue;
      const cfg = deps.getResolvedConfig(agent);
      const paths = resolveFileAccess(agent, cfg, deps.getDefaults());
      const prev = handle.getFsCapabilitySnapshot();
      try {
        const outcome = await runFsProbe(paths, deps.fsProbeDeps, prev);
        if (outcome.kind === "completed") {
          await writeFsSnapshot(agent, outcome.snapshot, deps.getFsCapabilityPath(agent), deps.fsSnapshotStoreDeps);
          handle.setFsCapabilitySnapshot(outcome.snapshot);
        }
      } catch (err) {
        deps.log.warn({ err, agent }, "fs-probe tick failed");
      }
    }
    deps.log.debug({ elapsedMs: Date.now() - startMs }, "fs-probe tick complete");
  },
};
```

    ```ts
    /**
     * Phase 96 Plan 07 — D-01 heartbeat layer of filesystem capability probe.
     * Mirrors src/heartbeat/checks/mcp-reconnect.ts (Phase 85 plan 01).
     * Tick interval 60s; iterates running agents; per-agent runFsProbe → writeFsSnapshot → setFsCapabilitySnapshot.
     */
    import type { Logger } from "pino";
    import { runFsProbe, type FsProbeDeps } from "../../manager/fs-probe.js";
    import { writeFsSnapshot, type FsSnapshotStoreDeps } from "../../manager/fs-snapshot-store.js";
    import { resolveFileAccess } from "../../config/loader.js";
    import type { SessionHandle } from "../../manager/persistent-session-handle.js";

    export interface FsProbeCheckDeps {
      readonly listAgents: () => readonly string[];
      readonly getHandle: (agent: string) => SessionHandle | undefined;
      readonly getResolvedConfig: (agent: string) => { fileAccess?: readonly string[] } | undefined;
      readonly getDefaults: () => { fileAccess?: readonly string[] };
      readonly getFsCapabilityPath: (agent: string) => string;
      readonly fsProbeDeps: FsProbeDeps;
      readonly fsSnapshotStoreDeps: FsSnapshotStoreDeps;
      readonly log: Logger;
    }

    export interface CheckModule {
      readonly name: string;
      readonly intervalMs: number;
      readonly tick: (deps: FsProbeCheckDeps) => Promise<void>;
    }

    export const fsProbeCheck: CheckModule = {
      name: "fs-probe",
      intervalMs: 60_000,  // D-01: 60s, same cadence as Phase 85 mcp-reconnect
      async tick(deps) {
        const startMs = Date.now();
        const agents = deps.listAgents();
        // Per-agent independence: failures don't block other agents
        await Promise.all(agents.map(async (agent) => {
          const handle = deps.getHandle(agent);
          if (!handle) return;  // agent not running; skip silently

          const cfg = deps.getResolvedConfig(agent);
          if (!cfg) return;
          const paths = resolveFileAccess(agent, cfg, deps.getDefaults());

          const prev = handle.getFsCapabilitySnapshot();

          try {
            const outcome = await runFsProbe(paths, deps.fsProbeDeps, prev);
            if (outcome.kind === "completed") {
              await writeFsSnapshot(
                agent,
                outcome.snapshot,
                deps.getFsCapabilityPath(agent),
                deps.fsSnapshotStoreDeps,
              );
              handle.setFsCapabilitySnapshot(outcome.snapshot);
            }
            // outcome.kind === "failed" → no persist, no in-memory update; debug-log
          } catch (err) {
            deps.log.warn({ err, agent }, "fs-probe tick failed for agent");
          }
        }));
        deps.log.debug({ elapsedMs: Date.now() - startMs, agentCount: agents.length }, "fs-probe tick complete");
      },
    };
    ```
