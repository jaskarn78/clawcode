```ts
// src/prompt/filesystem-capability-block.ts (NEW)

import type { FsCapabilitySnapshot } from "../manager/persistent-session-handle.js";

export interface FlapHistoryEntry {
  readonly windowStart: string;     // ISO8601 — start of current 5min flap window
  readonly transitions: number;     // count of ready↔non-ready transitions
  readonly stickyDegraded: boolean; // true if 3+ flaps in window
}

export const FS_FLAP_WINDOW_MS = 5 * 60 * 1000;        // 5 minutes
export const FS_FLAP_TRANSITION_THRESHOLD = 3;

export interface RenderFsBlockOptions {
  /** DI'd flap-stability tracker; caller owns mutability across calls. */
  readonly flapHistory?: Map<string, FlapHistoryEntry>;
  readonly now?: () => Date;
}

/**
 * Pure: produces XML-tagged stable-prefix block. Empty snapshot → empty string
 * (cache-stability for v2.5 fixtures without fileAccess declaration).
 *
 * Three subsections by classification:
 *   1. My workspace (full RW) — entries with canonicalPath.startsWith(agentWorkspaceRoot + sep)
 *      OR canonicalPath === agentWorkspaceRoot
 *   2. Operator-shared paths (per ACL) — entries NOT under agentWorkspaceRoot with mode='ro'
 *   3. Off-limits — single static line "Anything outside the above."
 *
 * Only status='ready' entries appear in subsections 1+2 (degraded HIDDEN from LLM).
 * Within each subsection, entries sorted by canonicalPath ASCII-ascending (deterministic).
 *
 * Flap-stability: if a path has 3+ transitions in the 5-min window, treat as
 * degraded for the rest of the window (excluded from output) — prevents
 * prompt-cache prefix-hash yo-yo. Mirrors Phase 94 plan 02 filterToolsByCapabilityProbe.
 */
export function renderFilesystemCapabilityBlock(
  snapshot: ReadonlyMap<string, FsCapabilitySnapshot>,
  agentWorkspaceRoot: string,
  options?: RenderFsBlockOptions,
): string;

/** Pure helper: returns true if entry should be included in rendered block. */
export function isFsEntryAdvertisable(
  canonicalPath: string,
  state: FsCapabilitySnapshot,
  flapHistory: Map<string, FlapHistoryEntry> | undefined,
  now: Date,
): boolean;
```

    ```ts
    /**
     * Phase 96 Plan 02 — system-prompt <filesystem_capability> block renderer.
     * Pure module. No SDK imports, no node:fs, no logger, no Date constructor.
     * Produces a 3-section XML-tagged block; only status='ready' entries appear.
     * Empty snapshot → empty string (cache-stability for v2.5 fixtures).
     */
    import { sep } from "node:path";
    import type { FsCapabilitySnapshot } from "../manager/persistent-session-handle.js";

    export const FS_FLAP_WINDOW_MS = 5 * 60 * 1000;
    export const FS_FLAP_TRANSITION_THRESHOLD = 3;

    export interface FlapHistoryEntry {
      readonly windowStart: string;
      readonly transitions: number;
      readonly stickyDegraded: boolean;
    }

    export interface RenderFsBlockOptions {
      readonly flapHistory?: Map<string, FlapHistoryEntry>;
      readonly now?: () => Date;
    }

    export function isFsEntryAdvertisable(
      canonicalPath: string,
      state: FsCapabilitySnapshot,
      flapHistory: Map<string, FlapHistoryEntry> | undefined,
      now: Date,
    ): boolean {
      // Hide degraded + unknown
      if (state.status !== "ready") return false;

      // Sticky-degraded flap-stability — Phase 94 plan 02 mirror
      const flap = flapHistory?.get(canonicalPath);
      if (flap?.stickyDegraded) {
        const windowStartMs = new Date(flap.windowStart).getTime();
        if (now.getTime() - windowStartMs < FS_FLAP_WINDOW_MS) {
          return false;
        }
      }

      return true;
    }

    function isUnderRoot(canonicalPath: string, agentWorkspaceRoot: string): boolean {
      // Within-workspace anchoring — uses startsWith intra-workspace (NOT cross-workspace per D-06).
      // D-06 forbids startsWith for cross-workspace boundary; intra-workspace under agent root is OK
      // because there's no symlink risk inside the agent's own workspace.
      if (canonicalPath === agentWorkspaceRoot) return true;
      return canonicalPath.startsWith(agentWorkspaceRoot.endsWith(sep) ? agentWorkspaceRoot : agentWorkspaceRoot + sep);
    }

    export function renderFilesystemCapabilityBlock(
      snapshot: ReadonlyMap<string, FsCapabilitySnapshot>,
      agentWorkspaceRoot: string,
      options?: RenderFsBlockOptions,
    ): string {
      // Empty-snapshot short-circuit — preserves v2.5 stable-prefix hash
      if (snapshot.size === 0) return "";

      const now = (options?.now ?? (() => new Date()))();
      const myWorkspace: string[] = [];
      const operatorShared: string[] = [];

      // Sort entries deterministically by canonicalPath ASCII-ascending
      const sortedEntries = Array.from(snapshot.entries()).sort(([a], [b]) => (a < b ? -1 : a > b ? 1 : 0));

      let anyAdvertisable = false;
      for (const [canonicalPath, state] of sortedEntries) {
        if (!isFsEntryAdvertisable(canonicalPath, state, options?.flapHistory, now)) continue;
        anyAdvertisable = true;
        if (isUnderRoot(canonicalPath, agentWorkspaceRoot)) {
          myWorkspace.push(`- ${canonicalPath}`);
        } else if (state.mode === "ro") {
          operatorShared.push(`- ${canonicalPath} (RO, ACL)`);
        }
        // mode='denied' is unreachable here because status='ready' implies fs.access succeeded,
        // which means mode is 'rw' or 'ro' — never 'denied'.
      }

      // If snapshot is non-empty but ALL entries are degraded/unknown/sticky → return empty string
      // (no informational block to render — caller's stable-prefix is unchanged).
      if (!anyAdvertisable) return "";

      const lines: string[] = [];
      lines.push("<filesystem_capability>");
      lines.push("## My workspace (full RW)");
      if (myWorkspace.length > 0) {
        lines.push(...myWorkspace);
      } else {
        lines.push("- (none)");
      }
      lines.push("");
      lines.push("## Operator-shared paths (per ACL)");
      if (operatorShared.length > 0) {
        lines.push(...operatorShared);
      } else {
        lines.push("- (none)");
      }
      lines.push("");
      lines.push("## Off-limits — do not attempt");
      lines.push("- Anything outside the above.");
      lines.push("</filesystem_capability>");

      return lines.join("\n");
    }
    ```
