```ts
/** Phase 96 D-CONTEXT — 3-value status enum (vs Phase 94's 5-value enum because
 *  filesystem capability has no reconnect/failed analog — operator-driven ACL
 *  changes don't transition through transient connect states). */
export type FsCapabilityStatus =
  | "ready"          // fs.access(R_OK) succeeded — path readable now
  | "degraded"       // declared in fileAccess but fs.access failed (verbatim error captured)
  | "unknown";       // never probed (boot pre-warm-path)

export type FsCapabilityMode = "rw" | "ro" | "denied";

export interface FsCapabilitySnapshot {
  readonly status: FsCapabilityStatus;
  readonly mode: FsCapabilityMode;
  readonly lastProbeAt: string;       // ISO8601
  readonly lastSuccessAt?: string;    // ISO8601 of most recent ready
  readonly error?: string;            // verbatim from fs.access — Phase 85 TOOL-04 inheritance
}
```

```ts
// src/manager/fs-probe.ts (NEW)

import type { Logger } from "pino";
import type { FsCapabilitySnapshot } from "./persistent-session-handle.js";

export type FsProbeOutcome =
  | { readonly kind: "completed"; readonly snapshot: ReadonlyMap<string, FsCapabilitySnapshot>; readonly durationMs: number }
  | { readonly kind: "failed"; readonly error: string };

export const FS_PROBE_TIMEOUT_MS = 5_000;        // D-CONTEXT: 5s per path

export interface FsProbeDeps {
  /** Wraps node:fs/promises.access. */
  readonly fsAccess: (path: string, mode: number) => Promise<void>;
  readonly fsConstants: { readonly R_OK: number; readonly W_OK: number };
  /** Wraps node:fs/promises.realpath — canonical-path resolution (D-06). */
  readonly realpath: (path: string) => Promise<string>;
  readonly now?: () => Date;
  readonly log: Logger;
}

/** Pure orchestrator. Probes ALL paths in parallel via Promise.all + per-path timeout.
 *  Failures don't block siblings. Verbatim error pass-through. */
export async function runFsProbe(
  paths: readonly string[],
  deps: FsProbeDeps,
  prevSnapshot?: ReadonlyMap<string, FsCapabilitySnapshot>,
): Promise<FsProbeOutcome>;
```

    ```ts
     * Phase 96 Plan 01 — filesystem capability probe primitive.
     * Pure DI module. No SDK imports, no node:fs, no Date constructor (use deps.now).
     * 3-value status enum (ready|degraded|unknown) — diverges from Phase 94's 5-value
     * because filesystem capability has no reconnect/failed analog.
     */

    import type {
      FsCapabilityMode,
      FsCapabilitySnapshot,
      FsCapabilityStatus,
    } from "./persistent-session-handle.js";

    export const FS_PROBE_TIMEOUT_MS = 5_000;

    export type FsProbeOutcome =
      | { readonly kind: "completed"; readonly snapshot: ReadonlyMap<string, FsCapabilitySnapshot>; readonly durationMs: number }
      | { readonly kind: "failed"; readonly error: string };

    export interface FsProbeDeps {
      readonly fsAccess: (path: string, mode: number) => Promise<void>;
      readonly fsConstants: { readonly R_OK: number; readonly W_OK: number };
      readonly realpath: (path: string) => Promise<string>;
      readonly resolve?: (path: string) => string;  // node:path.resolve fallback
      readonly now?: () => Date;
      readonly log: Logger;
    }

    function withTimeout<T>(p: Promise<T>, ms: number, label: string): Promise<T> {
      return Promise.race([
        p,
        new Promise<T>((_, reject) =>
          setTimeout(() => reject(new Error(`fs probe ${label} timeout after ${ms}ms`)), ms),
        ),
      ]);
    }

    async function canonicalize(rawPath: string, deps: FsProbeDeps): Promise<string> {
      try {
        return await deps.realpath(rawPath);
      } catch {
        // ENOENT / fall back to resolve-only
        return deps.resolve ? deps.resolve(rawPath) : rawPath;
      }
    }

    async function probeOne(
      rawPath: string,
      deps: FsProbeDeps,
      now: Date,
      prevEntry: FsCapabilitySnapshot | undefined,
    ): Promise<readonly [string, FsCapabilitySnapshot]> {
      const canonical = await canonicalize(rawPath, deps);
      try {
        await withTimeout(deps.fsAccess(canonical, deps.fsConstants.R_OK), FS_PROBE_TIMEOUT_MS, canonical);
        const status: FsCapabilityStatus = "ready";
        const mode: FsCapabilityMode = "ro";  // promotion to RW requires explicit fileAccess + W_OK probe (deferred)
        return [
          canonical,
          {
            status,
            mode,
            lastProbeAt: now.toISOString(),
            lastSuccessAt: now.toISOString(),
          },
        ] as const;
      } catch (err) {
        const errMsg = err instanceof Error ? err.message : String(err);
        const status: FsCapabilityStatus = "degraded";
        const mode: FsCapabilityMode = "denied";
        return [
          canonical,
          {
            status,
            mode,
            lastProbeAt: now.toISOString(),
            error: errMsg,
            ...(prevEntry?.lastSuccessAt !== undefined ? { lastSuccessAt: prevEntry.lastSuccessAt } : {}),
          },
        ] as const;
      }
    }

    export async function runFsProbe(
      paths: readonly string[],
      deps: FsProbeDeps,
      prevSnapshot?: ReadonlyMap<string, FsCapabilitySnapshot>,
    ): Promise<FsProbeOutcome> {
      const now = (deps.now ?? (() => new Date()))();
      const startMs = now.getTime();

      try {
        const settled = await Promise.all(
          paths.map(async (rawPath) => {
            try {
              return await probeOne(rawPath, deps, now, prevSnapshot?.get(rawPath));
            } catch (err) {
              // Defensive — probeOne already swallows; this catches programmer errors only
              const errMsg = err instanceof Error ? err.message : String(err);
              const fallbackKey = deps.resolve ? deps.resolve(rawPath) : rawPath;
              return [

                {
                  status: "degraded" as const,
                  mode: "denied" as const,
                  lastProbeAt: now.toISOString(),
                  error: errMsg,
                },
              ] as const;
            }
          }),
        );
        const snapshot = new Map(settled);
        const durationMs = ((deps.now ?? (() => new Date()))()).getTime() - startMs;
        return { kind: "completed", snapshot, durationMs };
      } catch (err) {
        return { kind: "failed", error: err instanceof Error ? err.message : String(err) };
      }
    }
    ```
