export const preDeploySnapshotSchema: z.ZodObject<{
  version: z.ZodLiteral<1>;

  snapshotPid: z.ZodNumber;         // positive int

    name: z.ZodString;              // min(1)
    sessionId: z.ZodNullable<z.ZodString>;
  }>>;
}>;

export type PreDeploySnapshot = z.infer<typeof preDeploySnapshotSchema>;

export const DEFAULT_PRE_DEPLOY_SNAPSHOT_PATH: string;

export function writePreDeploySnapshot(
  filePath: string,
  runningAgents: ReadonlyArray<{ name: string; sessionId: string | null }>,
  log: Logger,
): Promise<void>;
// Writes atomically (tmp + rename). mkdir -p the parent. Logs at info with
// component="snapshot-restore". Throws on unrecoverable fs errors (caller
// in daemon.ts wraps in try/catch and logs warn).

export function readAndConsumePreDeploySnapshot(
  filePath: string,
  knownAgentNames: ReadonlySet<string>,
  maxAgeHours: number,

// Returns empty set on: ENOENT, malformed JSON, schema fail, stale (>maxAgeHours).
// Always deletes the file on read attempt (except ENOENT). Never throws to caller.
