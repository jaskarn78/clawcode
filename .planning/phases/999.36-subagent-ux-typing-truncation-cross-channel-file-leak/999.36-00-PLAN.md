private fireTypingIndicator(message: Message, turn: Turn | undefined): void {
  let span: Span | undefined;
  try {
    span = turn?.startSpan("typing_indicator", {});
    if ("sendTyping" in message.channel && typeof message.channel.sendTyping === "function") {
      void message.channel.sendTyping().catch((err) => {
        this.log.debug(
          { error: (err as Error).message, channelId: message.channelId },
          "sendTyping failed — observational, non-fatal",
        );
      });
    }
  } catch (err) {
    this.log.debug(
      { error: (err as Error).message, channelId: message.channelId },
      "typing indicator setup failed — observational, non-fatal",
    );
  } finally {
    try { span?.end(); } catch { /* non-fatal */ }
  }
}

<threat_model>
- **Typing indicator failure mode**: `setInterval` not cleared → leak. MUST clear in finally OR via explicit stopTypingLoop handle returned to caller.
- **Bot-token rate limit**: Discord limits sendTyping calls per second. The 8s cadence keeps us well under any limit.
- **Subagent boot >5min**: Typing indicator will refresh ~37 times. No issue. Discord's typing extension is 10s per call; 8s cadence guarantees no flicker.
- **Thread archival mid-typing**: If thread is archived between two `sendTyping` calls, the next call rejects. Catch-and-swallow per bridge.ts pattern — observational only.
- **Diagnostic log volume**: With 14 active agents and ~5 subagent spawns/day, the diagnostic logs add ~50 lines/day. Negligible. Do NOT downgrade to debug; operators read INFO via journalctl.
- **Compliance**: Logs do not include user content — only chunk byte indices + lengths + source-tag strings. No PII risk.
