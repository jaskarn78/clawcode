// dream-pass.ts EXPORTS

export const dreamResultSchema = z.object({
  newWikilinks: z.array(z.object({
    from: z.string(),
    to: z.string(),
    rationale: z.string(),
  })),
  promotionCandidates: z.array(z.object({
    chunkId: z.string(),
    currentPath: z.string(),
    rationale: z.string(),
    priorityScore: z.number().min(0).max(100),
  })),
  themedReflection: z.string(),
  suggestedConsolidations: z.array(z.object({
    sources: z.array(z.string()),
    newPath: z.string(),
    rationale: z.string(),
  })),
});
export type DreamResult = z.infer<typeof dreamResultSchema>;

export const dreamPassOutcomeSchema = z.discriminatedUnion("kind", [
  z.object({ kind: z.literal("completed"), result: dreamResultSchema, durationMs: z.number(), tokensIn: z.number(), tokensOut: z.number(), model: z.string() }),
  z.object({ kind: z.literal("skipped"), reason: z.enum(["agent-active", "disabled"]) }),
  z.object({ kind: z.literal("failed"), error: z.string() }),
]);
export type DreamPassOutcome = z.infer<typeof dreamPassOutcomeSchema>;

export interface RunDreamPassDeps {
  readonly memoryStore: { getRecentChunks(agent: string, limit: number): Promise<MemoryChunk[]> };
  readonly conversationStore: { getRecentSummaries(agent: string, limit: number): Promise<ConversationSummary[]> };
  readonly readFile: (path: string) => Promise<string>;
  readonly dispatch: (req: DreamDispatchRequest) => Promise<DreamDispatchResponse>;
  readonly resolvedDreamConfig: { enabled: boolean; idleMinutes: number; model: string; retentionDays?: number };
  readonly memoryRoot: string; // ~clawcode/.clawcode/agents/<agent>/memory
  readonly now?: (deps: unknown) => Date; // optional for test injection
  readonly log: { info: (m: string) => void; warn: (m: string) => void; error: (m: string) => void };
}

export async function runDreamPass(agentName: string, deps: RunDreamPassDeps): Promise<DreamPassOutcome>;

// dream-prompt-builder.ts EXPORTS

export interface DreamPromptInput {
  readonly recentChunks: MemoryChunk[];
  readonly memoryMd: string;
  readonly recentSummaries: ConversationSummary[];
  readonly graphEdges: string;
  readonly agentName: string;
}
export const DREAM_PROMPT_INPUT_TOKEN_BUDGET = 32_000;
export function buildDreamPrompt(input: DreamPromptInput): { systemPrompt: string; userPrompt: string; estimatedInputTokens: number };

// idle-window-detector.ts EXPORTS

export const IDLE_HARD_FLOOR_MS = 5 * 60 * 1000;
export const IDLE_HARD_CEILING_MS = 6 * 60 * 60 * 1000;

  readonly lastTurnAt: Date | null;
  readonly idleMinutes: number;
  readonly now: () => Date;
}
export function isAgentIdle(deps: IsAgentIdleDeps): { idle: boolean; reason: "active" | "idle-threshold-met" | "idle-ceiling-bypass" | "no-prior-turn" };
export function findIdleAgents(agents: Array<{ name: string; lastTurnAt: Date | null; dreamConfig: { enabled: boolean; idleMinutes: number } }>, now: () => Date): string[];

  retentionDays: z.number().int().min(1).max(365).optional(),
});
export type DreamConfig = z.infer<typeof dreamConfigSchema>;
