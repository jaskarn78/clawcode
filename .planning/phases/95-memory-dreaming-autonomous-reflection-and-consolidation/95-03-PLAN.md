       ```
       export async function handleRunDreamPassIpc(
         req: RunDreamPassRequest,
         deps: {
           runDreamPass: (agent: string, model?: string) => Promise<DreamPassOutcome>;
           applyDreamResult: (agent: string, outcome: DreamPassOutcome) => Promise<DreamApplyOutcome>;
           isAgentIdle: (agent: string) => { idle: boolean; reason: string };
           getResolvedDreamConfig: (agent: string) => { enabled: boolean; idleMinutes: number; model: string } | null;
           knownAgents: () => string[];
           now: () => Date;
           log: { info: (m: string) => void; warn: (m: string) => void; error: (m: string) => void };
         },
       ): Promise<RunDreamPassResponse> {
         if (!deps.knownAgents().includes(req.agent)) {
           throw new ManagerError(`agent not found: ${req.agent}`, { code: -32602 });
         }
         const startedAt = deps.now().toISOString();
         const baseConfig = deps.getResolvedDreamConfig(req.agent);
         if (!baseConfig) {
           // No dream block at all → operator must --force
           if (!req.force) {
             return { outcome: { kind: 'skipped', reason: 'disabled' }, applied: { kind: 'skipped', reason: 'no-completed-result' }, agent: req.agent, startedAt };
           }
         } else if (!baseConfig.enabled && !req.force) {
           return { outcome: { kind: 'skipped', reason: 'disabled' }, applied: { kind: 'skipped', reason: 'no-completed-result' }, agent: req.agent, startedAt };
         }
         if (!req.idleBypass) {
           const idle = deps.isAgentIdle(req.agent);
           if (!idle.idle) {
             return { outcome: { kind: 'skipped', reason: 'agent-active' }, applied: { kind: 'skipped', reason: 'no-completed-result' }, agent: req.agent, startedAt };
           }
         }
         const model = req.modelOverride ?? baseConfig?.model ?? 'haiku';
         const outcome = await deps.runDreamPass(req.agent, model);
         const applied = await deps.applyDreamResult(req.agent, outcome);
         deps.log.info(`[run-dream-pass] ${req.agent}: outcome=${outcome.kind} applied=${applied.kind}`);
         return { outcome, applied, agent: req.agent, startedAt };
       }
       ```
