   ```
     systemPrompt += "\n\n## Available MCP Tools\n\n";
   ```

   ```typescript
   export type McpHealthResult = {
     readonly name: string;
     readonly healthy: boolean;
     readonly latencyMs: number;
     readonly error?: string;
   };

   export async function checkMcpServerHealth(
     server: { name: string; command: string; args: readonly string[]; env: Readonly<Record<string, string>> },
     timeoutMs?: number,
   ): Promise<McpHealthResult>;
   ```
   Implementation:
   - Use `execa` to spawn the server process with `{ stdin: "pipe", stdout: "pipe", stderr: "pipe", env: { ...process.env, ...server.env } }`
   - Send JSON-RPC initialize request: `{"jsonrpc":"2.0","id":"health","method":"initialize","params":{"protocolVersion":"2024-11-05","capabilities":{},"clientInfo":{"name":"clawcode-health","version":"0.1.0"}}}`
   - Wait for a JSON-RPC response on stdout (parse line by line)
   - If valid response received within timeout (default 5000ms), return healthy: true with latencyMs
   - If process exits or timeout, return healthy: false with error message
   - Always kill the spawned process after check completes
