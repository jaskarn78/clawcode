       ```typescript
       export const BROWSER_ERROR_TYPES = [
         "timeout",
         "element_not_found",
         "navigation_failed",
         "launch_failed",
         "invalid_argument",
         "internal",
       ] as const;
       export type BrowserErrorType = (typeof BROWSER_ERROR_TYPES)[number];

       export class BrowserError extends Error {
         readonly type: BrowserErrorType;
         readonly selector?: string;
         readonly timeoutMs?: number;
         constructor(
           type: BrowserErrorType,
           message: string,
           opts: { selector?: string; timeoutMs?: number; cause?: unknown } = {},
         ) {
           super(message, opts.cause ? { cause: opts.cause } : undefined);
           this.name = "BrowserError";
           this.type = type;
           this.selector = opts.selector;
           this.timeoutMs = opts.timeoutMs;
         }
       }

       /**
        * Normalize any thrown value into a structured BrowserToolError envelope.
        * Playwright's TimeoutError carries `name === "TimeoutError"`.
        */
       export function toBrowserToolError(err: unknown): import("./types.js").BrowserToolError {
         if (err instanceof BrowserError) {
           return Object.freeze({
             ok: false as const,
             error: Object.freeze({
               type: err.type,
               message: err.message,
               ...(err.selector !== undefined ? { selector: err.selector } : {}),
               ...(err.timeoutMs !== undefined ? { timeoutMs: err.timeoutMs } : {}),
             }),
           });
         }
         if (err instanceof Error && err.name === "TimeoutError") {
           return Object.freeze({
             ok: false as const,
             error: Object.freeze({ type: "timeout" as const, message: err.message }),
           });
         }
         const msg = err instanceof Error ? err.message : String(err);
         return Object.freeze({
           ok: false as const,
           error: Object.freeze({ type: "internal" as const, message: msg }),
         });
       }
       ```

       ```typescript
        * Return the given path if a non-zero-byte state.json exists, else undefined.

           if (s.size === 0) return undefined;
           return statePath;
         } catch {
           return undefined;
         }
       }

       /**
        * Save context storageState atomically: write to <statePath>.tmp, then
        * rename. Rename is atomic on the same filesystem. Pitfall 10 guard:
        * clean up stale .tmp on entry (best-effort — ignore errors).
        */
       export async function saveState(
         ctx: BrowserContext,
         statePath: string,
       ): Promise<void> {
         const tmp = `${statePath}.tmp`;
         try {
           await mkdir(dirname(statePath), { recursive: true });
           await unlink(tmp).catch(() => { /* ignore */ });
           await ctx.storageState({ path: tmp, indexedDB: true });
           await rename(tmp, statePath);
         } catch (err) {
           throw new BrowserError("internal", `storageState save failed: ${(err as Error).message}`, { cause: err });
         }
       }

       /**
        * Debounced best-effort saver. Multiple rapid trigger() calls collapse
        * to a single saveFn() after waitMs of quiet. flush() awaits any
        * pending or in-flight save.
        *
        * Errors from saveFn are logged (if logger provided) but not thrown —
        * storageState loss is not fatal (agent recovers next warm) so we do
        * NOT surface this to tool callers as a failure.
        */
       export function makeDebouncedSaver(
         saveFn: () => Promise<void>,
         waitMs: number,
         log?: BrowserLogger,
       ): { trigger: () => void; flush: () => Promise<void> } {
         let timer: NodeJS.Timeout | null = null;
         let inflight: Promise<void> | null = null;

         const run = async (): Promise<void> => {
           timer = null;
           try {
             inflight = saveFn();
             await inflight;
           } catch (err) {
             log?.warn({ err: (err as Error).message }, "debounced storageState save failed");
           } finally {
             inflight = null;
           }
         };

         return {
           trigger(): void {
             if (timer) clearTimeout(timer);
             timer = setTimeout(() => { void run(); }, waitMs);
           },
           async flush(): Promise<void> {
             if (timer) {
               clearTimeout(timer);
               timer = null;
               await run();
               return;
             }
             if (inflight) await inflight;
           },
         };
       }
       ```

       ```typescript
         private browser: BrowserLike | null = null;
         private warmPromise: Promise<void> | null = null;
         private readonly contexts = new Map<string, { ctx: BrowserContext; workspace: string }>();
         private readonly savers = new Map<string, { trigger(): void; flush(): Promise<void> }>();

             throw new BrowserError("launch_failed", "browser not warmed — call warm() first");

           if (cached) return cached.ctx;

           );
           ctx.on("close", () => {
             this.contexts.delete(agent);
             this.savers.delete(agent);
           });

         }

           const entry = this.contexts.get(agent);
           if (!entry) return;
           const saver = this.savers.get(agent);
           if (saver) await saver.flush();
           const statePath = join(entry.workspace, "browser", "state.json");
           await saveState(entry.ctx, statePath);
         }

         async close(): Promise<void> {
           if (this.closed) return;
           this.closed = true;
           // Pitfall 10: save BEFORE close.
           for (const [agent] of this.contexts) {
             try {
               await this.saveAgentStateNow(agent);
             } catch (err) {
               this.log?.warn({ agent, err: (err as Error).message }, "per-agent storageState save failed on shutdown");
             }
           }
           for (const [agent, entry] of this.contexts) {
             try {
               await entry.ctx.close();
             } catch (err) {
               this.log?.warn({ agent, err: (err as Error).message }, "per-agent context close failed");
             }
           }
           this.contexts.clear();
           this.savers.clear();
           if (this.browser) {
             try {
               await this.browser.close();
             } catch (err) {
               this.log?.warn({ err: (err as Error).message }, "browser close failed");
             }
             this.browser = null;
           }
           this.warmPromise = null;
           this.log?.info("browser manager closed");
         }
       }
       ```
