       ```typescript
       import type { Database } from "better-sqlite3";

       export class ApiKeySessionIndex {
         constructor(private readonly db: Database) {}
         lookup(keyHash: string): { session_id: string; agent_name: string } | null {
           const row = this.db.prepare("SELECT session_id, agent_name FROM api_key_sessions WHERE key_hash = ?").get(keyHash) as { session_id: string; agent_name: string } | undefined;
           return row ?? null;
         }
         record(keyHash: string, agentName: string, sessionId: string): void {
           const now = Date.now();
           this.db.prepare(`
             INSERT INTO api_key_sessions (key_hash, agent_name, session_id, created_at, last_used_at)
             VALUES (?, ?, ?, ?, ?)
             ON CONFLICT(key_hash) DO UPDATE SET session_id = excluded.session_id, last_used_at = excluded.last_used_at
           `).run(keyHash, agentName, sessionId, now, now);
         }
         touch(keyHash: string): void {
           this.db.prepare("UPDATE api_key_sessions SET last_used_at = ? WHERE key_hash = ?").run(Date.now(), keyHash);
         }
         delete(keyHash: string): boolean {
           const r = this.db.prepare("DELETE FROM api_key_sessions WHERE key_hash = ?").run(keyHash);
           return r.changes > 0;
         }
         listForAgent(agentName: string): ReadonlyArray<{ key_hash: string; session_id: string; last_used_at: number }> {
           return this.db.prepare("SELECT key_hash, session_id, last_used_at FROM api_key_sessions WHERE agent_name = ? ORDER BY last_used_at DESC").all(agentName) as ReadonlyArray<{ key_hash: string; session_id: string; last_used_at: number }>;
         }
       }
       // Convenience wrappers
       export function lookupSessionForKey(db: Database, keyHash: string) { return new ApiKeySessionIndex(db).lookup(keyHash); }
       export function recordSessionForKey(db: Database, keyHash: string, agent: string, sessionId: string): void { new ApiKeySessionIndex(db).record(keyHash, agent, sessionId); }
       ```
