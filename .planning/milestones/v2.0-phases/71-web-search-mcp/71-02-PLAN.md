    ```typescript
    import { webSearch, webFetchUrl } from "./tools.js";
    import type { SearchConfig } from "../config/schema.js";
    import type { ResolvedAgentConfig } from "../shared/types.js";
    import type { IpcSearchToolCallParams } from "../ipc/types.js";
    import type { SearchToolOutcome } from "./types.js";

    export interface SearchDaemonHandlerDeps {
      readonly searchConfig: SearchConfig;
      readonly resolvedAgents: readonly ResolvedAgentConfig[];
      readonly braveClient: { search(q: string, opts?: { numResults?: number }): Promise<SearchToolOutcome<unknown>> };
      readonly exaClient:   { search(q: string, opts?: { numResults?: number }): Promise<SearchToolOutcome<unknown>> };
      readonly fetcher: typeof import("./fetcher.js").fetchUrl;
    }

    export async function handleSearchToolCall(
      deps: SearchDaemonHandlerDeps,
      params: IpcSearchToolCallParams,
    ): Promise<SearchToolOutcome<unknown>> {
      // 1. Disabled guard → internal
      if (!deps.searchConfig.enabled) return Object.freeze({ ok: false, error: Object.freeze({ type: "internal", message: "search MCP disabled (defaults.search.enabled=false)" }) });
      // 2. Agent resolution → invalid_argument
      const resolvedAgent = deps.resolvedAgents.find(a => a.name === params.agent);
      if (!resolvedAgent) return Object.freeze({ ok: false, error: Object.freeze({ type: "invalid_argument", message: `unknown agent: ${params.agent}` }) });
      // 3. Dispatch
      const toolDeps = { config: deps.searchConfig, braveClient: deps.braveClient, exaClient: deps.exaClient, fetcher: (url, opts) => deps.fetcher(url, opts) };
      try {
        switch (params.toolName) {
          case "web_search":     return await webSearch(params.args as { query: string; numResults?: number }, toolDeps);
          case "web_fetch_url":  return await webFetchUrl(params.args as { url: string; mode?: "readability" | "raw"; maxBytes?: number }, toolDeps);
          default: {
            const unknown: string = params.toolName;
            return Object.freeze({ ok: false, error: Object.freeze({ type: "invalid_argument", message: `unknown search tool: ${unknown}` }) });
          }
        }
      } catch (err) {
        // Defensive — tool handlers should never throw, but protect the IPC boundary anyway.
        const message = err instanceof Error ? err.message : String(err);
        return Object.freeze({ ok: false, error: Object.freeze({ type: "internal", message }) });
      }
    }
    ```
    Note: NO `saveAgentState` equivalent here — search has no per-agent persistent state. No lazy-warm either — providers are ephemeral HTTP clients constructed at daemon startup (see Task 2).
