    Step 4 — Add `migrateConversationTurnsFts()` to `src/memory/store.ts` (private method, placed after `migrateInstructionFlags`) and wire the call into the constructor chain at line ~79 (AFTER `this.migrateInstructionFlags()`). Full SQL verbatim:

    ```typescript
    /**
     * Phase 68 — RETR-02. Create FTS5 full-text index over conversation_turns.content
     * in external-content mode, plus AI/AD/AU triggers for automatic synchronization,
     * plus one-shot backfill gated on sqlite_master lookup for idempotency across
     * daemon restarts. Tokenizer: unicode61 remove_diacritics 2 (SQLite default).
     */
    private migrateConversationTurnsFts(): void {
      const existing = this.db
        .prepare(
          "SELECT name FROM sqlite_master WHERE type='table' AND name='conversation_turns_fts'"
        )
        .get();
      const needsBackfill = !existing;

      this.db.exec(`
        CREATE VIRTUAL TABLE IF NOT EXISTS conversation_turns_fts USING fts5(
          content,
          content='conversation_turns',
          content_rowid='rowid',
          tokenize='unicode61 remove_diacritics 2'
        );

        CREATE TRIGGER IF NOT EXISTS conversation_turns_ai
        AFTER INSERT ON conversation_turns BEGIN
          INSERT INTO conversation_turns_fts(rowid, content)
            VALUES (new.rowid, new.content);
        END;

        CREATE TRIGGER IF NOT EXISTS conversation_turns_ad
        AFTER DELETE ON conversation_turns BEGIN
          INSERT INTO conversation_turns_fts(conversation_turns_fts, rowid, content)
            VALUES ('delete', old.rowid, old.content);
        END;

        CREATE TRIGGER IF NOT EXISTS conversation_turns_au
        AFTER UPDATE ON conversation_turns BEGIN
          INSERT INTO conversation_turns_fts(conversation_turns_fts, rowid, content)
            VALUES ('delete', old.rowid, old.content);
          INSERT INTO conversation_turns_fts(rowid, content)
            VALUES (new.rowid, new.content);
        END;
      `);

      if (needsBackfill) {
        this.db.exec(`
          INSERT INTO conversation_turns_fts(rowid, content)
            SELECT rowid, content FROM conversation_turns;
        `);
      }
    }
    ```

    /**
     * Scoped conversation search orchestrator.
     *
     * Merges semantic (sqlite-vec KNN via SemanticSearch) + full-text (FTS5 via
     * ConversationStore.searchTurns) candidates, applies time-decay weighting,
     * deduplicates scope="all" results (prefers session-summary over raw turns),
     * truncates snippets, and returns a single paginated page.
     *
     * Pure dependency-injected function. All I/O via `deps`. `now: Date` is
     * injectable for deterministic decay tests.
     *
     * Phase 68 — RETR-01 (wiring surface), RETR-02 (FTS5 path), RETR-03 (decay + pagination).
     */

    import { calculateRelevanceScore } from "./decay.js";
    import { SemanticSearch } from "./search.js";
    import type { MemoryEntry } from "./types.js";
    import type { ConversationTurnSearchResult } from "./conversation-types.js";
    import {
      SNIPPET_MAX_CHARS,
      MAX_RESULTS_PER_PAGE,
      DEFAULT_RETRIEVAL_HALF_LIFE_DAYS,
      type ScopedSearchDeps,
      type ScopedSearchOptions,
      type ScopedSearchPage,
      type ScopedSearchResult,
    } from "./conversation-search.types.js";

    const SEMANTIC_WEIGHT = 0.7;
    const DECAY_WEIGHT = 0.3;
    const CONVERSATION_TURN_DEFAULT_IMPORTANCE = 0.5;

    /** Truncate content to SNIPPET_MAX_CHARS, appending ellipsis if clipped. */
    function makeSnippet(content: string): string {
      if (content.length <= SNIPPET_MAX_CHARS) return content;
      return content.slice(0, SNIPPET_MAX_CHARS) + "…";
    }

    /** Convert FTS5 BM25 (negative; lower = better) to [0, 1] relevance score. */
    function bm25ToRelevance(bm25: number): number {
      return 1 / (1 + Math.abs(bm25));
    }

    export async function searchByScope(
      deps: ScopedSearchDeps,
      options: ScopedSearchOptions,
    ): Promise<ScopedSearchPage> {
      const clampedLimit = Math.min(
        Math.max(options.limit, 1),
        MAX_RESULTS_PER_PAGE,
      );
      const offset = Math.max(options.offset, 0);
      const now = options.now ?? new Date();
      const halfLifeDays = options.halfLifeDays ?? DEFAULT_RETRIEVAL_HALF_LIFE_DAYS;
      const decayParams = { halfLifeDays };

      const candidates: ScopedSearchResult[] = [];

      // 1. Semantic path (for scope='memories' and 'all'; and 'conversations' for session-summaries)
      if (options.scope === "memories" || options.scope === "all") {
        const queryEmbedding = await deps.embedder.embed(options.query);
        const semanticSearch = new SemanticSearch(deps.memoryStore);
        const semanticResults = semanticSearch.searchWithScore(
          queryEmbedding,
          MAX_RESULTS_PER_PAGE * 3,  // over-fetch for dedup room
        );

        for (const r of semanticResults) {
          const isSummary = r.memory.tags.includes("session-summary");
          // For scope='memories', EXCLUDE session-summaries (they belong to 'conversations' bucket)
          if (options.scope === "memories" && isSummary) continue;

          const sessionId = extractSessionId(r.memory.tags);
          const decay = calculateRelevanceScore(
            r.memory.importance,
            r.memory.accessedAt,
            now,
            decayParams,
          );
          candidates.push(buildResult({
            id: r.memory.id,
            content: r.memory.content,
            origin: isSummary ? "session-summary" : "memory",
            relevanceScore: r.combinedScore,  // SemanticSearch already semantic-weighted
            decayScore: decay,
            tags: r.memory.tags,
            createdAt: r.memory.createdAt,
            sessionId,
          }));
        }
      }

      // 2. Semantic path — session-summaries ONLY for scope='conversations'
      if (options.scope === "conversations") {
        const summaries = deps.memoryStore.findByTag("session-summary");
        for (const m of summaries) {
          // Naive "contains" match for conversations-only scope; SemanticSearch would be ideal
          // but requires the KNN path. For MVP simplicity: textual substring over findByTag results.
          // Alternative: run SemanticSearch and filter to tag. Chose direct filter for determinism.
          if (!m.content.toLowerCase().includes(options.query.toLowerCase())) continue;
          const decay = calculateRelevanceScore(m.importance, m.accessedAt, now, decayParams);
          candidates.push(buildResult({
            id: m.id,
            content: m.content,
            origin: "session-summary",
            relevanceScore: m.importance,  // no KNN score available here
            decayScore: decay,
            tags: m.tags,
            createdAt: m.createdAt,
            sessionId: extractSessionId(m.tags),
          }));
        }
      }

      // 3. FTS5 raw-turn path (for 'conversations' and 'all')
      if (options.scope === "conversations" || options.scope === "all") {
        const turnPage = deps.conversationStore.searchTurns(options.query, {
          limit: MAX_RESULTS_PER_PAGE * 3,
          offset: 0,
        });
        for (const t of turnPage.results) {
          const decay = calculateRelevanceScore(
            CONVERSATION_TURN_DEFAULT_IMPORTANCE,
            t.createdAt,
            now,
            decayParams,
          );
          const relevance = bm25ToRelevance(t.bm25Score);
          candidates.push(buildResult({
            id: t.turnId,
            content: t.content,
            origin: "conversation-turn",
            relevanceScore: relevance,
            decayScore: decay,
            tags: [],
            createdAt: t.createdAt,
            sessionId: t.sessionId,
          }));
        }
      }

      // 4. Dedup for scope='all' — prefer session-summary over raw-turn for same sessionId
      const deduped = options.scope === "all"
        ? dedupPreferSummary(candidates)
        : candidates;

      // 5. Sort by combinedScore descending
      deduped.sort((a, b) => b.combinedScore - a.combinedScore);

      const totalCandidates = deduped.length;
      const pageSlice = deduped.slice(offset, offset + clampedLimit);
      const hasMore = offset + pageSlice.length < totalCandidates;
      const nextOffset = hasMore ? offset + pageSlice.length : null;

      return Object.freeze({
        results: Object.freeze(pageSlice),
        hasMore,
        nextOffset,
        totalCandidates,
      });
    }

    /** Extract sessionId from a `session:<id>` tag if present. */
    function extractSessionId(tags: readonly string[]): string | null {
      for (const t of tags) {
        if (t.startsWith("session:")) return t.slice("session:".length);
      }
      return null;
    }

    /** Build a ScopedSearchResult combining relevance * 0.7 + decay * 0.3. */
    function buildResult(r: {
      id: string;
      content: string;
      origin: ScopedSearchResult["origin"];
      relevanceScore: number;
      decayScore: number;
      tags: readonly string[];
      createdAt: string;
      sessionId: string | null;
    }): ScopedSearchResult {
      const combinedScore = r.relevanceScore * SEMANTIC_WEIGHT + r.decayScore * DECAY_WEIGHT;
      return Object.freeze({
        id: r.id,
        content: r.content,
        snippet: makeSnippet(r.content),
        origin: r.origin,
        relevanceScore: r.relevanceScore,
        combinedScore,
        tags: Object.freeze([...r.tags]),
        createdAt: r.createdAt,
        sessionId: r.sessionId,
      });
    }

    /** Deduplicate: for any sessionId with a session-summary present, drop conversation-turn results for the same sessionId. */
    function dedupPreferSummary(results: ScopedSearchResult[]): ScopedSearchResult[] {
      const sessionsWithSummary = new Set<string>();
      for (const r of results) {
        if (r.origin === "session-summary" && r.sessionId) {
          sessionsWithSummary.add(r.sessionId);
        }
      }
      return results.filter((r) => {
        if (r.origin !== "conversation-turn") return true;
        if (!r.sessionId) return true;
        return !sessionsWithSummary.has(r.sessionId);
      });
    }
