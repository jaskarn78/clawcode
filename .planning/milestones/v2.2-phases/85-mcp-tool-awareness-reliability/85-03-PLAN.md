<objective>
Ship `/clawcode-tools` — a Discord slash command (and matching CLI subcommand) that surfaces the per-agent MCP server state populated by Plan 01's readiness gate + heartbeat reconnect. Operator sees at a glance: which MCPs are `ready` vs `degraded` vs `failed`, when each last succeeded, and how many failures have accumulated in the rolling window. Makes the phantom-error contradiction visible: if an agent claims "1Password isn't logged in", the operator runs `/clawcode-tools` and sees `1password: ready · last success 12s ago · 0 failures` — the agent is wrong, and the operator has proof.

Purpose: Close the observability loop. Plan 01 collected the state; Plan 02 put it in the prompt; Plan 03 puts it in the operator's hands via Discord UI. Without this, an operator debugging a phantom error has no way to compare what the agent believes to what's actually true — they'd have to SSH in and tail daemon logs.

Output: `/clawcode-tools` as a CONTROL_COMMANDS entry (daemon-routed, zero LLM cost per invocation), `EmbedBuilder`-rendered status view (UI-01 compliance), `clawcode tools` CLI parity subcommand, per-server fields with status emoji + last-success timestamp + failure count.
</objective>

    ```typescript
    private async handleToolsCommand(
      interaction: ChatInputCommandInteraction,
    ): Promise<void> {
      const explicitAgent = interaction.options.get("agent")?.value;
      const agentName = typeof explicitAgent === "string" && explicitAgent.length > 0
        ? explicitAgent
        : getAgentForChannel(this.routingTable, interaction.channelId);

      if (!agentName) {
        await interaction.reply({
          content: "This channel is not bound to an agent and no agent was provided.",
          ephemeral: true,
        });
        return;
      }

      await interaction.deferReply({ ephemeral: true });

      let response: {
        agent: string;
        servers: ReadonlyArray<{
          name: string;
          status: "ready" | "degraded" | "failed" | "reconnecting" | "unknown";
          lastSuccessAt: number | null;
          lastFailureAt: number | null;
          failureCount: number;
          optional: boolean;
          lastError: string | null;
        }>;
      };
      try {
        response = (await sendIpcRequest(SOCKET_PATH, "list-mcp-status", {
          agent: agentName,
        })) as typeof response;
      } catch (err) {
        const msg = err instanceof Error ? err.message : String(err);
        await interaction.editReply(`Failed to read MCP state: ${msg}`);
        return;
      }

      if (response.servers.length === 0) {
        await interaction.editReply(`No MCP servers configured for ${agentName}`);
        return;
      }

      const embed = new EmbedBuilder()
        .setTitle(`MCP Tools · ${agentName}`)
        .setColor(this.resolveEmbedColor(response.servers));

      for (const s of response.servers) {
        const emoji = STATUS_EMOJI[s.status] ?? "⚪";
        const optSuffix = s.optional && s.status !== "ready" ? " (optional)" : "";
        const lastSuccess = s.lastSuccessAt
          ? `${formatRelativeTime(Date.now() - s.lastSuccessAt)} ago`
          : "never";
        const errLine = s.lastError ? `\nerror: ${s.lastError}` : "";
        embed.addFields({
          name: `${emoji} ${s.name}${optSuffix}`,
          value: `status: ${s.status}\nlast success: ${lastSuccess}\nfailures: ${s.failureCount}${errLine}`,
          inline: false,
        });
      }

      await interaction.editReply({ embeds: [embed] });
    }
    ```

    Add the module-level helpers in the same file (top of file near other helpers, or near `DISCORD_MAX_LENGTH`):
    ```typescript
    import { EmbedBuilder } from "discord.js";

    const STATUS_EMOJI: Record<string, string> = {
      ready: "🟢",
      degraded: "🟡",
      failed: "🔴",
      reconnecting: "🟠",
      unknown: "⚪",
    };

    /**
     * Discord embed color driven by worst-state server in the set.
     * Green if all ready; yellow if any degraded; red if any failed.
     */
    function resolveEmbedColor(
      servers: ReadonlyArray<{ status: string }>,
    ): number {
      if (servers.some((s) => s.status === "failed")) return 0xEA4335;      // red
      if (servers.some((s) => s.status === "degraded")) return 0xFBBC05;    // yellow
      if (servers.some((s) => s.status === "reconnecting")) return 0xFB8C00; // orange
      return 0x34A853;                                                       // green
    }

    /** Relative time formatter: "3s", "12m", "4h", "2d". Keep short for embed fields. */
    function formatRelativeTime(deltaMs: number): string {
      const s = Math.floor(deltaMs / 1000);
      if (s < 60) return `${s}s`;
      const m = Math.floor(s / 60);
      if (m < 60) return `${m}m`;
      const h = Math.floor(m / 60);
      if (h < 24) return `${h}h`;
      const d = Math.floor(h / 24);
      return `${d}d`;
    }
    ```

    ```typescript
    import type { Command } from "commander";
    import { sendIpcRequest } from "../../ipc/client.js";
    import { SOCKET_PATH } from "../../manager/daemon.js";
    import { ManagerNotRunningError } from "../../shared/errors.js";
    import { cliLog, cliError } from "../output.js";

    export type ToolsServer = {
      readonly name: string;
      readonly status: "ready" | "degraded" | "failed" | "reconnecting" | "unknown";
      readonly lastSuccessAt: number | null;
      readonly lastFailureAt: number | null;
      readonly failureCount: number;
      readonly optional: boolean;
      readonly lastError: string | null;
    };

    export type ToolsResponse = {
      readonly agent: string;
      readonly servers: readonly ToolsServer[];
    };

    function formatRelative(ts: number | null, now: number = Date.now()): string {
      if (ts === null) return "never";
      const delta = Math.max(0, now - ts);
      const s = Math.floor(delta / 1000);
      if (s < 60) return `${s}s ago`;
      const m = Math.floor(s / 60);
      if (m < 60) return `${m}m ago`;
      const h = Math.floor(m / 60);
      if (h < 24) return `${h}h ago`;
      return `${Math.floor(h / 24)}d ago`;
    }

    export function formatToolsTable(resp: ToolsResponse, now: number = Date.now()): string {
      if (resp.servers.length === 0) {
        return `No MCP servers configured for ${resp.agent}`;
      }
      const rows = resp.servers.map((s) => ({
        agent: resp.agent,
        server: s.name + (s.optional ? " (opt)" : ""),
        status: s.status,
        lastSuccess: formatRelative(s.lastSuccessAt, now),
        failures: String(s.failureCount),
        lastError: s.lastError ?? "",
      }));
      const w = (col: keyof (typeof rows)[number], header: string) =>
        Math.max(header.length, ...rows.map((r) => String(r[col]).length));
      const widths = {
        agent: w("agent", "AGENT"),
        server: w("server", "SERVER"),
        status: w("status", "STATUS"),
        lastSuccess: w("lastSuccess", "LAST SUCCESS"),
        failures: w("failures", "FAILURES"),
        lastError: w("lastError", "LAST ERROR"),
      };
      const header = [
        "AGENT".padEnd(widths.agent),
        "SERVER".padEnd(widths.server),
        "STATUS".padEnd(widths.status),
        "LAST SUCCESS".padEnd(widths.lastSuccess),
        "FAILURES".padEnd(widths.failures),
        "LAST ERROR".padEnd(widths.lastError),
      ].join("  ");
      const separator = "-".repeat(Object.values(widths).reduce((a, b) => a + b, 0) + 10);
      const body = rows.map((r) => [
        r.agent.padEnd(widths.agent),
        r.server.padEnd(widths.server),
        r.status.padEnd(widths.status),
        r.lastSuccess.padEnd(widths.lastSuccess),
        r.failures.padEnd(widths.failures),
        r.lastError.padEnd(widths.lastError),
      ].join("  "));
      return [header, separator, ...body].join("\n");
    }

    export function registerToolsCommand(program: Command): void {
      program
        .command("tools")
        .description("Show per-agent MCP tool readiness (ready / degraded / failed)")
        .requiredOption("-a, --agent <name>", "Agent to query")
        .action(async (opts: { agent: string }) => {
          try {
            const result = (await sendIpcRequest(SOCKET_PATH, "list-mcp-status", {
              agent: opts.agent,
            })) as ToolsResponse;
            cliLog(formatToolsTable(result));
          } catch (error) {
            if (error instanceof ManagerNotRunningError) {
              cliError("Manager is not running. Start it with: clawcode start-all");
              process.exit(1);
              return;
            }
            const msg = error instanceof Error ? error.message : String(error);
            cliError(`Error: ${msg}`);
            process.exit(1);
          }
        });
    }
    ```
