<objective>
Extend the v1.6 warm-path readiness gate with a JSON-RPC `initialize` handshake against every configured MCP server at agent startup. The daemon refuses to flip an agent to `status: running` until every **mandatory** MCP server responds successfully (optional servers log + continue). Reuse the v1.3 heartbeat to auto-reconnect MCPs that drop mid-session, with state transitions `ready → degraded → failed → reconnecting → ready` that downstream slash commands + system prompt can read. Ensure JSON-RPC tool errors flow back to the agent verbatim (not wrapped).

Purpose: Kill the phantom-error class at its root. The user-reported symptom ("1Password isn't logged in when it is") happens because (a) no proactive health gate catches misconfigured MCPs at startup, (b) the heartbeat doesn't reconnect on flap, and (c) when a tool does fail the agent sees a generic failure and confabulates misconfiguration. This plan makes each failure path honest: refuse to start, detect + reconnect, and surface the actual RPC error. Plans 02 and 03 then put that state into the prompt + Discord.

Output: `performMcpReadinessHandshake` pure module; `mcp-reconnect` heartbeat check; `optional` field on mcpServerSchema; warm-path-check `mcpProbe` hook; `getMcpState` on persistent-session-handle; verbatim JSON-RPC error pass-through pinned by tests; three integration tests proving the three truths (mandatory refuses, optional permits, reconnect recovers).
</objective>

    **Step B — Create `src/mcp/readiness.ts` with the pure handshake function.**

    ```typescript
    import { checkMcpServerHealth, type McpHealthResult } from "./health.js";
    import type { McpServerSchemaConfig } from "../config/schema.js";

    export type McpServerState = {
      readonly name: string;
      readonly status: "ready" | "degraded" | "failed" | "reconnecting";
      readonly lastSuccessAt: number | null;       // epoch ms
      readonly lastFailureAt: number | null;
      readonly lastError: { code?: number; message: string } | null;
      readonly failureCount: number;                // rolling last-N, see MAX
      readonly optional: boolean;
    };

    export type McpReadinessReport = {
      readonly ready: boolean;                      // true iff every mandatory server.status === "ready"
      readonly stateByName: ReadonlyMap<string, McpServerState>;
      readonly errors: readonly string[];           // mandatory failures ONLY (gate-blocking)
      readonly optionalErrors: readonly string[];   // optional failures (advisory)
    };

    export const MCP_HANDSHAKE_TIMEOUT_MS = 5000;
    export const MCP_FAILURE_WINDOW = 20;          // last-N for /clawcode-tools count

    /**
     * TOOL-01 — Run JSON-RPC `initialize` against every configured MCP server
     * in parallel (Promise.all). Each result classifies as ready/failed based
     * on checkMcpServerHealth's healthy bool + error string. JSON-RPC errors
     * flow through verbatim (TOOL-04); we neither rewrite the message nor swap
     * it for a generic "tool unavailable" string.
     */
    export async function performMcpReadinessHandshake(
      servers: readonly McpServerSchemaConfig[],
      opts: { readonly timeoutMs?: number; readonly now?: () => number } = {},
    ): Promise<McpReadinessReport> {
      const timeoutMs = opts.timeoutMs ?? MCP_HANDSHAKE_TIMEOUT_MS;
      const now = opts.now ?? (() => Date.now());
      const stateByName = new Map<string, McpServerState>();
      const errors: string[] = [];
      const optionalErrors: string[] = [];

      if (servers.length === 0) {
        return Object.freeze({
          ready: true,
          stateByName,
          errors: Object.freeze([]) as readonly string[],
          optionalErrors: Object.freeze([]) as readonly string[],
        });
      }

      const results: McpHealthResult[] = await Promise.all(
        servers.map((s) => checkMcpServerHealth(s, timeoutMs)),
      );

      for (let i = 0; i < servers.length; i++) {
        const s = servers[i]!;
        const r = results[i]!;
        const ts = now();
        if (r.healthy) {
          stateByName.set(s.name, Object.freeze({
            name: s.name,
            status: "ready",
            lastSuccessAt: ts,
            lastFailureAt: null,
            lastError: null,
            failureCount: 0,
            optional: s.optional === true,
          }));
          continue;
        }
        const msg = r.error ?? "unknown error";
        const scoped = `mcp: ${s.name}: ${msg}`;
        stateByName.set(s.name, Object.freeze({
          name: s.name,
          status: "failed",
          lastSuccessAt: null,
          lastFailureAt: ts,
          // TOOL-04 — preserve the verbatim error. checkMcpServerHealth does
          // not carry a JSON-RPC error code (transport-level failure), so
          // `code` stays undefined. When we wire tool-call errors in a later
          // task, the code field WILL be populated from the RPC envelope.
          lastError: { message: msg },
          failureCount: 1,
          optional: s.optional === true,
        }));
        if (s.optional === true) {
          optionalErrors.push(scoped);
        } else {
          errors.push(scoped);
        }
      }

      return Object.freeze({
        ready: errors.length === 0,
        stateByName,
        errors: Object.freeze([...errors]) as readonly string[],
        optionalErrors: Object.freeze([...optionalErrors]) as readonly string[],
      });
    }
    ```

    **Step B — Create the heartbeat check `src/heartbeat/checks/mcp-reconnect.ts`.**

    ```typescript
    import type { CheckModule, CheckContext, CheckResult } from "../types.js";
    import {
      performMcpReadinessHandshake,
      type McpServerState,
    } from "../../mcp/readiness.js";

    const MAX_RECONNECTS_PER_CYCLE = 3;
    const BACKOFF_RESET_MS = 5 * 60_000; // 5min

    const mcpReconnectCheck: CheckModule = {
      name: "mcp-reconnect",
      interval: 60,     // seconds — matches v1.3 default; per-agent override honored
      timeout: 20,      // seconds — caps total probe duration per tick

      async execute(ctx: CheckContext): Promise<CheckResult> {
        const agentConfig = ctx.sessionManager.getAgentConfig(ctx.agentName);
        const servers = agentConfig?.mcpServers ?? [];
        if (servers.length === 0) {
          return { status: "ok", message: "no MCP servers configured" };
        }

        const priorState = ctx.sessionManager.getMcpStateForAgent(ctx.agentName);
        const rep = await performMcpReadinessHandshake(servers);
        const merged = new Map<string, McpServerState>();

        let ready = 0, degraded = 0, failed = 0;
        for (const [name, fresh] of rep.stateByName) {
          const prior = priorState.get(name);
          let next: McpServerState = fresh;

          if (fresh.status === "ready") {
            // Recovered OR steady. Reset failureCount. Preserve optional flag.
            next = Object.freeze({ ...fresh, failureCount: 0 });
            ready++;
          } else if (fresh.status === "failed") {
            const priorFailures = prior?.failureCount ?? 0;
            const sinceLastSuccess = prior?.lastSuccessAt
              ? Date.now() - prior.lastSuccessAt
              : Number.POSITIVE_INFINITY;
            const resetWindow = sinceLastSuccess > BACKOFF_RESET_MS;
            const newCount = resetWindow ? 1 : priorFailures + 1;
            // First-flap from ready → degraded; subsequent failures → failed.
            const status = prior?.status === "ready" ? "degraded" : "failed";
            if (status === "degraded") degraded++; else failed++;
            next = Object.freeze({
              ...fresh,
              status,
              failureCount: newCount,
              // Preserve the lastSuccessAt from prior so backoff logic can see
              // how long this server has been down.
              lastSuccessAt: prior?.lastSuccessAt ?? null,
            });
          }
          merged.set(name, next);
        }

        // Persist merged state.
        ctx.sessionManager.setMcpStateForAgent(ctx.agentName, merged);
        // Also update the handle so TurnDispatcher-scope reads see fresh data.
        const handle = (ctx.sessionManager as unknown as {
          sessions: Map<string, { setMcpState?: (m: ReadonlyMap<string, McpServerState>) => void }>
        }).sessions.get(ctx.agentName);
        handle?.setMcpState?.(merged);

        const totalFailures = degraded + failed;
        if (failed > 0) {
          return {
            status: "critical",
            message: `${ready} ready, ${degraded} degraded, ${failed} failed`,
            metadata: { ready, degraded, failed },
          };
        }
        if (degraded > 0) {
          return {
            status: "warn",
            message: `${ready} ready, ${degraded} degraded, ${failed} failed`,
            metadata: { ready, degraded, failed },
          };
        }
        return {
          status: "ok",
          message: `${ready} ready, 0 degraded, 0 failed`,
          metadata: { ready, degraded: 0, failed: 0 },
        };
      },
    };

    export default mcpReconnectCheck;
    ```
