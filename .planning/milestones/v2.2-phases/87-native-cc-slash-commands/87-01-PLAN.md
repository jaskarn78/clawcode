       ```ts
       import type { SlashCommand } from "./sdk-types.js";
       import type { SlashCommandDef } from "../discord/slash-types.js";

       /**
        * Phase 87 CMD-01 — native-behavior classifier.
        *
        * Decides how a native CC command dispatches:
        *   - "control-plane" → SDK Query.setX() (model, permissions, effort)
        *   - "prompt-channel" → TurnDispatcher prompt input (compact, context, cost, help, hooks, ...)
        *   - "skip" → do NOT register (clear/export not SDK-dispatchable; mcp covered by /clawcode-tools)
        *
        * Safe default for unknown future commands is "prompt-channel" — this
        * matches the CMD-00 spike's conclusion that most non-setter commands
        * are prompt-routable via SDKLocalCommandOutputMessage.
        */
       const CONTROL_PLANE: ReadonlySet<string> = new Set(["model", "permissions", "effort"]);
       const SKIP: ReadonlySet<string> = new Set(["clear", "export", "mcp"]);

       export function classifyCommand(name: string): "control-plane" | "prompt-channel" | "skip" {
         if (CONTROL_PLANE.has(name)) return "control-plane";
         if (SKIP.has(name)) return "skip";
         return "prompt-channel";
       }

       export type CommandAcl = { readonly denied: ReadonlySet<string> };

       /** Build Discord SlashCommandDef[] from SDK-reported SlashCommand[]. */
       export function buildNativeCommandDefs(
         sdkCommands: readonly SlashCommand[],
         acl: CommandAcl,
       ): readonly SlashCommandDef[] {
         const out: SlashCommandDef[] = [];
         for (const cmd of sdkCommands) {
           if (acl.denied.has(cmd.name)) continue;
           const behavior = classifyCommand(cmd.name);
           if (behavior === "skip") continue;
           out.push({
             name: `clawcode-${cmd.name}`,
             description: cmd.description.slice(0, 100) || `Native /${cmd.name}`,
             claudeCommand: "",  // populated by Plan 03 dispatch site
             options: cmd.argumentHint.length > 0
               ? [{ name: "args", type: 3, description: cmd.argumentHint, required: false }]
               : [],
             nativeBehavior: behavior,
           });
         }
         return out;
       }

       /** Merge existing (control + default) with native; native wins on name collision. */
       export function mergeAndDedupe(
         existing: readonly SlashCommandDef[],
         native: readonly SlashCommandDef[],
       ): readonly SlashCommandDef[] {
         const byName = new Map<string, SlashCommandDef>();
         for (const cmd of existing) byName.set(cmd.name, cmd);
         for (const cmd of native) byName.set(cmd.name, cmd);  // native wins
         return [...byName.values()];
       }
       ```

       ```ts
        * Expected format:
        *   ## Command ACLs
        *   - deny: [init, security-review, batch]
        *
        * Missing file → "allow" (permissive default)
        * Missing `## Command ACLs` section → "allow"
        * `deny:` list matching commandName (exact, no leading slash) → "deny"
        * Otherwise → "allow"
        */
       export async function resolveCommandAcl(
         securityMdPath: string,
         commandName: string,
       ): Promise<"allow" | "deny"> {
         let content: string;
         try {
           content = await readFile(securityMdPath, "utf-8");
         } catch { return "allow"; }
         const sectionMatch = content.match(
           /##\s*Command\s+ACLs\s*\n([\s\S]*?)(?=\n##\s|\n#\s|$)/,
         );
         if (!sectionMatch) return "allow";
         const denyMatch = sectionMatch[1].match(/^\s*-\s*deny:\s*\[([^\]]*)\]/m);

         const denied = denyMatch[1]

       }
       ```
