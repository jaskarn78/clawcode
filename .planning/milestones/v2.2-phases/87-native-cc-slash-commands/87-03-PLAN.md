       /**
        * Phase 87 CMD-03 — canonical prompt-string builder for prompt-channel
        * native-CC dispatch.
        *
        * The SDK's Options docstring confirms slash commands are processed when
        * sent as prompt input (sdk.d.ts). The string format MUST be the bare
        * `/<name>` — NOT the clawcode-<name> Discord prefix (the SDK has no
        * knowledge of the clawcode namespace; it parses the literal /<name>
        * against its own slash-command dispatcher).
        *
        * Empty / missing args → no trailing space. Present args → single space
        * separator, passed through verbatim (no escaping — SDK parses as-is).
        */
       export function buildNativePromptString(
         commandName: string,
         args: string | undefined,
       ): string {
         const bare = commandName.replace(/^clawcode-/, "");
         const trimmed = (args ?? "").trim();
         return trimmed.length > 0 ? `/${bare} ${trimmed}` : `/${bare}`;
       }
       ```

       /**
        * Phase 87 CMD-03 — lookup a registered native-CC command by name and
        * return it iff it has nativeBehavior="prompt-channel".
        *
        * Implementation reads from the same resolved set register() builds for
        * the guild — delegates to resolveAgentCommands(agentConfig.slashCommands)
        * merged with mergeAndDedupe(agentCommands, nativeDefs) from Plan 01.
        * For simplicity we expose a resolvedCommandDefsForChannel helper that
        * Plan 01 wires or we inline the lookup against the agent's commands.
        */
       private findNativePromptChannelCommand(name: string): SlashCommandDef | null {
         // Walk resolvedAgents → their merged commands → match by name.
         // Prompt-channel entries are the same for every agent registering them
         // (pure name match; the dispatch target is the channel's bound agent).
         for (const agent of this.resolvedAgents) {
           for (const cmd of resolveAgentCommands(agent.slashCommands)) {
             if (cmd.name === name && cmd.nativeBehavior === "prompt-channel") {
               return cmd;
             }
           }
         }
         return null;

       /**
        * Phase 87 CMD-03 / CMD-06 — dispatch a prompt-channel native-CC command.
        *
        * Flow (mirrors the existing agent-routed streaming flow at lines 380-560
        * but with a canonical `/<name> <args>` prompt instead of the agent's
        * claudeCommand template):
        *   1. Resolve agent by channel; reject if unbound.
        *   2. deferReply + "Thinking..." feedback.
        *   3. Build prompt string via buildNativePromptString.
        *   4. Stream via TurnDispatcher.dispatchStream with ProgressiveMessageEditor.
        *   5. On SDK error, surface the actual error text in the ephemeral reply.
        */
       private async dispatchNativePromptCommand(
         interaction: ChatInputCommandInteraction,
         cmd: SlashCommandDef,
       ): Promise<void> {
         const agentName = getAgentForChannel(this.routingTable, interaction.channelId);
         if (!agentName) {
           try {
             await interaction.reply({
               content: "This channel is not bound to an agent.",
               ephemeral: true,
             });
           } catch { /* expired */ }
           return;
         }
         try {
           await interaction.deferReply();
         } catch { return; }
         const argsRaw = interaction.options.get("args")?.value;
         const args = typeof argsRaw === "string" ? argsRaw : undefined;
         const prompt = buildNativePromptString(cmd.name, args);

         try { await interaction.editReply("Thinking..."); } catch { /* noop */ }

         const editor = new ProgressiveMessageEditor({
           editFn: async (content: string) => {
             const truncated = content.length > DISCORD_MAX_LENGTH
               ? content.slice(0, DISCORD_MAX_LENGTH - 3) + "..."
               : content;
             await interaction.editReply(truncated);
           },
           editIntervalMs: 1500,
         });

         try {
           if (!this.turnDispatcher) {
             throw new Error("Turn dispatcher not wired — cannot dispatch native-CC prompt command");
           }
           const origin = makeRootOrigin("discord", interaction.channelId);
           const response = await this.turnDispatcher.dispatchStream(
             origin, agentName, prompt,
             (accumulated) => editor.update(accumulated),
           );
           await editor.flush();
           const text = response.trim();
           if (text.length === 0) {
             await interaction.editReply("(No response from agent)");
             return;
           }
           const truncated = text.length > DISCORD_MAX_LENGTH
             ? text.slice(0, DISCORD_MAX_LENGTH - 3) + "..."
             : text;
           await interaction.editReply(truncated);
         } catch (err) {
           editor.dispose();
           const msg = err instanceof Error ? err.message : String(err);
           this.log.error(
             { agent: agentName, command: cmd.name, prompt, error: msg },
             "native-CC prompt command failed",
           );
           // CMD-03 truth #4 — surface the ACTUAL error text. Phase 85's
           // TOOL-04 pattern (verbatim error pass-through) applied to the
           // slash-command layer.
           try {
             await interaction.editReply(`Command failed: ${msg}`);
           } catch { /* expired */ }
         }
       }
       ```
