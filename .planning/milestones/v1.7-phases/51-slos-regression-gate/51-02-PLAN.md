       ```typescript
       import { readFileSync } from "node:fs";
       import { parse as parseYaml } from "yaml";
       import { z } from "zod/v4";
       import { BenchmarkConfigError } from "./types.js";

       const promptDefinitionSchema = z.object({
         id: z.string().min(1),
         prompt: z.string().min(1),
         description: z.string().optional(),
       });

       const promptsFileSchema = z.object({
         prompts: z.array(promptDefinitionSchema).min(1),
       });

       export type PromptDefinition = z.infer<typeof promptDefinitionSchema>;

       /**
        * Load and validate the bench prompt set. Throws BenchmarkConfigError
        * on any read / parse / schema failure. Returns frozen array.
        */
       export function loadPrompts(path: string): readonly PromptDefinition[] {
         let raw: string;
         try { raw = readFileSync(path, "utf-8"); }
         catch (err) {
           const msg = err instanceof Error ? err.message : "unknown";
           throw new BenchmarkConfigError(`read failed: ${msg}`, path);
         }
         let parsed: unknown;
         try { parsed = parseYaml(raw); }
         catch (err) {
           const msg = err instanceof Error ? err.message : "unknown";
           throw new BenchmarkConfigError(`yaml parse failed: ${msg}`, path);
         }
         const result = promptsFileSchema.safeParse(parsed ?? {});
         if (!result.success) {
           throw new BenchmarkConfigError(
             `schema invalid: ${result.error.issues.map((i) => i.message).join("; ")}`,
             path,
           );
         }
         return Object.freeze(result.data.prompts.map((p) => Object.freeze(p)));
       }
       ```

       ```typescript
         readonly spawner?: (cmd: string, args: readonly string[], env: NodeJS.ProcessEnv) => ChildProcess | { pid: number; kill: (sig: NodeJS.Signals) => void };
       };

       /**
        * Spawn an isolated daemon in a tempdir-rooted HOME so MANAGER_DIR
        * resolves to <tmpHome>/.clawcode/manager. Returns a handle with the
        * resulting socket path and a stop() that SIGTERMs the daemon and
        * unlinks the socket.
        */
       export async function spawnIsolatedDaemon(opts: SpawnOpts): Promise<DaemonHandle> {
         await mkdir(join(opts.tmpHome, ".clawcode", "manager"), { recursive: true });
         const entryScript = resolve(process.cwd(), "src/manager/daemon-entry.ts");
         const env: NodeJS.ProcessEnv = {
           ...process.env,
           HOME: opts.tmpHome,
         };
         delete env.ANTHROPIC_API_KEY; // Phase 50 daemon spawn pattern
         const spawner = opts.spawner ?? ((cmd, args, e) => spawn(cmd, [...args], { detached: true, stdio: "ignore", env: e, cwd: process.cwd() }));
         const child = spawner("npx", ["tsx", entryScript, "--config", opts.configPath], env);
         const pid = child.pid;
         if (typeof pid !== "number") throw new Error("spawnIsolatedDaemon: child has no pid");
         const socketPath = join(opts.tmpHome, ".clawcode", "manager", "clawcode.sock");
         return Object.freeze({
           pid,
           socketPath,
           stop: async () => {
             try { child.kill("SIGTERM"); } catch { /* already dead */ }
             try { await access(socketPath); await unlink(socketPath); } catch { /* socket already gone */ }
           },
         });
       }

       /**
        * Poll the daemon's `status` IPC method until it responds OR maxAttempts
        * exhausted. Returns true on success, false on timeout.
        */
       export async function awaitDaemonReady(
         socketPath: string,
         opts: { maxAttempts?: number; delayMs?: number; ipcClient?: typeof sendIpcRequest } = {},
       ): Promise<boolean> {
         const maxAttempts = opts.maxAttempts ?? 30;
         const delayMs = opts.delayMs ?? 500;
         const client = opts.ipcClient ?? sendIpcRequest;
         for (let i = 0; i < maxAttempts; i++) {
           await new Promise((r) => setTimeout(r, delayMs));
           try {
             await client(socketPath, "status", {});
             return true;
           } catch { /* not ready yet */ }
         }
         return false;
       }

       /**
        * Write a minimal clawcode.yaml that defines a single bench-agent.
        * No Discord, no MCP, no scheduler entries — fastest possible boot.
        */
       export async function writeBenchAgentConfig(
         tmpHome: string,
         opts: { agentName: string; model?: "haiku" | "sonnet" | "opus" },
       ): Promise<string> {
         const cfg = {
           version: 1,
           defaults: { model: opts.model ?? "haiku" },
           agents: [{
             name: opts.agentName,
             channels: [] as string[],
             model: opts.model ?? "haiku",
             heartbeat: false,
             workspace: join(tmpHome, ".clawcode-bench-workspace", opts.agentName),
           }],
         };
         const path = join(tmpHome, "clawcode-bench.yaml");
         await writeFile(path, stringifyYaml(cfg), "utf-8");
         return path;
       }
       ```

       ```typescript
       import { readFileSync, writeFileSync, mkdirSync } from "node:fs";
       import { dirname } from "node:path";
       import { baselineSchema, type Baseline, type BenchReport, BenchmarkConfigError, type PromptResult } from "./types.js";
       import { CANONICAL_SEGMENTS } from "../performance/types.js";

       /** Read and validate baseline.json. Throws BenchmarkConfigError on any failure. */
       export function readBaseline(path: string): Baseline {
         let raw: string;
         try { raw = readFileSync(path, "utf-8"); }
         catch (err) {
           const msg = err instanceof Error ? err.message : "unknown";
           throw new BenchmarkConfigError(`read failed: ${msg}`, path);
         }
         let parsed: unknown;
         try { parsed = JSON.parse(raw); }
         catch (err) {
           const msg = err instanceof Error ? err.message : "unknown";
           throw new BenchmarkConfigError(`json parse failed: ${msg}`, path);
         }
         const result = baselineSchema.safeParse(parsed);
         if (!result.success) {
           throw new BenchmarkConfigError(
             `schema invalid: ${result.error.issues.map((i) => i.message).join("; ")}`,
             path,
           );
         }
         return Object.freeze(result.data) as Baseline;
       }

       /**
        * Write baseline.json from a fresh BenchReport with provenance fields
        * (updated_at + updated_by stamped now). Creates parent dir if missing.
        */
       export function writeBaseline(
         path: string,
         report: BenchReport,
         provenance: { username: string; gitSha?: string },
       ): Baseline {
         const baseline: Baseline = Object.freeze({
           ...report,
           git_sha: provenance.gitSha ?? report.git_sha,
           updated_at: new Date().toISOString(),
           updated_by: provenance.username,
         });
         mkdirSync(dirname(path), { recursive: true });
         writeFileSync(path, JSON.stringify(baseline, null, 2) + "\n", "utf-8");
         return baseline;
       }

       /**
        * Format a side-by-side diff table for human review. Used by both
        * --update-baseline (print before confirm) and --check-regression
        * (print on failure for CI logs).
        */
       export function formatDiffTable(report: BenchReport, baseline: Baseline | null): string {
         const headers = ["Segment", "Baseline p95", "Current p95", "Delta", "Delta %"];
         const baselineBySeg = new Map(baseline?.overall_percentiles.map((r) => [r.segment, r]) ?? []);
         const dataRows = CANONICAL_SEGMENTS.map((seg) => {
           const cur = report.overall_percentiles.find((r) => r.segment === seg);
           const base = baselineBySeg.get(seg);
           if (!base) return [seg, "(no baseline yet)", `${cur?.p95 ?? "—"} ms`, "—", "—"];
           if (base.p95 === null || cur?.p95 === null || cur?.p95 === undefined) {
             return [seg, `${base.p95 ?? "—"} ms`, `${cur?.p95 ?? "—"} ms`, "—", "—"];
           }
           const deltaMs = cur.p95 - base.p95;
           const deltaPct = base.p95 === 0 ? 0 : (deltaMs / base.p95) * 100;
           return [
             seg,
             `${base.p95} ms`,
             `${cur.p95} ms`,
             `${deltaMs >= 0 ? "+" : ""}${deltaMs} ms`,
             `${deltaMs >= 0 ? "+" : ""}${deltaPct.toFixed(1)}%`,
           ];
         });
         const allRows = [headers, ...dataRows];
         const widths = headers.map((_, i) => Math.max(...allRows.map((r) => (r[i] ?? "").length)));
         const sep = widths.map((w) => "-".repeat(w)).join("  ");
         const fmt = allRows.map((r, i) => {
           const line = r.map((cell, c) => c === 0 ? cell.padEnd(widths[c]!) : cell.padStart(widths[c]!)).join("  ");
           return i === 0 ? `${line}\n${sep}` : line;
         });
         return fmt.join("\n");
       }
       ```

       ```typescript
       export type RunBenchOpts = {
         readonly promptsPath: string;
         readonly agent?: string;
         readonly repeats?: number;
         readonly since?: string;
         readonly reportsDir: string;
         /** DI hooks for tests. */
         readonly harness?: { spawn: typeof spawnIsolatedDaemon; awaitReady: typeof awaitDaemonReady; writeConfig: typeof writeBenchAgentConfig };
         readonly ipcClient?: typeof sendIpcRequest;
         readonly tmpHomeFactory?: () => string;
       };

       export type RunBenchResult = {
         readonly report: BenchReport;
         readonly reportPath: string;
       };

       /**
        * Orchestrate one bench run: spawn isolated daemon, run each prompt N
        * times, query latency percentiles, write the JSON report. Tears down
        * the daemon in finally{} so the operator never has to clean up.
        */
       export async function runBench(opts: RunBenchOpts): Promise<RunBenchResult> {
         const repeats = opts.repeats ?? 5;
         const since = opts.since ?? "1h";
         const agentName = opts.agent ?? "bench-agent";
         const prompts = loadPrompts(opts.promptsPath);

         const tmpHome = opts.tmpHomeFactory?.() ?? mkdtempSync(join(tmpdir(), "clawcode-bench-"));
         const harness = opts.harness ?? { spawn: spawnIsolatedDaemon, awaitReady: awaitDaemonReady, writeConfig: writeBenchAgentConfig };
         const client = opts.ipcClient ?? sendIpcRequest;

         const configPath = await harness.writeConfig(tmpHome, { agentName, model: "haiku" });
         const handle: DaemonHandle = await harness.spawn({ tmpHome, configPath });

         try {
           const ready = await harness.awaitReady(handle.socketPath);
           if (!ready) throw new Error(`bench daemon failed to become ready at ${handle.socketPath}`);
           // Best-effort start of the bench-agent (idempotent — daemon may have already started it on boot).
           try { await client(handle.socketPath, "start", { name: agentName }); } catch { /* may already be running */ }

           const promptResults: PromptResult[] = [];
           for (const prompt of prompts) {
             const turnIds: string[] = [];
             for (let i = 0; i < repeats; i++) {
               const res = await client(handle.socketPath, "bench-run-prompt", {
                 agent: agentName,
                 prompt: prompt.prompt,
                 turnIdPrefix: `bench:${prompt.id}:`,
               }) as { turnId: string; response: string };
               turnIds.push(res.turnId);
             }
             // After running all repeats for this prompt, capture the percentile snapshot for THIS prompt's window
             // We use 'since: 1h' (the bench window) for the per-prompt percentiles too — they fold into the overall.
             const latencyForPrompt = await client(handle.socketPath, "latency", {
               agent: agentName,
               since,
             }) as LatencyReport;
             promptResults.push({
               id: prompt.id,
               turnIds,
               percentiles: [...latencyForPrompt.segments],
             });
           }

           // Final overall_percentiles snapshot across all prompts in the window
           const overallLatency = await client(handle.socketPath, "latency", {
             agent: agentName,
             since,
           }) as LatencyReport;
           // Ensure all 4 canonical segments are present, even if count=0
           const segMap = new Map(overallLatency.segments.map((s) => [s.segment, s]));
           const overall_percentiles: PercentileRow[] = CANONICAL_SEGMENTS.map((seg) =>
             segMap.get(seg) ?? { segment: seg, p50: null, p95: null, p99: null, count: 0 }
           );

           let gitSha = "unknown";
           try { gitSha = execSync("git rev-parse HEAD", { encoding: "utf-8" }).trim(); }
           catch { /* not in a git checkout — leave 'unknown' */ }

           const report: BenchReport = Object.freeze({
             run_id: nanoid(12),
             started_at: new Date().toISOString(),
             git_sha: gitSha,
             node_version: process.version,
             prompt_results: Object.freeze(promptResults) as PromptResult[],
             overall_percentiles: Object.freeze(overall_percentiles) as PercentileRow[],
           });

           mkdirSync(opts.reportsDir, { recursive: true });
           const reportPath = join(opts.reportsDir, `${report.run_id}.json`);
           writeFileSync(reportPath, JSON.stringify(report, null, 2) + "\n", "utf-8");
           return Object.freeze({ report, reportPath });
         } finally {
           await handle.stop();
         }
       }
       ```
