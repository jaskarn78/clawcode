The daemon merges per-agent `perf.slos` overrides with DEFAULT_SLOS.          
          ```typescript
          /**
           * Augment a LatencyReport's segments with slo_status, slo_threshold_ms,
           * and slo_metric per row, using DEFAULT_SLOS merged with per-agent
           * perf.slos? overrides.
           *
           * The threshold + metric are emitted ALONGSIDE the status so the dashboard
           * can render the "SLO target" subtitle directly from the response —
           * preserving single source of truth end-to-end. A segment with no configured
           * SLO emits slo_threshold_ms: null and slo_metric: null; the dashboard
           * falls back to the no-data cell class and omits the subtitle.
           *
           * Pure; safe to call with undefined override array. Exported for unit testing.
           */
          export function augmentWithSloStatus(
            segments: readonly PercentileRow[],
            agentSlos: readonly SloEntry[] | undefined,
          ): readonly PercentileRow[] {
            const effectiveSlos = agentSlos && agentSlos.length > 0
              ? mergeSloOverrides(DEFAULT_SLOS, agentSlos)
              : DEFAULT_SLOS;
            const slosBySeg = new Map<string, SloEntry>();
            for (const s of effectiveSlos) {
              if (!slosBySeg.has(s.segment)) slosBySeg.set(s.segment, s);
            }
            return Object.freeze(segments.map((row) => {
              const slo = slosBySeg.get(row.segment);
              if (!slo) {
                // No SLO configured for this segment — still emit the fields so the
                // response shape is consistent, but with nulls so the dashboard omits subtitle.
                return Object.freeze({
                  ...row,
                  slo_threshold_ms: null,
                  slo_metric: null,
                });
              }
              return Object.freeze({
                ...row,
                slo_status: evaluateSloStatus(row, slo.thresholdMs, slo.metric),
                slo_threshold_ms: slo.thresholdMs,
                slo_metric: slo.metric,
              });
            }));
          }
          ```
