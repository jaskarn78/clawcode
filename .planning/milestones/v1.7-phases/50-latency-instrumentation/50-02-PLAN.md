**Rationale for caller-owned Turn (documented here so no future revision re-opens this):**
- DiscordBridge.handleMessage wants a `receive` span opened ASAP (before session dispatch). That requires the Turn to exist BEFORE streamFromAgent is called. Caller-owned Turn is the only contract that supports this.
- Scheduler has a similar need: it constructs `scheduler:<nanoid>` turnIds (50-02b).
- SessionManager CANNOT own the Turn — it has no concept of "receive time" or scheduler turnId semantics.
- Passing a `turnId: string` instead of a `Turn` forces SessionManager to become a Turn factory — duplicated responsibility and unable to express the pre-SessionManager work that bridge does.
- Caller-owned Turn is the minimal-surface contract: SessionManager is pure passthrough; session-adapter is pure span-emitter; bridge/scheduler are pure turn-lifecycle-owners.

4. Extract a shared instrumentation helper to avoid triplicating the span loop across `send`, `sendAndCollect`, `sendAndStream` (Pitfall 2 warns about divergence). Add a private function inside the adapter factory (around line 380, before sendAndStream):

```typescript
/**
 * Shared SDK stream iteration with tracing.
 * - Opens end_to_end and first_token spans on entry (if turn provided)
 * - Filters subagent messages via parent_tool_use_id (Pitfall 6)
 * - Starts tool_call.<name> spans on tool_use content blocks; ends on matching user tool_use_result
 * - Ends end_to_end and any remaining open spans on result or throw
 * - NOTE: does NOT call turn.end() — caller owns turn lifecycle
 *
 * @returns the accumulated text result from the assistant
 */
async function iterateWithTracing(
  q: ReturnType<typeof sdk.query>,
  turn: Turn | undefined,
  onAssistantText: ((accumulated: string) => void) | null,
): Promise<string> {
  const endToEnd = turn?.startSpan("end_to_end");
  const firstToken = turn?.startSpan("first_token");
  let firstTokenEnded = false;
  const activeTools = new Map<string, import("../performance/trace-collector.js").Span>();
  const textParts: string[] = [];

  const closeAllSpans = () => {
    for (const span of activeTools.values()) span.end();
    activeTools.clear();
    if (!firstTokenEnded) { firstToken?.end(); firstTokenEnded = true; }
    endToEnd?.end();
  };

  try {
    for await (const msg of q) {
      if (msg.type === "assistant") {
        const parentToolUseId = (msg as { parent_tool_use_id?: string | null }).parent_tool_use_id ?? null;
        if (parentToolUseId === null) {
          // Only the PARENT assistant message counts for first_token and tool_call starts (Pitfall 6)
          const contentBlocks = ((msg as { message?: { content?: unknown[] } }).message?.content ?? []) as unknown[];
          for (const raw of contentBlocks) {
            const block = raw as { type?: string; name?: string; id?: string };
            if (block.type === "text" && !firstTokenEnded) {
              firstToken?.end();
              firstTokenEnded = true;
            }
            if (block.type === "tool_use" && block.id && block.name) {
              const span = turn?.startSpan(`tool_call.${block.name}`, { tool_use_id: block.id });
              if (span) activeTools.set(block.id, span);
            }
          }
        }
        if (typeof msg.content === "string" && msg.content.length > 0) {
          textParts.push(msg.content);
          onAssistantText?.(textParts.join("\n"));
        }
      }

      if (msg.type === "user") {
        const toolUseId = (msg as { parent_tool_use_id?: string | null }).parent_tool_use_id;
        if (toolUseId) {
          const span = activeTools.get(toolUseId);
          if (span) {
            span.end();
            activeTools.delete(toolUseId);
          }
        }
      }

      if (msg.type === "result") {
        if (msg.session_id) sessionId = msg.session_id;
        extractUsage(msg, usageCallback);
        closeAllSpans();
        if ("result" in msg && typeof msg.result === "string" && msg.result.length > 0) {
          return msg.result;
        }
        if (msg.subtype !== "success") {
          if ("is_error" in msg && msg.is_error) {
            throw new Error(`Agent error: ${msg.subtype}`);
          }
        }
        return textParts.join("\n");
      }
    }
    closeAllSpans();
    return textParts.join("\n");
  } catch (err) {
    closeAllSpans();
    throw err;
  }
}
```
