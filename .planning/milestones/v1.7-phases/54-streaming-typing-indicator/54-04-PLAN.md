    In src/manager/daemon.ts near `augmentWithSloStatus` (around line 108), add a sibling helper:

    ```typescript
    /**
     * Phase 54 — evaluate the First Token headline object emitted on the
     * `latency` IPC response. Cold-start guard: if the window has fewer than
     * COLD_START_MIN_TURNS first_token spans, status becomes "no_data" (gray)
     * regardless of the measured percentile. Protects operators from seeing
     * red on a newly-started agent.
     */
    const COLD_START_MIN_TURNS = 5;

    export type FirstTokenHeadline = {
      readonly p50: number | null;
      readonly p95: number | null;
      readonly p99: number | null;
      readonly count: number;
      readonly slo_status: SloStatus;
      readonly slo_threshold_ms: number | null;
      readonly slo_metric: SloMetric | null;
    };

    export function evaluateFirstTokenHeadline(
      row: PercentileRow,
      agentSlos: readonly SloEntry[] | undefined,
    ): FirstTokenHeadline {
      const effectiveSlos =
        agentSlos && agentSlos.length > 0
          ? mergeSloOverrides(DEFAULT_SLOS, agentSlos)
          : DEFAULT_SLOS;
      const slo = effectiveSlos.find(
        (s) => s.segment === "first_token",
      );

      // Cold-start guard — preempts healthy/breach coloring.
      if (row.count < COLD_START_MIN_TURNS) {
        return Object.freeze({
          p50: row.p50,
          p95: row.p95,
          p99: row.p99,
          count: row.count,
          slo_status: "no_data",
          slo_threshold_ms: slo?.thresholdMs ?? null,
          slo_metric: slo?.metric ?? null,
        });
      }

      if (!slo) {
        return Object.freeze({
          p50: row.p50,
          p95: row.p95,
          p99: row.p99,
          count: row.count,
          slo_status: "no_data",
          slo_threshold_ms: null,
          slo_metric: null,
        });
      }

      return Object.freeze({
        p50: row.p50,
        p95: row.p95,
        p99: row.p99,
        count: row.count,
        slo_status: evaluateSloStatus(row, slo.thresholdMs, slo.metric),
        slo_threshold_ms: slo.thresholdMs,
        slo_metric: slo.metric,
      });
    }
    ```
