           if (config && !this.budget.canEscalate(agentName, this.config.escalationModel, config)) {
             const threshold = this.budget.checkAlerts(agentName, this.config.escalationModel, config);
             if (threshold && this.budget.shouldAlert(agentName, this.config.escalationModel, threshold)) {
               this.alertCallback?.(agentName, this.config.escalationModel, threshold);
