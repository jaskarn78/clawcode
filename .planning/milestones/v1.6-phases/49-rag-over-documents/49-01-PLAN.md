       ```typescript
       export type DocumentChunk = {
         readonly id: string;           // nanoid
         readonly source: string;       // file path or identifier
         readonly chunkIndex: number;   // 0-based position in document
         readonly content: string;      // chunk text
         readonly startChar: number;    // start character offset in original
         readonly endChar: number;      // end character offset in original
         readonly createdAt: string;    // ISO timestamp
       };

       export type IngestResult = {
         readonly source: string;
         readonly chunksCreated: number;
         readonly totalChars: number;
       };

       export type DocumentSearchResult = {
         readonly chunkId: string;
         readonly source: string;
         readonly chunkIndex: number;
         readonly content: string;
         readonly similarity: number;   // 0-1, higher is better
         readonly contextBefore: string | null;  // previous chunk content
         readonly contextAfter: string | null;   // next chunk content
       };
       ```

    3. Create `src/documents/chunker.ts`:
       - Export `chunkText(text: string, targetTokens = 500, overlapTokens = 50): ChunkInput[]`
       - Where `ChunkInput = { content: string; chunkIndex: number; startChar: number; endChar: number }`
       - Use word-count heuristic: 1 token ~ 0.75 words. So targetTokens=500 means ~375 words per chunk, overlapTokens=50 means ~37 words overlap.
       - Split text by whitespace into words. Walk forward in steps of (targetWords - overlapWords). For each step, join the words slice and record startChar/endChar by scanning the original text.
       - Return empty array for empty/whitespace-only text.
       - Export `async chunkPdf(buffer: Buffer, targetTokens = 500, overlapTokens = 50): Promise<ChunkInput[]>`
       - Uses `pdf-parse` to extract text from buffer, then calls `chunkText` on the extracted text.

       - `export class DocumentStore` — takes a better-sqlite3 Database instance (NOT a path; reuse MemoryStore's db).
       - Constructor calls `initSchema()` which creates:
         ```sql
         CREATE TABLE IF NOT EXISTS document_chunks (
           id TEXT PRIMARY KEY,
           source TEXT NOT NULL,
           chunk_index INTEGER NOT NULL,
           content TEXT NOT NULL,
           start_char INTEGER NOT NULL,
           end_char INTEGER NOT NULL,
           created_at TEXT NOT NULL
         );
         CREATE INDEX IF NOT EXISTS idx_doc_chunks_source ON document_chunks(source);
         CREATE INDEX IF NOT EXISTS idx_doc_chunks_source_index ON document_chunks(source, chunk_index);

         CREATE VIRTUAL TABLE IF NOT EXISTS vec_document_chunks USING vec0(
           chunk_id TEXT PRIMARY KEY,
           embedding float[384] distance_metric=cosine
         );
         ```
       - Prepare statements for: insertChunk, insertVec, deleteBySource, deleteVecBySource, getChunkCount, getAdjacentChunks.
       - `ingest(source: string, chunks: readonly ChunkInput[], embeddings: readonly Float32Array[]): IngestResult`
         - Transaction: delete existing chunks for source first (overwrite), then insert all new chunks + vectors.
         - Generate nanoid for each chunk.
         - After insert, check getChunkCount(). If > 10_000, log a warning using console.warn (no pino dependency in this module).
         - Return `{ source, chunksCreated: chunks.length, totalChars }`.
       - `search(embedding: Float32Array, limit = 5, source?: string): DocumentSearchResult[]`
         - Use KNN via vec0 MATCH (same pattern as SemanticSearch).
         - Fetch up to `limit` results. If `source` provided, filter with `WHERE d.source = ?` (use two prepared statements: one filtered, one unfiltered).
         - For each result, fetch chunk_index-1 and chunk_index+1 from same source for context.
         - Convert cosine distance to similarity: `1 - distance`.
         - Return frozen array of DocumentSearchResult.
         - Clamp limit to max 20.
       - `deleteDocument(source: string): number` — delete from document_chunks and vec_document_chunks WHERE source matches. Return number of chunks deleted.
       - `getChunkCount(): number` — SELECT COUNT(*) from document_chunks.
       - `listSources(): readonly string[]` — SELECT DISTINCT source from document_chunks ORDER BY source.
