    Step 4 — Add a finmentum-aware source resolver helper:

    ```typescript
    /**
     * Resolve the workspace-copy plan for a single agent.
     *
     * Finmentum family source layout (verified on-box 2026-04-20):
     *   - workspace-finmentum/        — primary, has SOUL/IDENTITY + memory/ + .learnings/ + archive/
     *   - workspace-finmentum-content-creator/ — dedicated, own SOUL/IDENTITY
     *   - workspace-fin-acquisition/  — only uploads/
     *   - workspace-fin-research/     — only uploads/
     *   - workspace-fin-playground/   — only uploads/
     *   - workspace-fin-tax/          — only uploads/
     *
     * Resolution rules:
     *   1. Non-finmentum agents: always full-workspace copy to targetBasePath.
     *   2. Finmentum agent whose source workspace has SOUL.md present:
     *      full-workspace copy to targetBasePath (shared basePath for
     *      finmentum family). Phase 78 already wrote per-agent soulFile/
     *      identityFile paths pointing into this shared tree.
     *   3. Finmentum agent whose source workspace has NO SOUL.md but has
     *      uploads/: copy uploads/ to <targetBasePath>/uploads/<agentId>/.
     *   4. First-wins: once any finmentum agent has done a full-workspace
     *      copy to the shared basePath, subsequent finmentum agents that
     *      would ALSO do a full copy (e.g. finmentum + finmentum-content-
     *      creator with distinct source workspaces) are ADDITIVE — each
     *      writes to its own per-agent soulFile/identityFile target path
     *      from Phase 78 config-mapper. The shared directory root is
     *      already populated by the first copy; the second copy lands
     *      differently-named files into the same tree (SOUL.md via
     *      finmentum, finmentum-content-creator.md via content-creator's
     *      soulFile Phase 78 target) — no overwrite conflict because
     *      soulFile paths are distinct per agent.
     *
     * Implementation detail: for finmentum-content-creator specifically,
     * Phase 78 config-mapper produces soulFile=<shared>/SOUL.md path
     * because it's in the isFinmentumFamily set — but content-creator has
     * its own on-disk SOUL.md with different content. To honor this
     * distinction, we do NOT try to merge: we simply let both be full-
     * workspace copies; the SECOND copy (content-creator → shared) will
     * overwrite the SOUL.md at the shared path. This matches 79-CONTEXT's
     * "5 agents share basePath but content-creator has its own content"
     * decision — last write wins at the shared SOUL.md, and that's the
     * content the content-creator sub-agent needs.
     *
     * If this collision becomes a concern, Phase 81 can introduce a
     * per-agent SOUL.<id>.md naming scheme; Phase 79 keeps behavior simple.
     */
    type WorkspaceCopyPlan =
      | { readonly mode: "full"; readonly source: string; readonly target: string }
      | { readonly mode: "uploads-only"; readonly source: string; readonly target: string }
      | { readonly mode: "skip-empty-source"; readonly reason: string };

    function resolveWorkspaceCopyPlan(
      sourceWorkspace: string,
      targetBasePath: string,
      agentId: string,
    ): WorkspaceCopyPlan {
      if (!existsSync(sourceWorkspace)) {
        return {
          mode: "skip-empty-source",
          reason: `source workspace not found: ${sourceWorkspace}`,
        };
      }
      const hasSoul = existsSync(join(sourceWorkspace, "SOUL.md"));
      if (hasSoul) {
        return {
          mode: "full",
          source: sourceWorkspace,
          target: targetBasePath,
        };
      }
      const uploadsSrc = join(sourceWorkspace, "uploads");
      if (existsSync(uploadsSrc)) {
        return {
          mode: "uploads-only",
          source: uploadsSrc,
          target: join(targetBasePath, "uploads", agentId),
        };
      }
      return {
        mode: "skip-empty-source",
        reason: `no SOUL.md and no uploads/ at ${sourceWorkspace}`,
      };
    }
    ```
