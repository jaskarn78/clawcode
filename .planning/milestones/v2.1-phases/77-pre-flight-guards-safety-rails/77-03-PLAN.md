    ```typescript
    /**
     * Phase 77 MIGR-07 runtime belt-and-suspenders: patches node:fs write APIs
     * to refuse any path resolving under `~/.openclaw/`. Install once at the
     * CLI apply entry point, uninstall on exit.
     *
     * Companion to src/migration/guards.ts::assertReadOnlySource (the static
     * helper). This module wires the helper into the node fs surface at
     * runtime so accidental new-code that forgets to call assertReadOnlySource
     * still cannot write under the source tree.
     *
     * Static guard: CI grep asserts no literal "~/.openclaw/" string in a
     * write-context call across src/migration/.
     *
     * DO NOT:
     *   - Patch node:fs read APIs — reads are fine (and necessary for the
     *     openclaw.json reader, sqlite ATTACH, etc.).
     *   - Leave the guard installed past the command boundary — other CLI
     *     subcommands (status, costs, etc.) may legitimately write anywhere.
     *   - Double-wrap — track `installed` state and short-circuit repeat installs.
     *   - Block appendFile to .planning/migration/ledger.jsonl — the ledger
     *     is NOT under ~/.openclaw/; assertReadOnlySource already permits it.
     */
    import * as fs from "node:fs";
    import * as fsp from "node:fs/promises";
    import { fileURLToPath } from "node:url";
    import { assertReadOnlySource } from "./guards.js";

    type FsPathLike = string | Buffer | URL | number;

    function extractPath(pathArg: FsPathLike): string | undefined {
      if (typeof pathArg === "number") return undefined;           // fd — skip
      if (Buffer.isBuffer(pathArg)) return pathArg.toString("utf8");
      if (pathArg instanceof URL) return fileURLToPath(pathArg);
      if (typeof pathArg === "string") return pathArg;
      return undefined;
    }

    type Originals = {
      writeFile: typeof fsp.writeFile;
      appendFile: typeof fsp.appendFile;
      mkdir: typeof fsp.mkdir;
      writeFileSync: typeof fs.writeFileSync;
      appendFileSync: typeof fs.appendFileSync;
      mkdirSync: typeof fs.mkdirSync;
    };

    let installed = false;
    let originals: Originals | undefined;

    export function installFsGuard(): void {
      if (installed) return;
      originals = {
        writeFile: fsp.writeFile,
        appendFile: fsp.appendFile,
        mkdir: fsp.mkdir,
        writeFileSync: fs.writeFileSync,
        appendFileSync: fs.appendFileSync,
        mkdirSync: fs.mkdirSync,
      };

      const wrap = <A extends unknown[], R>(
        orig: (p: FsPathLike, ...rest: A) => R,
      ) => (p: FsPathLike, ...rest: A): R => {
        const str = extractPath(p);
        if (str !== undefined) assertReadOnlySource(str);
        return orig(p, ...rest);
      };

      // NOTE: assigning to imported namespace members is allowed at runtime in
      // ESM via the internal shape; use Object.defineProperty to be explicit.
      Object.defineProperty(fsp, "writeFile", { value: wrap(originals.writeFile), configurable: true });
      Object.defineProperty(fsp, "appendFile", { value: wrap(originals.appendFile), configurable: true });
      Object.defineProperty(fsp, "mkdir", { value: wrap(originals.mkdir), configurable: true });
      Object.defineProperty(fs, "writeFileSync", { value: wrap(originals.writeFileSync), configurable: true });
      Object.defineProperty(fs, "appendFileSync", { value: wrap(originals.appendFileSync), configurable: true });
      Object.defineProperty(fs, "mkdirSync", { value: wrap(originals.mkdirSync), configurable: true });

      installed = true;
    }

    export function uninstallFsGuard(): void {
      if (!installed || !originals) return;
      Object.defineProperty(fsp, "writeFile", { value: originals.writeFile, configurable: true });
      Object.defineProperty(fsp, "appendFile", { value: originals.appendFile, configurable: true });
      Object.defineProperty(fsp, "mkdir", { value: originals.mkdir, configurable: true });
      Object.defineProperty(fs, "writeFileSync", { value: originals.writeFileSync, configurable: true });
      Object.defineProperty(fs, "appendFileSync", { value: originals.appendFileSync, configurable: true });
      Object.defineProperty(fs, "mkdirSync", { value: originals.mkdirSync, configurable: true });
      originals = undefined;
      installed = false;
    }
    ```

          ```typescript
          export async function runApplyAction(
            opts: { only?: string },
            deps: { execaRunner?: (cmd: string, args: string[]) => Promise<{ stdout: string; exitCode: number | null }> } = {},
          ): Promise<number> {
            const paths = resolvePaths();
            const inventory = await readOpenclawInventory(paths.openclawJson);
            const chunkCounts = await gatherChunkCounts(inventory, paths.openclawMemoryDir);
            const report = buildPlan({
              inventory,
              chunkCounts,

              targetFilter: opts.only,

              cliError(`Unknown OpenClaw agent: '${unknownFilter.agent}'. Available: ${available}`);
              return 1;
            }
            installFsGuard();

                inventory,
                report,
                existingConfigPath: paths.clawcodeConfigPath,
                ledgerPath: paths.ledgerPath,
                sourceHash: report.planHash,
                filter: opts.only,
                execaRunner: deps.execaRunner,

              // All 4 guards passed — but Phase 77 has no actual apply body.
              cliError(APPLY_NOT_IMPLEMENTED_MESSAGE);
              return 1;
            } finally {
              uninstallFsGuard();
            }
          }
          ```
