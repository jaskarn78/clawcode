    ```typescript
    import { createHash } from "node:crypto";
    import { join } from "node:path";
    import {
      FINMENTUM_FAMILY_IDS,
      isFinmentumFamily,
      type OpenclawSourceEntry,
      type OpenclawSourceInventory,
    } from "./openclaw-config-reader.js";
    import type { ChunkCountResult } from "./source-memory-reader.js";

    export const WARNING_KINDS = [
      "missing-discord-binding",
      "empty-source-memory",
      "source-db-no-chunks-table",
      "unknown-agent-filter",
    ] as const;
    export type WarningKind = typeof WARNING_KINDS[number];

    export type PlanWarning = {
      readonly kind: WarningKind;
      readonly agent: string;
      readonly detail?: string;
    };

    export type AgentPlan = {
      readonly sourceId: string;                  // openclaw id (= clawcode agent name)
      readonly sourceName: string;                // openclaw human name
      readonly sourceWorkspace: string;           // absolute path under ~/.openclaw/
      readonly sourceAgentDir: string;            // absolute path under ~/.openclaw/agents/
      readonly sourceModel: string;               // model.primary
      readonly memoryChunkCount: number;          // 0 if empty/missing
      readonly memoryStatus: "present" | "empty" | "missing" | "table-absent";
      readonly discordChannelId: string | undefined;
      readonly isFinmentumFamily: boolean;
      readonly targetBasePath: string;            // where clawcode will live
      readonly targetMemoryPath: string;          // memoryPath override (equals basePath for dedicated)
      readonly targetAgentName: string;           // slug used in clawcode.yaml
    };

    export type PlanReport = {
      readonly agents: readonly AgentPlan[];      // sorted by sourceId ASC
      readonly warnings: readonly PlanWarning[];  // sorted by (kind, agent) ASC
      readonly sourcePath: string;                // inventory.sourcePath
      readonly targetRoot: string;                // clawcodeAgentsRoot
      readonly generatedAt: string;               // ISO timestamp — EXCLUDED from planHash
      readonly planHash: string;                  // SHA256 hex of deterministic serialization (see below)
    };

    /** Target basePath resolver. Finmentum family collapses to shared dir. */
    export function getTargetBasePath(
      sourceId: string,
      clawcodeAgentsRoot: string,
    ): string {
      if (isFinmentumFamily(sourceId)) {
        return join(clawcodeAgentsRoot, "finmentum");
      }
      return join(clawcodeAgentsRoot, sourceId);
    }

    /** memoryPath resolver. Finmentum family gets <basePath>/memory/<id>; dedicated equals basePath. */
    export function getTargetMemoryPath(
      sourceId: string,
      clawcodeAgentsRoot: string,
    ): string {
      const basePath = getTargetBasePath(sourceId, clawcodeAgentsRoot);
      if (isFinmentumFamily(sourceId)) {
        return join(basePath, "memory", sourceId);
      }
      return basePath;
    }

    /** Pure hasher — canonical JSON serialization of a PlanReport MINUS generatedAt MINUS planHash. */
    export function computePlanHash(report: Omit<PlanReport, "planHash" | "generatedAt">): string {
      // Stable key order: sort object keys alphabetically at every nesting level.
      const canonical = canonicalize(report);
      return createHash("sha256").update(canonical, "utf8").digest("hex");
    }

    function canonicalize(v: unknown): string {
      if (v === null || typeof v !== "object") return JSON.stringify(v);
      if (Array.isArray(v)) return "[" + v.map(canonicalize).join(",") + "]";
      const obj = v as Record<string, unknown>;
      const keys = Object.keys(obj).sort();
      return (
        "{" +
        keys.map((k) => JSON.stringify(k) + ":" + canonicalize(obj[k])).join(",") +
        "}"
      );
    }

    export type BuildPlanArgs = {
      readonly inventory: OpenclawSourceInventory;
      readonly chunkCounts: ReadonlyMap<string, ChunkCountResult>;
      readonly clawcodeAgentsRoot: string;        // e.g. "/home/jjagpal/.clawcode/agents" or a test tmpdir
      readonly targetFilter?: string;             // optional --agent filter
      readonly now?: () => Date;                  // DI for tests (defaults to () => new Date())
    };

    export function buildPlan(args: BuildPlanArgs): PlanReport {
      const { inventory, chunkCounts, clawcodeAgentsRoot, targetFilter } = args;
      const now = args.now ?? (() => new Date());
      const warnings: PlanWarning[] = [];

      // Filter handling (Success criteria #4: --agent <name> exit 1 on unknown).
      let source = inventory.agents;
      if (targetFilter !== undefined) {
        const matched = source.filter((a) => a.id === targetFilter);
        if (matched.length === 0) {
          warnings.push({ kind: "unknown-agent-filter", agent: targetFilter });
          source = [];
        } else {
          source = matched;
        }
      }

      const agents: AgentPlan[] = source.map((entry): AgentPlan => {
        const chunkResult = chunkCounts.get(entry.id) ?? { count: 0, missing: true, tableAbsent: false };
        const memoryStatus: AgentPlan["memoryStatus"] = chunkResult.missing
          ? "missing"
          : chunkResult.tableAbsent
            ? "table-absent"
            : chunkResult.count === 0
              ? "empty"
              : "present";
        return {
          sourceId: entry.id,
          sourceName: entry.name,
          sourceWorkspace: entry.workspace,
          sourceAgentDir: entry.agentDir,
          sourceModel: entry.model.primary,
          memoryChunkCount: chunkResult.count,
          memoryStatus,
          discordChannelId: entry.discordChannelId,
          isFinmentumFamily: entry.isFinmentumFamily,
          targetBasePath: getTargetBasePath(entry.id, clawcodeAgentsRoot),
          targetMemoryPath: getTargetMemoryPath(entry.id, clawcodeAgentsRoot),
          targetAgentName: entry.id,
        };
      });

      // Emit warnings per plan, then sort for determinism.
      for (const a of agents) {
        if (a.discordChannelId === undefined) {
          warnings.push({ kind: "missing-discord-binding", agent: a.sourceId });
        }
        if (a.memoryStatus === "missing") {
          warnings.push({ kind: "empty-source-memory", agent: a.sourceId, detail: "sqlite file not found" });
        } else if (a.memoryStatus === "table-absent") {
          warnings.push({ kind: "source-db-no-chunks-table", agent: a.sourceId });
        } else if (a.memoryStatus === "empty") {
          warnings.push({ kind: "empty-source-memory", agent: a.sourceId, detail: "chunks table present but empty" });
        }
      }
      warnings.sort((a, b) => (a.kind + a.agent).localeCompare(b.kind + b.agent));

      // agents is already sorted-by-id because inventory.agents is sorted-by-id (Plan 01 guarantees).
      // But re-sort defensively to honor the hash-stability invariant independent of caller.
      const sortedAgents = [...agents].sort((a, b) => a.sourceId.localeCompare(b.sourceId));

      const base = {
        agents: sortedAgents,
        warnings,
        sourcePath: inventory.sourcePath,
        targetRoot: clawcodeAgentsRoot,
      };
      const planHash = computePlanHash(base);
      return {
        ...base,
        generatedAt: now().toISOString(),
        planHash,
      };
    }
    ```

    Rationale for determinism:
    - `canonicalize` produces JSON with sorted keys at every level — two logically equal reports hash identically regardless of insertion order quirks.
    - `generatedAt` is deliberately EXCLUDED from `planHash`: the hash represents the plan's semantic content, not when it was computed. Success criteria #2 ("same input → same hash") would otherwise be unreachable.
    - The `now` parameter is DI'd for test determinism; production default is `new Date()`.
    - Warnings are sorted by `(kind + agent)` so the order is stable across runs.
    - We defensively re-sort `agents` even though Plan 01 already sorts — this protects against a future refactor of Plan 01 that forgets the sort.
