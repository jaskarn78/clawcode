    ```typescript
    import Database from "better-sqlite3";
    import { existsSync, statSync } from "node:fs";
    import { join } from "node:path";

    export type ChunkCountResult = {
      readonly count: number;
      readonly missing: boolean;       // true if sqlite file absent
      readonly tableAbsent: boolean;   // true if sqlite present but no 'chunks' table
    };

    export function getMemorySqlitePath(agentId: string, memoryDir: string): string {
      return join(memoryDir, `${agentId}.sqlite`);
    }

    export function readChunkCount(sqlitePath: string): ChunkCountResult {
      if (!existsSync(sqlitePath)) {
        return { count: 0, missing: true, tableAbsent: false };
      }
      // Read-only open is mandatory (PITFALLS Pitfall 3 — live daemon may be writing).
      // `fileMustExist: true` would throw; we already checked existsSync.
      const db = new Database(sqlitePath, { readonly: true, fileMustExist: true });
      try {
        const tableRow = db
          .prepare(
            "SELECT name FROM sqlite_master WHERE type='table' AND name='chunks' LIMIT 1",
          )
          .get() as { name: string } | undefined;
        if (!tableRow) {
          return { count: 0, missing: false, tableAbsent: true };
        }
        const row = db
          .prepare("SELECT COUNT(*) AS n FROM chunks")
          .get() as { n: number };
        return { count: row.n ?? 0, missing: false, tableAbsent: false };
      } finally {
        db.close();
      }
    }
    ```

    Rationale:
    - `readonly: true` is the PITFALLS Pitfall 3 mitigation: even with the live OpenClaw daemon writing WAL, a readonly handle cannot corrupt. We do NOT open in `?mode=ro` URI form — better-sqlite3's `{ readonly: true }` option is documented and simpler.
    - `sqlite_master` table-existence check covers the 8-of-15-agents-have-empty-dbs case AND future-proofs against schema drift.
    - `try/finally` + `db.close()` is mandatory — better-sqlite3 holds a file handle until explicitly closed; leaking handles across 15 agents in `list` would hit the OS FD limit in CI.

    Test setup:
    - Test fixture creation: in `beforeEach`, create a tmp sqlite via `new Database(":memory:")` → backup to tmp path via `db.backup(tmpPath)` with a `CREATE TABLE chunks(id TEXT PRIMARY KEY, text TEXT)` seeded with 3 rows. Alternative: `new Database(tmpPath); db.exec("CREATE TABLE chunks ...")`. Use `fs.mkdtempSync` for a tmp dir and clean up in `afterEach`.
    - mtime test: `statSync(path).mtimeMs` before + after `readChunkCount`; assert strictly equal. If linux's mtime resolution is coarse, test passes trivially; the test is a regression guard against accidentally swapping to `readonly: false`.

    DO NOT:
    - Open with `readonly: false` — this would corrupt the source WAL if OpenClaw is running.
    - Use `db.pragma('query_only = ON')` — `{ readonly: true }` is the better-sqlite3 idiom and stronger.
    - Read any column other than COUNT(*) — this phase does not translate memory content; Phase 80 does.
    - Skip the table-existence check — 8 agents have empty or absent dbs; throwing on missing table would make `list` unusable.

    import { z } from "zod/v4";
    import { appendFile, mkdir, readFile } from "node:fs/promises";
    import { existsSync } from "node:fs";
    import { dirname } from "node:path";

    export const LEDGER_ACTIONS = ["plan", "apply", "verify", "rollback", "cutover"] as const;
    export type LedgerAction = typeof LEDGER_ACTIONS[number];

    export const LEDGER_STATUSES = [
      "pending",
      "migrated",
      "verified",
      "rolled-back",
      "re-planned",
    ] as const;
    export type LedgerStatus = typeof LEDGER_STATUSES[number];

    export const ledgerRowSchema = z.object({
      ts: z.string().refine(
        (v) => !Number.isNaN(Date.parse(v)) && v.includes("T"),
        "ts must be ISO 8601",
      ),
      action: z.enum(LEDGER_ACTIONS),
      agent: z.string().min(1),
      status: z.enum(LEDGER_STATUSES),
      source_hash: z.string().min(1),
      target_hash: z.string().optional(),
      notes: z.string().optional(),
    });
    export type LedgerRow = z.infer<typeof ledgerRowSchema>;

    export const DEFAULT_LEDGER_PATH = ".planning/migration/ledger.jsonl";

    export async function appendRow(ledgerPath: string, row: LedgerRow): Promise<void> {
      // Validate before write — appending a bad row and relying on readRows to catch it would corrupt the append-only invariant.
      const parsed = ledgerRowSchema.safeParse(row);
      if (!parsed.success) {
        const issues = parsed.error.issues
          .map((i) => `${i.path.join(".") || "(root)"}: ${i.message}`)
          .join("; ");
        throw new Error(`Invalid LedgerRow: ${issues}`);
      }
      await mkdir(dirname(ledgerPath), { recursive: true });
      await appendFile(ledgerPath, JSON.stringify(parsed.data) + "\n", "utf8");
    }

    export async function readRows(ledgerPath: string): Promise<readonly LedgerRow[]> {
      if (!existsSync(ledgerPath)) return [];
      const text = await readFile(ledgerPath, "utf8");
      const lines = text.split("\n");
      const rows: LedgerRow[] = [];
      for (let i = 0; i < lines.length; i++) {
        const line = lines[i]?.trim();
        if (!line) continue;
        let parsed: unknown;
        try {
          parsed = JSON.parse(line);
        } catch (err) {
          const msg = err instanceof Error ? err.message : String(err);
