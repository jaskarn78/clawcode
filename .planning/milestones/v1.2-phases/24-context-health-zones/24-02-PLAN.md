```typescript
export type ContextZone = "green" | "yellow" | "orange" | "red";
export type ZoneThresholds = { readonly yellow: number; readonly orange: number; readonly red: number };
export type ZoneTransition = { readonly from: ContextZone; readonly to: ContextZone; readonly fillPercentage: number; readonly timestamp: string };
export type SnapshotCallback = (agentName: string, zone: ContextZone, fillPercentage: number) => Promise<void>;
export const ZONE_SEVERITY: Record<ContextZone, number>;
export const DEFAULT_ZONE_THRESHOLDS: ZoneThresholds;
export function classifyZone(fillPercentage: number, thresholds: ZoneThresholds): ContextZone;
export class ContextZoneTracker {
  constructor(opts: { agentName: string; thresholds: ZoneThresholds; onSnapshot?: SnapshotCallback });
  update(fillPercentage: number): Promise<ZoneTransition | null>;
  get zone(): ContextZone;
  reset(): void;
}
```
